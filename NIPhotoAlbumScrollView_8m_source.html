64 - (void)willDisplayPage:(NIPhotoScrollView *)page {
65  // When we ask the data source for the image we expect the following to happen:
66  // 1) If the data source has any image at this index, it should return it and set the
67  // photoSize accordingly.
68  // 2) If the returned photo is not the highest quality available, the data source should
69  // start loading the high quality photo and set isLoading to YES.
70  // 3) If no photo was available, then the data source should start loading the photo
71  // at its highest available quality and nil should be returned. The loadingImage property
72  // will be displayed until the image is loaded. isLoading should be set to YES.
73  NIPhotoScrollViewPhotoSize photoSize = NIPhotoScrollViewPhotoSizeUnknown;
74  BOOL isLoading = NO;
75  CGSize originalPhotoDimensions = CGSizeZero;
76  UIImage* image = [self.dataSource photoAlbumScrollView: self
77  photoAtIndex: page.pageIndex
78  photoSize: &photoSize
79  isLoading: &isLoading
80  originalPhotoDimensions: &originalPhotoDimensions];
81 
82  page.photoDimensions = originalPhotoDimensions;
83  page.loading = isLoading;
84 
85  if (nil == image) {
86  page.zoomingIsEnabled = NO;
87  [page setImage:self.loadingImage photoSize:NIPhotoScrollViewPhotoSizeUnknown];
88 
89  } else {
90  BOOL updateImage = photoSize > page.photoSize;
91  if (updateImage) {
92  [page setImage:image photoSize:photoSize];
93  }
94 
95  // Configure this after the image is set otherwise if the page's image isn't there
96  // e.g. (after prepareForReuse), zooming will always be disabled
97  page.zoomingIsEnabled = ([self isZoomingEnabled]
98  && (NIPhotoScrollViewPhotoSizeOriginal == photoSize));
99 
100  if (updateImage && NIPhotoScrollViewPhotoSizeOriginal == photoSize) {
101  [self notifyDelegatePhotoDidLoadAtIndex:page.pageIndex];
102  }
103  }
104 }

129 - (UIView<NIPagingScrollViewPage> *)pagingScrollView:(NIPagingScrollView *)pagingScrollView
130  pageViewForIndex:(NSInteger)pageIndex {
131  UIView<NIPagingScrollViewPage>* pageView = nil;
132  NSString* reuseIdentifier = @"photo";
133  pageView = [pagingScrollView dequeueReusablePageWithIdentifier:reuseIdentifier];
134  if (nil == pageView) {
135  pageView = [[NIPhotoScrollView alloc] init];
136  pageView.reuseIdentifier = reuseIdentifier;
137  pageView.backgroundColor = self.photoViewBackgroundColor;
138  }
139 
140  NIPhotoScrollView* photoScrollView = (NIPhotoScrollView *)pageView;
141  photoScrollView.photoScrollViewDelegate = self;
142  photoScrollView.zoomingAboveOriginalSizeIsEnabled = [self isZoomingAboveOriginalSizeEnabled];
143 
144  return pageView;
145 }

147 - (void)didLoadPhoto: (UIImage *)image
148  atIndex: (NSInteger)pageIndex
149  photoSize: (NIPhotoScrollViewPhotoSize)photoSize {
150  // This modifies the UI and therefor MUST be executed on the main thread.
151  NIDASSERT([NSThread isMainThread]);
152 
153  for (NIPhotoScrollView* page in self.visiblePages) {
154  if (page.pageIndex == pageIndex) {
155 
156  // Only replace the photo if it's of a higher quality than one we're already showing.
157  if (photoSize > page.photoSize) {
158  page.loading = NO;
159  [page setImage:image photoSize:photoSize];
160 
161  page.zoomingIsEnabled = ([self isZoomingEnabled]
162  && (NIPhotoScrollViewPhotoSizeOriginal == photoSize));
163 
164  // Notify the delegate that the photo has been loaded.
165  if (NIPhotoScrollViewPhotoSizeOriginal == photoSize) {
166  [self notifyDelegatePhotoDidLoadAtIndex:pageIndex];
167  }
168  }
169  break;
170  }
171  }
172 }
