33 UIImage* NISnapshotOfViewWithTransparencyOption(UIView* view, BOOL transparency) {
34  // Passing 0 as the last argument ensures that the image context will match the current device's
35  // scaling mode.
36  UIGraphicsBeginImageContextWithOptions(view.bounds.size, !transparency, 0);
37 
38  CGContextRef cx = UIGraphicsGetCurrentContext();
39 
40  // Views that can scroll do so by modifying their bounds. We want to capture the part of the view
41  // that is currently in the frame, so we offset by the bounds of the view accordingly.
42  CGContextTranslateCTM(cx, -view.bounds.origin.x, -view.bounds.origin.y);
43 
44  [view.layer renderInContext:cx];
45 
46  UIImage* image = UIGraphicsGetImageFromCurrentImageContext();
47  UIGraphicsEndImageContext();
48 
49  return image;
50 }

104 - (void)willRotateToInterfaceOrientation:(UIInterfaceOrientation)toInterfaceOrientation duration:(NSTimeInterval)duration {
105  if (!self.isSupportedOS) {
106  return;
107  }
108 
109  UIView* containerView = [self.delegate containerViewForSnapshotRotation:self];
110  UIView* rotationView = [self.delegate rotatingViewForSnapshotRotation:self];
111 
112  // The container view must not be the same as the rotation view.
113  NIDASSERT(containerView != rotationView);
114  if (containerView == rotationView) {
115  return;
116  }
117 
118  self.frameBeforeRotation = rotationView.frame;
119  self.snapshotViewBeforeRotation = NISnapshotViewOfView(rotationView);
120  [containerView insertSubview:self.snapshotViewBeforeRotation aboveSubview:rotationView];
121 }
122 
123 - (void)willAnimateRotationToInterfaceOrientation:(UIInterfaceOrientation)toInterfaceOrientation duration:(NSTimeInterval)duration {
124  if (!self.isSupportedOS) {
125  return;
126  }
127 
128  UIView* containerView = [self.delegate containerViewForSnapshotRotation:self];
129  UIView* rotationView = [self.delegate rotatingViewForSnapshotRotation:self];
130 
131  // The container view must not be the same as the rotation view.
132  NIDASSERT(containerView != rotationView);
133  if (containerView == rotationView) {
134  return;
135  }
136 
137  self.frameAfterRotation = rotationView.frame;
138 
139  [UIView setAnimationsEnabled:NO];
140 
141  self.snapshotViewAfterRotation = NISnapshotViewOfView(rotationView);
142  // Set the new frame while maintaining the old frame's height.
143  self.snapshotViewAfterRotation.frame = CGRectMake(self.frameBeforeRotation.origin.x,
144  self.frameBeforeRotation.origin.y,
145  self.frameBeforeRotation.size.width,
146  self.snapshotViewAfterRotation.frame.size.height);
147 
148  UIImage* imageBeforeRotation = self.snapshotViewBeforeRotation.image;
149  UIImage* imageAfterRotation = self.snapshotViewAfterRotation.image;
150 
151  if ([self.delegate respondsToSelector:@selector(fixedInsetsForSnapshotRotation:)]) {
152  UIEdgeInsets fixedInsets = [self.delegate fixedInsetsForSnapshotRotation:self];
153 
154  imageBeforeRotation = [imageBeforeRotation resizableImageWithCapInsets:fixedInsets resizingMode:UIImageResizingModeStretch];
155  imageAfterRotation = [imageAfterRotation resizableImageWithCapInsets:fixedInsets resizingMode:UIImageResizingModeStretch];
156  }
157 
158  self.snapshotViewBeforeRotation.image = imageBeforeRotation;
159  self.snapshotViewAfterRotation.image = imageAfterRotation;
160 
161  [UIView setAnimationsEnabled:YES];
162 
163  if (imageAfterRotation.size.height < imageBeforeRotation.size.height) {
164  self.snapshotViewAfterRotation.alpha = 0;
165 
166  [containerView insertSubview:self.snapshotViewAfterRotation aboveSubview:self.snapshotViewBeforeRotation];
167 
168  self.snapshotViewAfterRotation.alpha = 1;
169 
170  } else {
171  [containerView insertSubview:self.snapshotViewAfterRotation belowSubview:self.snapshotViewBeforeRotation];
172  self.snapshotViewBeforeRotation.alpha = 0;
173  }
174 
175  self.snapshotViewAfterRotation.frame = self.frameAfterRotation;
176  self.snapshotViewBeforeRotation.frame = CGRectMake(self.frameAfterRotation.origin.x,
177  self.frameAfterRotation.origin.y,
178  self.frameAfterRotation.size.width,
179  self.snapshotViewBeforeRotation.frame.size.height);
180 
181  rotationView.hidden = YES;
182 }
183 
184 - (void)didRotateFromInterfaceOrientation:(UIInterfaceOrientation)fromInterfaceOrientation {
185  if (!self.isSupportedOS) {
186  return;
187  }
188 
189  UIView* containerView = [self.delegate containerViewForSnapshotRotation:self];
190  UIView* rotationView = [self.delegate rotatingViewForSnapshotRotation:self];
191 
192  // The container view must not be the same as the rotation view.
193  NIDASSERT(containerView != rotationView);
194  if (containerView == rotationView) {
195  return;
196  }
197 
198  [self.snapshotViewBeforeRotation removeFromSuperview];
199  [self.snapshotViewAfterRotation removeFromSuperview];
200  self.snapshotViewBeforeRotation = nil;
201  self.snapshotViewAfterRotation = nil;
202 
203  rotationView.hidden = NO;
204 }

269 - (UIEdgeInsets)fixedInsetsForSnapshotRotation:(NISnapshotRotation *)snapshotRotation {
270  UIEdgeInsets insets = UIEdgeInsetsZero;
271 
272  // Find the right edge of the content view in the coordinate space of the UITableView.
273  UIView* rotatingView = [self.forwardingDelegate rotatingViewForSnapshotRotation:snapshotRotation];
274  NIDASSERT([rotatingView isKindOfClass:[UITableView class]]);
275  if ([rotatingView isKindOfClass:[UITableView class]]) {
276  UITableView* tableView = (UITableView *)rotatingView;
277 
278  NSArray* visibleCells = tableView.visibleCells;
279  if (visibleCells.count > 0) {
280  UIView* contentView = [[visibleCells objectAtIndex:0] contentView];
281  CGFloat contentViewRightEdge = [tableView convertPoint:CGPointMake(contentView.bounds.size.width, 0) fromView:contentView].x;
282 
283  CGFloat fixedRightWidth = tableView.bounds.size.width - contentViewRightEdge;
284  CGFloat fixedLeftWidth = MIN(snapshotRotation.frameAfterRotation.size.width, snapshotRotation.frameBeforeRotation.size.width) - fixedRightWidth - 1;
285  insets = UIEdgeInsetsMake(0, fixedLeftWidth, 0, fixedRightWidth);
286  }
287  }
288  return insets;
289 }
