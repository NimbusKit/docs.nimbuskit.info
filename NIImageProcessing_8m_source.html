30 + (CGRect)sourceRectWithImageSize:(CGSize)imageSize
31  displaySize:(CGSize)displaySize
32  contentMode:(UIViewContentMode)contentMode {
33  if (UIViewContentModeScaleToFill == contentMode) {
34  // Scale to fill draws the original image by squashing it to fit the destination's
35  // aspect ratio, so the source and destination rects aren't modified.
36  return CGRectMake(0, 0, imageSize.width, imageSize.height);
37 
38  } else if (UIViewContentModeScaleAspectFit == contentMode) {
39  // Aspect fit grabs the entire original image and squashes it down to a frame that fits
40  // the destination and leaves the unfilled space transparent.
41  return CGRectMake(0, 0, imageSize.width, imageSize.height);
42 
43  } else if (UIViewContentModeScaleAspectFill == contentMode) {
44  // Aspect fill requires that we take the destination rectangle and "fit" it within the
45  // source rectangle; this gives us the area of the source image we'll crop out to draw into
46  // the destination image.
47  CGFloat scale = MIN(imageSize.width / displaySize.width,
48  imageSize.height / displaySize.height);
49  CGSize scaledDisplaySize = CGSizeMake(displaySize.width * scale, displaySize.height * scale);
50  return CGRectMake(floorf((imageSize.width - scaledDisplaySize.width) / 2),
51  floorf((imageSize.height - scaledDisplaySize.height) / 2),
52  scaledDisplaySize.width,
53  scaledDisplaySize.height);
54 
55  } else if (UIViewContentModeCenter == contentMode) {
56  // We need to cut out a hole the size of the display in the center of the source image.
57  return CGRectMake(floorf((imageSize.width - displaySize.width) / 2),
58  floorf((imageSize.width - displaySize.width) / 2),
59  displaySize.width, displaySize.height);
60 
61  } else if (UIViewContentModeTop == contentMode) {
62  // We need to cut out a hole the size of the display in the top center of the source image.
63  return CGRectMake(floorf((imageSize.width - displaySize.width) / 2),
64  0,
65  displaySize.width, displaySize.height);
66 
67  } else if (UIViewContentModeBottom == contentMode) {
68  // We need to cut out a hole the size of the display in the bottom center of the source image.
69  return CGRectMake(floorf((imageSize.width - displaySize.width) / 2),
70  imageSize.height - displaySize.height,
71  displaySize.width, displaySize.height);
72 
73  } else if (UIViewContentModeLeft == contentMode) {
74  // We need to cut out a hole the size of the display in the left center of the source image.
75  return CGRectMake(0,
76  floorf((imageSize.width - displaySize.width) / 2),
77  displaySize.width, displaySize.height);
78 
79  } else if (UIViewContentModeRight == contentMode) {
80  // We need to cut out a hole the size of the display in the right center of the source image.
81  return CGRectMake(imageSize.width - displaySize.width,
82  floorf((imageSize.width - displaySize.width) / 2),
83  displaySize.width, displaySize.height);
84 
85  } else if (UIViewContentModeTopLeft == contentMode) {
86  // We need to cut out a hole the size of the display in the top left of the source image.
87  return CGRectMake(0,
88  0,
89  displaySize.width, displaySize.height);
90 
91  } else if (UIViewContentModeTopRight == contentMode) {
92  // We need to cut out a hole the size of the display in the top right of the source image.
93  return CGRectMake(imageSize.width - displaySize.width,
94  0,
95  displaySize.width, displaySize.height);
96 
97  } else if (UIViewContentModeBottomLeft == contentMode) {
98  // We need to cut out a hole the size of the display in the bottom left of the source image.
99  return CGRectMake(0,
100  imageSize.height - displaySize.height,
101  displaySize.width, displaySize.height);
102 
103  } else if (UIViewContentModeBottomRight == contentMode) {
104  // We need to cut out a hole the size of the display in the bottom right of the source image.
105  return CGRectMake(imageSize.width - displaySize.width,
106  imageSize.height - displaySize.height,
107  displaySize.width, displaySize.height);
108 
109  } else {
110  // Not implemented
111  NIDERROR(@"The following content mode has not been implemented: %d", contentMode);
112  return CGRectMake(0, 0, imageSize.width, imageSize.height);
113  }
114 }

120 + (CGRect)destinationRectWithImageSize:(CGSize)imageSize
121  displaySize:(CGSize)displaySize
122  contentMode:(UIViewContentMode)contentMode {
123  if (UIViewContentModeScaleAspectFit == contentMode) {
124  // Fit the image right in the center of the source frame and maintain the aspect ratio.
125  CGFloat scale = MIN(displaySize.width / imageSize.width,
126  displaySize.height / imageSize.height);
127  CGSize scaledImageSize = CGSizeMake(imageSize.width * scale, imageSize.height * scale);
128  return CGRectMake(floorf((displaySize.width - scaledImageSize.width) / 2),
129  floorf((displaySize.height - scaledImageSize.height) / 2),
130  scaledImageSize.width,
131  scaledImageSize.height);
132 
133  } else if (UIViewContentModeScaleToFill == contentMode
134  || UIViewContentModeScaleAspectFill == contentMode
135  || UIViewContentModeCenter == contentMode
136  || UIViewContentModeTop == contentMode
137  || UIViewContentModeBottom == contentMode
138  || UIViewContentModeLeft == contentMode
139  || UIViewContentModeRight == contentMode
140  || UIViewContentModeTopLeft == contentMode
141  || UIViewContentModeTopRight == contentMode
142  || UIViewContentModeBottomLeft == contentMode
143  || UIViewContentModeBottomRight == contentMode) {
144  // We're filling the entire destination, so the destination rect is the display rect.
145  return CGRectMake(0, 0, displaySize.width, displaySize.height);
146 
147  } else {
148  // Not implemented
149  NIDERROR(@"The following content mode has not been implemented: %d", contentMode);
150  return CGRectMake(0, 0, displaySize.width, displaySize.height);
151  }
152 }

154 + (UIImage *)imageFromSource:(UIImage *)src
155  withContentMode:(UIViewContentMode)contentMode
156  cropRect:(CGRect)cropRect
157  displaySize:(CGSize)displaySize
158  scaleOptions:(NINetworkImageViewScaleOptions)scaleOptions
159  interpolationQuality:(CGInterpolationQuality)interpolationQuality {
160 
161  UIImage* resultImage = src;
162 
163  CGImageRef srcImageRef = src.CGImage;
164  CGImageRef croppedImageRef = nil;
165  CGImageRef trimmedImageRef = nil;
166 
167  CGRect srcRect = CGRectMake(0, 0, src.size.width, src.size.height);
168 
169  // Cropping
170  if (!CGRectIsEmpty(cropRect)
171  && !CGRectEqualToRect(cropRect, CGRectMake(0, 0, 1, 1))) {
172  CGRect innerRect = CGRectMake(floorf(src.size.width * cropRect.origin.x),
173  floorf(src.size.height * cropRect.origin.y),
174  floorf(src.size.width * cropRect.size.width),
175  floorf(src.size.height * cropRect.size.height));
176 
177  // Create a new image containing only the cropped inner rect.
178  srcImageRef = CGImageCreateWithImageInRect(srcImageRef, innerRect);
179  croppedImageRef = srcImageRef;
180 
181  // This new image will likely have a different width and height, so we have to update
182  // the source rect as a result.
183  srcRect = CGRectMake(0, 0, CGRectGetWidth(innerRect), CGRectGetHeight(innerRect));
184  }
185 
186  // Display
187  if (0 < displaySize.width
188  && 0 < displaySize.height) {
189 
190  if ((NINetworkImageViewScaleToFillLeavesExcess
191  == (NINetworkImageViewScaleToFillLeavesExcess & scaleOptions))
192  && UIViewContentModeScaleAspectFill == contentMode) {
193  // Make the display size match the aspect ratio of the source image by growing the
194  // display size.
195  CGFloat imageAspectRatio = srcRect.size.width / srcRect.size.height;
196  CGFloat displayAspectRatio = displaySize.width / displaySize.height;
197 
198  if (imageAspectRatio > displayAspectRatio) {
199  // The image is wider than the display, so let's increase the width.
200  displaySize.width = displaySize.height * imageAspectRatio;
201 
202  } else if (imageAspectRatio < displayAspectRatio) {
203  // The image is taller than the display, so let's increase the height.
204  displaySize.height = displaySize.width * (srcRect.size.height / srcRect.size.width);
205  }
206 
207  } else if ((NINetworkImageViewScaleToFitCropsExcess
208  == (NINetworkImageViewScaleToFitCropsExcess & scaleOptions))
209  && UIViewContentModeScaleAspectFit == contentMode) {
210  // Make the display size match the aspect ratio of the source image by shrinking the
211  // display size.
212  CGFloat imageAspectRatio = srcRect.size.width / srcRect.size.height;
213  CGFloat displayAspectRatio = displaySize.width / displaySize.height;
214 
215  if (imageAspectRatio > displayAspectRatio) {
216  // The image is wider than the display, so let's decrease the height.
217  displaySize.height = displaySize.width * (srcRect.size.height / srcRect.size.width);
218 
219  } else if (imageAspectRatio < displayAspectRatio) {
220  // The image is taller than the display, so let's decrease the width.
221  displaySize.width = displaySize.height * imageAspectRatio;
222  }
223  }
224 
225  CGRect srcCropRect = [self sourceRectWithImageSize: srcRect.size
226  displaySize: displaySize
227  contentMode: contentMode];
228  srcCropRect = CGRectMake(floorf(srcCropRect.origin.x),
229  floorf(srcCropRect.origin.y),
230  roundf(srcCropRect.size.width),
231  roundf(srcCropRect.size.height));
232 
233  // Do we need to crop the source?
234  if (!CGRectEqualToRect(srcCropRect, srcRect)) {
235  srcImageRef = CGImageCreateWithImageInRect(srcImageRef, srcCropRect);
236  trimmedImageRef = srcImageRef;
237 
238  srcRect = CGRectMake(0, 0,
239  CGRectGetWidth(srcCropRect),
240  CGRectGetHeight(srcCropRect));
241 
242  // Release the cropped image source to reduce this thread's memory consumption.
243  if (nil != croppedImageRef) {
244  CGImageRelease(croppedImageRef);
245  croppedImageRef = nil;
246  }
247  }
248 
249  // Calcuate the destination frame.
250  CGRect dstBlitRect = [self destinationRectWithImageSize: srcRect.size
251  displaySize: displaySize
252  contentMode: contentMode];
253  dstBlitRect = CGRectMake(floorf(dstBlitRect.origin.x),
254  floorf(dstBlitRect.origin.y),
255  roundf(dstBlitRect.size.width),
256  roundf(dstBlitRect.size.height));
257 
258  // Round any remainder on the display size dimensions.
259  displaySize = CGSizeMake(roundf(displaySize.width), roundf(displaySize.height));
260 
261  // See table "Supported Pixel Formats" in the following guide for support iOS bitmap formats:
262  // http://developer.apple.com/library/mac/#documentation/GraphicsImaging/Conceptual/drawingwithquartz2d/dq_context/dq_context.html
263  CGColorSpaceRef colorSpace = CGColorSpaceCreateDeviceRGB();
264  CGBitmapInfo bmi = kCGImageAlphaPremultipliedLast;
265 
266  // For screen sizes with higher resolutions, we create a larger image with a scale value
267  // so that it appears crisper on the screen.
268  CGFloat screenScale = NIScreenScale();
269 
270  // Create our final composite image.
271  CGContextRef dstBmp = CGBitmapContextCreate(NULL,
272  displaySize.width * screenScale,
273  displaySize.height * screenScale,
274  8,
275  0,
276  colorSpace,
277  bmi);
278 
279  // If this fails then we're likely creating an invalid bitmap and shit's about to go down.
280  // In production this will fail somewhat gracefully, in that we'll end up just using the
281  // source image instead of the cropped and resized image.
282  NIDASSERT(nil != dstBmp);
283 
284  if (nil != dstBmp) {
285  CGRect dstRect = CGRectMake(0, 0,
286  displaySize.width * screenScale,
287  displaySize.height * screenScale);
288 
289  // Render the source image into the destination image.
290  CGContextClearRect(dstBmp, dstRect);
291  CGContextSetInterpolationQuality(dstBmp, interpolationQuality);
292 
293  CGRect scaledBlitRect = CGRectMake(dstBlitRect.origin.x * screenScale,
294  dstBlitRect.origin.y * screenScale,
295  dstBlitRect.size.width * screenScale,
296  dstBlitRect.size.height * screenScale);
297  CGContextDrawImage(dstBmp, scaledBlitRect, srcImageRef);
298 
299  CGImageRef resultImageRef = CGBitmapContextCreateImage(dstBmp);
300 
301  if (nil != resultImageRef) {
302  resultImage = [UIImage imageWithCGImage:resultImageRef
303  scale:screenScale
304  orientation:src.imageOrientation];
305  CGImageRelease(resultImageRef);
306  }
307 
308  CGContextRelease(dstBmp);
309  }
310 
311  CGColorSpaceRelease(colorSpace);
312 
313  } else if (nil != croppedImageRef) {
314  resultImage = [UIImage imageWithCGImage:srcImageRef];
315  }
316 
317  // Memory cleanup.
318  if (nil != trimmedImageRef) {
319  CGImageRelease(trimmedImageRef);
320  }
321  if (nil != croppedImageRef) {
322  CGImageRelease(croppedImageRef);
323  }
324 
325  return resultImage;
326 }
