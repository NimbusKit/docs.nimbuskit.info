70 - (void)heartbeat {
71  [NIDeviceInfo beginCachedDeviceInfo];
72  NIOverviewDeviceLogEntry* logEntry =
73  [[NIOverviewDeviceLogEntry alloc] initWithTimestamp:[NSDate date]];
74  logEntry.bytesOfTotalDiskSpace = [NIDeviceInfo bytesOfTotalDiskSpace];
75  logEntry.bytesOfFreeDiskSpace = [NIDeviceInfo bytesOfFreeDiskSpace];
76  logEntry.bytesOfFreeMemory = [NIDeviceInfo bytesOfFreeMemory];
77  logEntry.bytesOfTotalMemory = [NIDeviceInfo bytesOfTotalMemory];
78  logEntry.batteryLevel = [NIDeviceInfo batteryLevel];
79  logEntry.batteryState = [NIDeviceInfo batteryState];
80  [NIDeviceInfo endCachedDeviceInfo];
81 
82  [self addDeviceLog:logEntry];
83 }

85 - (void)pruneEntriesFromLinkedList:(NSMutableOrderedSet *)ll {
86  NSDate* cutoffDate = [NSDate dateWithTimeIntervalSinceNow:-_oldestLogAge];
87  while ([[((NIOverviewLogEntry *)[ll firstObject])
88  timestamp] compare:cutoffDate] == NSOrderedAscending) {
89  [ll removeObjectAtIndex:0];
90  }
91 }
