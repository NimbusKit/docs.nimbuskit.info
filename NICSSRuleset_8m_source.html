68 // Maintain sanity with a preprocessor macro for the common cases
69 #define RULE_ELEMENT(name,Name,cssKey,type,converter) \
70 static NSString* const k ## Name ## Key = cssKey; \
71 -(BOOL)has ## Name { \
72 return nil != [_ruleset objectForKey: k ## Name ## Key]; \
73 } \
74 -(type)name { \
75 NIDASSERT([self has ## Name]); \
76 if (!_is.cached.Name) { \
77 _##name = [[self class] converter:[_ruleset objectForKey:k##Name##Key]]; \
78 _is.cached.Name = YES; \
79 } \
80 return _##name; \
81 }

217 - (UIFont *)font {
218  NIDASSERT([self hasFont]);
219 
220  if (_is.cached.Font) {
221  return _font;
222  }
223 
224  NSString* fontName = nil;
225  CGFloat fontSize = [UIFont systemFontSize];
226  BOOL fontIsBold = NO;
227  BOOL fontIsItalic = NO;
228 
229  NSArray* values = [_ruleset objectForKey:kFontWeightKey];
230  if (nil != values) {
231  NIDASSERT([values count] == 1);
232  fontIsBold = [[values objectAtIndex:0] isEqualToString:@"bold"];
233  }
234 
235  values = [_ruleset objectForKey:kFontStyleKey];
236  if (nil != values) {
237  NIDASSERT([values count] == 1);
238  fontIsItalic = [[values objectAtIndex:0] isEqualToString:@"italic"];
239  }
240 
241  // There are two ways to set font size and family: font and font-size/font-family.
242  // Newer definitions of these values should overwrite previous definitions so we must
243  // respect ordering here.
244  BOOL hasSetFontName = NO;
245  BOOL hasSetFontSize = NO;
246 
247  NSArray* order = [_ruleset objectForKey:kPropertyOrderKey];
248  for (NSString* name in [order reverseObjectEnumerator]) {
249  if (!hasSetFontName && [name isEqualToString:kFontFamilyKey]) {
250  values = [_ruleset objectForKey:name];
251  NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
252  fontName = [[values objectAtIndex:0] stringByTrimmingCharactersInSet:
253  [NSCharacterSet characterSetWithCharactersInString:@"\""]];
254  hasSetFontName = YES;
255 
256  } else if (!hasSetFontSize && [name isEqualToString:kFontSizeKey]) {
257  values = [_ruleset objectForKey:name];
258  NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
259  NSString* value = [values objectAtIndex:0];
260  if ([value isEqualToString:@"default"]) {
261  fontSize = [UIFont systemFontSize];
262 
263  } else {
264  fontSize = [value floatValue];
265  }
266  hasSetFontSize = YES;
267 
268  } else if (!hasSetFontSize && !hasSetFontName && [name isEqualToString:kFontKey]) {
269  values = [_ruleset objectForKey:name];
270  NIDASSERT([values count] <= 2); if ([values count] < 1) { continue; }
271 
272  if ([values count] >= 1) {
273  // Font size
274  fontSize = [[values objectAtIndex:0] floatValue];
275  hasSetFontSize = YES;
276  }
277  if ([values count] >= 2) {
278  // Font name
279  fontName = [[values objectAtIndex:1] stringByTrimmingCharactersInSet:
280  [NSCharacterSet characterSetWithCharactersInString:@"\""]];
281  hasSetFontName = YES;
282  }
283  }
284 
285  if (hasSetFontName && hasSetFontSize) {
286  // Once we've set all values then we can ignore any previous values.
287  break;
288  }
289  }
290 
291  UIFont* font = nil;
292  if (hasSetFontName) {
293  // If you wish to set the weight and style for a non-standard font family then you will need
294  // to set the font family to the given style manually.
295  NIDASSERT(!fontIsItalic && !fontIsBold);
296  font = [UIFont fontWithName:fontName size:fontSize];
297 
298  } else if (fontIsItalic && fontIsBold) {
299  // There is no easy way to create a bold italic font using the exposed UIFont methods.
300  // Please consider using the exact font name instead. E.g. font-name: Helvetica-BoldObliquei
301  NIDASSERT(!(fontIsItalic && fontIsBold));
302  font = [UIFont systemFontOfSize:fontSize];
303 
304  } else if (fontIsItalic) {
305  font = [UIFont italicSystemFontOfSize:fontSize];
306 
307  } else if (fontIsBold) {
308  font = [UIFont boldSystemFontOfSize:fontSize];
309 
310  } else {
311  font = [UIFont systemFontOfSize:fontSize];
312  }
313 
314  _font = font;
315  _is.cached.Font = YES;
316 
317  return font;
318 }

503 - (void)cacheBorderValues {
504  _borderWidth = 0;
505  _borderColor = nil;
506 
507  NSArray* values = nil;
508 
509  // There are two ways to set border color and width: border and border-color/border-width.
510  // Newer definitions of these values should overwrite previous definitions so we must
511  // respect ordering here.
512  BOOL hasSetBorderColor = NO;
513  BOOL hasSetBorderWidth = NO;
514 
515  NSArray* order = [_ruleset objectForKey:kPropertyOrderKey];
516  for (NSString* name in [order reverseObjectEnumerator]) {
517  if (!hasSetBorderColor && [name isEqualToString:kBorderColorKey]) {
518  values = [_ruleset objectForKey:name];
519  _borderColor = [[self class] colorFromCssValues:values
520  numberOfConsumedTokens:nil];
521  hasSetBorderColor = YES;
522 
523  } else if (!hasSetBorderWidth && [name isEqualToString:kBorderWidthKey]) {
524  values = [_ruleset objectForKey:name];
525  NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
526  _borderWidth = [[values objectAtIndex:0] floatValue];
527  hasSetBorderWidth = YES;
528 
529  } else if (!hasSetBorderColor && !hasSetBorderWidth && [name isEqualToString:kBorderKey]) {
530  values = [_ruleset objectForKey:name];
531 
532  if ([values count] >= 1) {
533  // Border width
534  _borderWidth = [[values objectAtIndex:0] floatValue];
535  hasSetBorderWidth = YES;
536  }
537  if ([values count] >= 3) {
538  // Border color
539  _borderColor = [[self class] colorFromCssValues:[values subarrayWithRange:NSMakeRange(2, [values count] - 2)]
540  numberOfConsumedTokens:nil];
541  hasSetBorderColor = YES;
542  }
543  }
544 
545  if (hasSetBorderColor && hasSetBorderWidth) {
546  // Once we've set all values then we can ignore any previous values.
547  break;
548  }
549  }
550 
551  _is.cached.BorderColor = hasSetBorderColor;
552  _is.cached.BorderWidth = hasSetBorderWidth;
553 }

1027 + (UIColor *)colorFromCssValues:(NSArray *)cssValues numberOfConsumedTokens:(NSInteger *)pNumberOfConsumedTokens {
1028  NSInteger bogus = 0;
1029  if (nil == pNumberOfConsumedTokens) {
1030  pNumberOfConsumedTokens = &bogus;
1031  }
1032  UIColor* color = nil;
1033 
1034  if ([cssValues count] >= 6 && [[cssValues objectAtIndex:0] isEqualToString:@"rgba("]) {
1035  // rgba( x x x x )
1036  color = RGBACOLOR([[cssValues objectAtIndex:1] floatValue],
1037  [[cssValues objectAtIndex:2] floatValue],
1038  [[cssValues objectAtIndex:3] floatValue],
1039  [[cssValues objectAtIndex:4] floatValue]);
1040  *pNumberOfConsumedTokens = 6;
1041 
1042  } else if ([cssValues count] >= 5 && [[cssValues objectAtIndex:0] isEqualToString:@"rgb("]) {
1043  // rgb( x x x )
1044  color = RGBCOLOR([[cssValues objectAtIndex:1] floatValue],
1045  [[cssValues objectAtIndex:2] floatValue],
1046  [[cssValues objectAtIndex:3] floatValue]);
1047  *pNumberOfConsumedTokens = 5;
1048 
1049  } else if ([cssValues count] == 1 && [[cssValues objectAtIndex:0] hasPrefix:@"url("]) {
1050  NSString *image = [cssValues objectAtIndex:0];
1051  image = [image substringWithRange:NSMakeRange(4, image.length - 5)];
1052  image = [image stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@" '\""]];
1053  color = [UIColor colorWithPatternImage:[UIImage imageNamed:image]];
1054  } else if ([cssValues count] >= 1) {
1055  NSString* cssString = [cssValues objectAtIndex:0];
1056 
1057  if ([cssString characterAtIndex:0] == '#') {
1058  unsigned long colorValue = 0;
1059 
1060  // #FFF
1061  if ([cssString length] == 4) {
1062  colorValue = strtol([cssString UTF8String] + 1, nil, 16);
1063  colorValue = ((colorValue & 0xF00) << 12) | ((colorValue & 0xF00) << 8)
1064  | ((colorValue & 0xF0) << 8) | ((colorValue & 0xF0) << 4)
1065  | ((colorValue & 0xF) << 4) | (colorValue & 0xF);
1066 
1067  // #FFFFFF
1068  } else if ([cssString length] == 7) {
1069  colorValue = strtol([cssString UTF8String] + 1, nil, 16);
1070  }
1071 
1072  color = RGBCOLOR(((colorValue & 0xFF0000) >> 16),
1073  ((colorValue & 0xFF00) >> 8),
1074  (colorValue & 0xFF));
1075  } else if ([cssString caseInsensitiveCompare:@"none"] == NSOrderedSame) {
1076  // Special case to "undo" a color that was set by some other rule
1077  color = nil;
1078  } else {
1079  color = [[self colorTable] objectForKey:cssString];
1080  }
1081 
1082  *pNumberOfConsumedTokens = 1;
1083  }
1084  return color;
1085 }
