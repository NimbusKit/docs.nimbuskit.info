77 + (BOOL)googleChromeWithURL:(NSURL *)url {
78  NSString *chromeScheme = nil;
79  if ([url.scheme isEqualToString:@"http"]) {
80  chromeScheme = sGoogleChromeHttpScheme;
81  } else if ([url.scheme isEqualToString:@"https"]) {
82  chromeScheme = sGoogleChromeHttpsScheme;
83  }
84 
85  if (chromeScheme) {
86  NSRange rangeForScheme = [[url absoluteString] rangeOfString:@":"];
87  NSString* urlNoScheme = [[url absoluteString] substringFromIndex:rangeForScheme.location + 1];
88  NSString* chromeUrlString = [chromeScheme stringByAppendingString:urlNoScheme];
89  NSURL* chromeUrl = [NSURL URLWithString:chromeUrlString];
90 
91  BOOL didOpen = [[UIApplication sharedApplication] openURL:chromeUrl];
92  if (!didOpen) {
93  didOpen = [self appStoreWithAppId:[self googleChromeAppStoreId]];
94  }
95 
96  return didOpen;
97  }
98 
99  return NO;
100 }

238 + (BOOL)mailWithInvocation:(NIMailAppInvocation *)invocation {
239  NSMutableDictionary* parameters = [NSMutableDictionary dictionary];
240 
241  NSString* urlPath = sMailScheme;
242 
243  if (NIIsStringWithAnyText(invocation.recipient)) {
244  urlPath = [urlPath stringByAppendingString:NIStringByAddingPercentEscapesForURLParameterString(invocation.recipient)];
245  }
246 
247  if (NIIsStringWithAnyText(invocation.cc)) {
248  [parameters setObject:invocation.cc forKey:@"cc"];
249  }
250  if (NIIsStringWithAnyText(invocation.bcc)) {
251  [parameters setObject:invocation.bcc forKey:@"bcc"];
252  }
253  if (NIIsStringWithAnyText(invocation.subject)) {
254  [parameters setObject:invocation.subject forKey:@"subject"];
255  }
256  if (NIIsStringWithAnyText(invocation.body)) {
257  [parameters setObject:invocation.body forKey:@"body"];
258  }
259 
260  urlPath = NIStringByAddingQueryDictionaryToString(urlPath, parameters);
261 
262  return [[UIApplication sharedApplication] openURL:[NSURL URLWithString:urlPath]];
263 }

381 + (BOOL)applicationWithScheme:(NSString *)applicationScheme
382  appStoreId:(NSString *)appStoreId
383  andPath:(NSString *)path {
384  BOOL didOpen = false;
385  NSString* urlPath = applicationScheme;
386 
387  // Were we passed a path?
388  if (path != nil) {
389  // Generate the full application URL
390  urlPath = [urlPath stringByAppendingFormat:@"%@", path];
391  }
392 
393  // Try to open the application URL
394  didOpen = [[UIApplication sharedApplication] openURL:[NSURL URLWithString:urlPath]];
395 
396  // Didn't open and we have an appStoreId
397  if (!didOpen && appStoreId != nil) {
398  // Open the app store instead
399  didOpen = [self appStoreWithAppId:appStoreId];
400  }
401 
402  return didOpen;
403 }

434 + (NSURL *)urlForInstagramImageAtFilePath:(NSString *)filePath error:(NSError **)error {
435  if (![self instagramIsInstalled]) {
436  return nil;
437  }
438 
439  UIImage* image = [[UIImage alloc] initWithContentsOfFile:filePath];
440 
441  // Unable to read the image.
442  if (nil == image) {
443  if (nil != error) {
444  *error = [NSError errorWithDomain: NSCocoaErrorDomain
445  code: NSFileReadUnknownError
446  userInfo: [NSDictionary dictionaryWithObject: filePath
447  forKey: NSFilePathErrorKey]];
448  }
449  return nil;
450  }
451 
452  // Instagram requires that images are at least 612x612 and preferably square.
453  if (image.size.width < 612
454  || image.size.height < 612) {
455  if (nil != error) {
456  *error = [NSError errorWithDomain: NINimbusErrorDomain
457  code: NIImageTooSmall
458  userInfo: [NSDictionary dictionaryWithObject: image
459  forKey: NIImageErrorKey]];
460  }
461  return nil;
462  }
463 
464  NSFileManager* fm = [NSFileManager defaultManager];
465  NSArray* paths = NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES);
466 
467  NIDASSERT(NIIsArrayWithObjects(paths));
468  if (!NIIsArrayWithObjects(paths)) {
469  return nil;
470  }
471 
472  NSString* documentsPath = [paths objectAtIndex:0];
473  NSString* destinationPath = [documentsPath stringByAppendingPathComponent:
474  [NSString stringWithFormat:@"nimbus-instagram-image-%.0f.ig",
475  [NSDate timeIntervalSinceReferenceDate]]];
476 
477  [fm copyItemAtPath: filePath
478  toPath: destinationPath
479  error: error];
480 
481  NIDASSERT(nil == error || nil == *error);
482  if (nil == error || nil == *error) {
483  return [NSURL URLWithString:[@"file:" stringByAppendingString:destinationPath]];
484 
485  } else {
486  return nil;
487  }
488 }
