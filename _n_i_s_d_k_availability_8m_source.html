00072 Class NIUITapGestureRecognizerClass(void) {
00073   static Class sClass = nil;
00074   static BOOL hasChecked = NO;
00075   if (!hasChecked) {
00076     hasChecked = YES;
00077 
00078     // An interesting gotcha: UITapGestureRecognizer actually *does* exist in iOS 3.0, but does
00079     // not conform to all of the same methods that the 3.2 implementation does. This can be
00080     // really confusing, so instead of returning the class, we'll always return nil on
00081     // pre-iOS 3.2 devices.
00082     if (NIDeviceOSVersionIsAtLeast(kCFCoreFoundationVersionNumber_iPhoneOS_3_2)) {
00083       sClass = NSClassFromString(@"UITapGestureRecognizer");
00084     }
00085   }
00086   return sClass;
