00028 static int              gNetworkTaskCount = 0;
00029 static pthread_mutex_t  gMutex = PTHREAD_MUTEX_INITIALIZER;

00032 void NINetworkActivityTaskDidStart(void) {
00033   pthread_mutex_lock(&gMutex);
00034 
00035   BOOL enableNetworkActivityIndicator = (0 == gNetworkTaskCount);
00036 
00037   ++gNetworkTaskCount;
00038 
00039   if (enableNetworkActivityIndicator) {
00040     [UIApplication sharedApplication].networkActivityIndicatorVisible = YES;
00041   }
00042 
00043   pthread_mutex_unlock(&gMutex);
00044 }

00048 void NINetworkActivityTaskDidFinish(void) {
00049   pthread_mutex_lock(&gMutex);
00050 
00051   --gNetworkTaskCount;
00052   // If this asserts, you don't have enough stop requests to match your start requests.
00053   NIDASSERT(gNetworkTaskCount >= 0);
00054   gNetworkTaskCount = MAX(0, gNetworkTaskCount);
00055 
00056   if (gNetworkTaskCount == 0) {
00057     [UIApplication sharedApplication].networkActivityIndicatorVisible = NO;
00058   }
00059 
00060   pthread_mutex_unlock(&gMutex);
00061 }
