88 - (void)testNIIsStringWithWhitespaceAndNewlines {
89  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(@""), @"Empty string should be whitespace and newlines.");
90  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(@" "), @"Space should be whitespace and newlines.");
91  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(@" \n\r"), @"Whitespace and newlines should be whitespace and newlines.");
92  STAssertFalse(NIIsStringWithWhitespaceAndNewlines(nil), @"nil is not a string");
93  STAssertFalse(NIIsStringWithWhitespaceAndNewlines(@"cat"), @"Words are not whitespace and newlines");
94 
95  for (unsigned short unicode = 0x000A; unicode <= 0x000D; ++unicode) {
96  NSString* str = [NSString stringWithFormat:@"%C", unicode];
97  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(str),
98  @"Unicode string #%X should be whitespace.", unicode);
99  }
100 
101  NSString* str = [NSString stringWithFormat:@"%C", (unsigned short)0x0085];
102  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(str), @"Unicode string should be whitespace.");
103 
104  STAssertTrue(NIIsStringWithWhitespaceAndNewlines(@" \t\r\n"), @"Empty string should be whitespace.");
105 
106  STAssertTrue(!NIIsStringWithWhitespaceAndNewlines(@"a"), @"Text should not be whitespace.");
107  STAssertTrue(!NIIsStringWithWhitespaceAndNewlines(@" \r\n\ta\r\n "), @"Text should not be whitespace.");
108 }
