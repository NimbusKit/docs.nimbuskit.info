00243 - (NSUInteger)countByEnumeratingWithState:(NSFastEnumerationState *)state
00244                                   objects:(id *)stackbuf
00245                                     count:(NSUInteger)len {
00246   // Initialization condition.
00247   if (0 == state->state) {
00248     // Whenever the linked list is modified, the modification number increases. This allows
00249     // enumeration to bail out if the linked list is modified mid-flight.
00250     state->mutationsPtr = &_modificationNumber;
00251   }
00252 
00253   NSUInteger numberOfItemsReturned = 0;
00254 
00255   // If there is no _tail (i.e. this is an empty list) then this will end immediately.
00256   if ((struct NILinkedListNode *)state->state != _tail) {
00257     state->itemsPtr = stackbuf;
00258 
00259     if (0 == state->state) {
00260       // Initialize the state here instead of above when we check 0 == state->state because
00261       // for single item linked lists head == tail. If we initialized it in the initialization
00262       // condition, state->state != _tail check would fail and we wouldn't return the single
00263       // object.
00264       state->state = (unsigned long)_head;
00265     }
00266 
00267     // Return *at most* the number of request objects.
00268     while ((0 != state->state) && (numberOfItemsReturned < len)) {
00269       struct NILinkedListNode* node = (struct NILinkedListNode *)state->state;
00270       stackbuf[numberOfItemsReturned] = node->object;
00271       state->state = (unsigned long)node->next;
00272       ++numberOfItemsReturned;
00273     }
00274 
00275     if (0 == state->state) {
00276       // Final step condition. We allow the above loop to overstep the end one iteration,
00277       // because we rewind it one step here (to ensure that the next time enumeration occurs,
00278       // state == _tail.
00279       state->state = (unsigned long)_tail;
00280     }
00281 
00282   } // else we've returned all of the items that we can; leave numberOfItemsReturned as 0 to
00283     // signal that there is nothing left to be done.
00284 
00285   return numberOfItemsReturned;
00286 }

00401   // nil objects can not be added to a linked list.
00402   NIDASSERT(nil != object);
00403   if (nil == object) {
