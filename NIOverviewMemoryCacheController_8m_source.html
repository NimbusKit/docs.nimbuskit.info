70 - (void)refreshModel {
71  NSMutableArray* contents = [NSMutableArray array];
72  NSString* summary = nil;
73 
74  // Display a summary.
75  if ([self.cache isKindOfClass:[NIImageMemoryCache class]]) {
76  NIImageMemoryCache* imageCache = (NIImageMemoryCache *)self.cache;
77  summary = [NSString stringWithFormat:
78  @"Number of images: %d\nNumber of pixels: %@/%@\nStress limit: %@",
79  self.cache.count,
80  NIStringFromBytes(imageCache.numberOfPixels),
81  NIStringFromBytes(imageCache.maxNumberOfPixels),
82  NIStringFromBytes(imageCache.maxNumberOfPixelsUnderStress)];
83 
84  } else {
85  summary = [NSString stringWithFormat:
86  @"Number of objects: %d",
87  self.cache.count];
88  }
89  [contents addObject:[NITableViewModelFooter footerWithTitle:summary]];
90 
91  NSDateFormatter* formatter = [[NSDateFormatter alloc] init];
92  // We care more about time than date here.
93  [formatter setDateStyle:NSDateFormatterShortStyle];
94  [formatter setTimeStyle:NSDateFormatterMediumStyle];
95 
96  // Add each of the cache objects to the model.
97  for (id cacheObject in self.cache.lruCacheObjects) {
98  NSString* name = nil;
99  UIImage* image = nil;
100  NSDate* lastAccessTime = nil;
101 
102  // We're accessing a private object from the core here, so we can't assume that any of
103  // these selectors will be around forever. That being said, we should try to remember to
104  // update this controller if the cache object internals ever change.
105  // If any of these assertions fire it means we've changed the cache object signatures but
106  // haven't gotten around to updating this controller yet.
107  NIDASSERT([cacheObject respondsToSelector:@selector(name)]);
108  if ([cacheObject respondsToSelector:@selector(name)]) {
109  name = [cacheObject performSelector:@selector(name)];
110  }
111  NIDASSERT([cacheObject respondsToSelector:@selector(object)]);
112  if ([cacheObject respondsToSelector:@selector(object)]) {
113  id object = [cacheObject performSelector:@selector(object)];
114  if ([object isKindOfClass:[UIImage class]]) {
115  image = object;
116  }
117  }
118  NIDASSERT([cacheObject respondsToSelector:@selector(lastAccessTime)]);
119  if ([cacheObject respondsToSelector:@selector(lastAccessTime)]) {
120  lastAccessTime = [cacheObject performSelector:@selector(lastAccessTime)];
121  }
122 
123  [contents addObject:
124  [NISubtitleCellObject objectWithTitle:name
125  subtitle:[NSString stringWithFormat:
126  @"Last access: %@",
127  [formatter stringFromDate:lastAccessTime]]
128  image:image]];
129  }
130 
131  if (0 == self.cache.count) {
132  [contents addObject:[NITitleCellObject objectWithTitle:@"No cache objects"]];
133  }
134 
135  [contents addObject:[NITableViewModelFooter footerWithTitle:
136  @"The most-recently-used object is here at the bottom"]];
137 
138  self.model = [[NITableViewModel alloc] initWithSectionedArray:contents
139  delegate:(id)[NICellFactory class]];
140  self.tableView.dataSource = self.model;
141  [self.tableView reloadData];
142 }
