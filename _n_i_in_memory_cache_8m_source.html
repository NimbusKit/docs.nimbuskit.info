00343   NSDictionary* cacheMap = [self.cacheMap copy];

00470 - (void)didSetObject:(id)object withName:(NSString *)name {
00471   // Reduce the cache size after the object has been set in case the cache size is smaller
00472   // than the object that's being added and we need to remove this object right away. If we
00473   // try to reduce the cache size before the object's been set, we won't have anything to remove
00474   // and we'll get stuck in an infinite loop.
00475   if (self.maxNumberOfPixels > 0) {
00476     // Remove least recently used images until we satisfy our memory constraints.
00477     while (self.numberOfPixels > self.maxNumberOfPixels
00478            && [self.lruCacheObjects count] > 0) {
00479       NIMemoryCacheInfo* info = [self.lruCacheObjects firstObject];
