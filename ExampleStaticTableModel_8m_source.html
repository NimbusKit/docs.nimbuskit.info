39 - (void)dealloc {

47 - (id)initWithNibName:(NSString *)nibNameOrNil bundle:(NSBundle *)nibBundleOrNil {
48  if ((self = [super initWithNibName:nibNameOrNil bundle:nibBundleOrNil])) {
49  NSArray* tableContents =
50  [NSArray arrayWithObjects:
51  @"Section 1",
52  [NSDictionary dictionaryWithObject:@"Row 1" forKey:@"title"],
53  [NSDictionary dictionaryWithObject:@"Row 2" forKey:@"title"],
54  [NSDictionary dictionaryWithObject:@"Row 3" forKey:@"title"],
55 
56  @"Section 2",
57  [NSDictionary dictionaryWithObject:@"Row 4" forKey:@"title"],
58  [NITableViewModelFooter footerWithTitle:@"Footer"],
59  nil];
60 
61  // This controller creates the table view cells.
62  _model = [[NITableViewModel alloc] initWithSectionedArray:tableContents
63  delegate:self];
64  }
65  return self;
66 }
67 
68 - (void)viewDidLoad {
69  [super viewDidLoad];
70 
71  // Only assign the table view's data source after the view has loaded.
72  // You must be careful when you call self.tableView in general because it will call loadView
73  // if the view has not been loaded yet. You do not need to clear the data source when the
74  // view is unloaded (more importantly: you shouldn't, due to the reason just outlined
75  // regarding loadView).
76  self.tableView.dataSource = _model;
77 }

79 - (UITableViewCell *)tableViewModel: (NITableViewModel *)tableViewModel
80  cellForTableView: (UITableView *)tableView
81  atIndexPath: (NSIndexPath *)indexPath
82  withObject: (id)object {
83  // A pretty standard implementation of creating table view cells follows.
84  UITableViewCell* cell = [tableView dequeueReusableCellWithIdentifier:@"row"];
85 
86  if (nil == cell) {
87  cell = [[[UITableViewCell alloc] initWithStyle: UITableViewCellStyleDefault
88  reuseIdentifier: @"row"]
89  autorelease];
90  cell.accessoryType = UITableViewCellAccessoryDisclosureIndicator;
91  }
92 
93  cell.textLabel.text = [object objectForKey:@"title"];
94 
95  return cell;
96 }
