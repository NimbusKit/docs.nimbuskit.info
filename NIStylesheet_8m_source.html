61 // Builds a map of significant scopes to full scopes.
62 //
63 // For example, consider the following rulesets:
64 //
65 // .root UIButton {
66 // }
67 // .apple UIButton {
68 // }
69 // UIButton UIView {
70 // }
71 // UIView {
72 // }
73 //
74 // The generated scope map will look like:
75 //
76 // UIButton => (.root UIButton, .apple UIButton)
77 // UIView => (UIButton UIView, UIView)
78 //
79 - (void)rebuildSignificantScopeToScopes {
80  NSMutableDictionary* significantScopeToScopes =
81  [[NSMutableDictionary alloc] initWithCapacity:[_rawRulesets count]];
82 
83  for (NSString* scope in _rawRulesets) {
84  NSArray* parts = [scope componentsSeparatedByString:@" "];
85  NSString* mostSignificantScopePart = [parts lastObject];
86 
87  // TODO (jverkoey Oct 6, 2011): We should respect CSS specificity. Right now this will
88  // give higher precedance to newer styles. Instead, we should prefer styles that have more
89  // selectors.
90  NSMutableArray* scopes = [significantScopeToScopes objectForKey:mostSignificantScopePart];
91  if (nil == scopes) {
92  scopes = [[NSMutableArray alloc] initWithObjects:scope, nil];
93  [significantScopeToScopes setObject:scopes forKey:mostSignificantScopePart];
94 
95  } else {
96  [scopes addObject:scope];
97  }
98  }
99 
100  _significantScopeToScopes = [significantScopeToScopes copy];
101 }
102

129 - (BOOL)loadFromPath:(NSString *)path
130  pathPrefix:(NSString *)pathPrefix
131  delegate:(id<NICSSParserDelegate>)delegate {
132  BOOL loadDidSucceed = NO;
133 
134  @synchronized(self) {
135  _rawRulesets = nil;
136  _significantScopeToScopes = nil;
137 
138  _ruleSets = [[NSMutableDictionary alloc] init];
139 
140  NICSSParser* parser = [[NICSSParser alloc] init];
141 
142  NSDictionary* results = [parser dictionaryForPath:path
143  pathPrefix:pathPrefix
144  delegate:delegate];
145  if (nil != results && ![parser didFailToParse]) {
146  _rawRulesets = results;
147  loadDidSucceed = YES;
148  }
149 
150  if (loadDidSucceed) {
151  [self ruleSetsDidChange];
152  }
153  }
154 
155  return loadDidSucceed;
156 }

158 - (void)addStylesheet:(NIStylesheet *)stylesheet {
159  NIDASSERT(nil != stylesheet);
160  if (nil == stylesheet) {
161  return;
162  }
163 
164  @synchronized(self) {
165  NSMutableDictionary* compositeRuleSets = [self.rawRulesets mutableCopy];
166 
167  BOOL ruleSetsDidChange = NO;
168 
169  for (NSString* selector in stylesheet.rawRulesets) {
170  NSDictionary* incomingRuleSet = [stylesheet.rawRulesets objectForKey:selector];
171  NSDictionary* existingRuleSet = [self.rawRulesets objectForKey:selector];
172 
173  // Don't bother adding empty rulesets.
174  if ([incomingRuleSet count] > 0) {
175  ruleSetsDidChange = YES;
176 
177  if (nil == existingRuleSet) {
178  // There is no rule set of this selector - simply add the new one.
179  [compositeRuleSets setObject:incomingRuleSet forKey:selector];
180  continue;
181  }
182 
183  NSMutableDictionary* compositeRuleSet = [existingRuleSet mutableCopy];
184  // Add the incoming rule set entries, overwriting any existing ones.
185  [compositeRuleSet addEntriesFromDictionary:incomingRuleSet];
186 
187  [compositeRuleSets setObject:compositeRuleSet forKey:selector];
188  }
189  }
190 
191  _rawRulesets = [compositeRuleSets copy];
192 
193  if (ruleSetsDidChange) {
194  [self ruleSetsDidChange];
195  }
196  }
197 }

199 - (NSString*)descriptionForView:(UIView *)view withClassName:(NSString *)className inDOM:(NIDOM *)dom andViewName:(NSString *)viewName {
200  NSMutableString *description = [[NSMutableString alloc] init];
201  NICSSRuleset *ruleset = [self rulesetForClassName:className];
202  if (nil != ruleset) {
203  NSRange r = [className rangeOfString:@":"];
204  if ([view respondsToSelector:@selector(descriptionWithRuleSet:forPseudoClass:inDOM:withViewName:)]) {
205  if (r.location != NSNotFound) {
206  [description appendString:[(id<NIStyleable>)view descriptionWithRuleSet:ruleset forPseudoClass:[className substringFromIndex:r.location+1] inDOM:dom withViewName:viewName]];
207  } else {
208  [description appendString:[(id<NIStyleable>)view descriptionWithRuleSet:ruleset forPseudoClass:nil inDOM:dom withViewName:viewName]];
209  }
210  } else {
211  [description appendFormat:@"// Description not supported for %@ with selector %@\n", view, className];
212  }
213  }
214  return description;
215 }

232 - (void)applyStyleToView:(UIView *)view withClassName:(NSString *)className inDOM:(NIDOM *)dom {
233  NICSSRuleset *ruleset = [self rulesetForClassName:className];
234  if (nil != ruleset) {
235  NSRange r = [className rangeOfString:@":"];
236  if (r.location != NSNotFound && [view respondsToSelector:@selector(applyStyleWithRuleSet:forPseudoClass:inDOM:)]) {
237  [(id<NIStyleable>)view applyStyleWithRuleSet:ruleset forPseudoClass: [className substringFromIndex:r.location+1] inDOM:dom];
238  } else {
239  [self applyRuleSet:ruleset toView:view inDOM:dom];
240  }
241  }

244 - (NICSSRuleset*) addSelectors: (NSArray*) selectors toRuleset: (NICSSRuleset*) ruleSet forClassName: (NSString*) className
245 {
246  if ([selectors count] > 0) {
247  // Gather all of the rule sets for this view into a composite rule set.
248  ruleSet = ruleSet ?: [_ruleSets objectForKey:className];
249 
250  if (nil == ruleSet) {
251  ruleSet = [[[NIStylesheet rulesetClass] alloc] init];
252 
253  // Composite the rule sets into one.
254  for (NSString* selector in selectors) {
255  [ruleSet addEntriesFromDictionary:[_rawRulesets objectForKey:selector]];
256  }
257 
258  NIDASSERT(nil != _ruleSets);
259  [_ruleSets setObject:ruleSet forKey:className];
260  }
261  }
262 
263  return ruleSet;
264 }

268  return [self addSelectors:selectors toRuleset:nil forClassName:className];
