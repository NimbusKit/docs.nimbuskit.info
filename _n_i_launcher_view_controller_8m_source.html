00110 - (UIButton *)launcherView: (NILauncherView *)launcherView
00111              buttonForPage: (NSInteger)page
00112                    atIndex: (NSInteger)buttonIndex {
00113   UIButton* button = [[[[self launcherButtonClass] alloc] init] autorelease];
00114 
00115   // launcherButtonClass must return a class that is a subclass of UIButton.
00116   NIDASSERT([button isKindOfClass:[UIButton class]]);
00117   if (![button isKindOfClass:[UIButton class]]) {
00118     return nil;
00119   }
00120 
00121   NILauncherItemDetails* item = [[_pages objectAtIndex:page] objectAtIndex:buttonIndex];
00122   [button setTitle:item.title forState:UIControlStateNormal];
00123   [button setImage: [UIImage imageWithContentsOfFile:item.imagePath]
00124           forState: UIControlStateNormal];
