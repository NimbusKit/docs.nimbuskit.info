00128 - (CGRect)frameForPagingScrollView {
00129   CGRect frame = self.bounds;
00130 
00131   // We make the paging scroll view a little bit wider on the side edges so that there
00132   // there is space between the pages when flipping through them.
00133   frame.origin.x -= self.pageHorizontalMargin;
00134   frame.size.width += (2 * self.pageHorizontalMargin);
00135 
00136   return frame;
00137 }

00141 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
00142   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00143   // placement. When the device is in landscape orientation, the frame will still be in
00144   // portrait because the pagingScrollView is the root view controller's view, so its
00145   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00146   // will be in landscape because it has a rotation transform applied.
00147   CGRect bounds = self.pagingScrollView.bounds;
00148   CGRect pageFrame = bounds;
00149 
00150   // We need to counter the extra spacing added to the paging scroll view in
00151   // frameForPagingScrollView:
00152   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00153   pageFrame.origin.x = (bounds.size.width * pageIndex) + self.pageHorizontalMargin;
00154 
00155   return pageFrame;
00156 }

00175 - (BOOL)isDisplayingPageForIndex:(NSInteger)pageIndex {
00176   BOOL foundPage = NO;
00177 
00178   // There will never be more than 3 visible pages in this array, so this lookup is
00179   // effectively O(C) constant time.

00192 - (NSInteger)currentVisiblePageIndex {
00193   CGPoint contentOffset = self.pagingScrollView.contentOffset;
00194   CGSize boundsSize = self.pagingScrollView.bounds.size;
00195 
00196   // Whatever image is currently displayed in the center of the screen is the currently
00197   // visible image.
00198   return boundi((NSInteger)(floorf((contentOffset.x + boundsSize.width / 2) / boundsSize.width)
00199                             + 0.5f),
00200                 0, self.numberOfPages - 1);

00206   if (0 >= _numberOfPages) {

00278 - (void)updateVisiblePages {
00279   NSRange visiblePageRange = [self visiblePageRange];
00280 
00281   // Recycle no-longer-visible pages. We copy _visiblePages because we may modify it while we're
00282   // iterating over it.
00283   for (UIView<NIPagingScrollViewPage>* page in [[_visiblePages copy] autorelease]) {
00284     if (!NSLocationInRange(page.pageIndex, visiblePageRange)) {
00285       [_viewRecycler recycleView:page];
00286       [page removeFromSuperview];
00287 
00288       [self didRecyclePage:page];
00289 
00290       [_visiblePages removeObject:page];
00291     }
00292   }
00293 
00294   NSInteger oldCenterPageIndex = self.centerPageIndex;
00295     
00296   if (_numberOfPages > 0) {
00297     _centerPageIndex = [self currentVisiblePageIndex];
00298       
00299     // Prioritize displaying the currently visible page.
00300     if (![self isDisplayingPageForIndex:_centerPageIndex]) {
00301       [self displayPageAtIndex:_centerPageIndex];
00302     }
00303       
00304     // Add missing pages.
00305     for (int pageIndex = visiblePageRange.location;
00306          pageIndex < NSMaxRange(visiblePageRange); ++pageIndex) {
00307       if (![self isDisplayingPageForIndex:pageIndex]) {
00308         [self displayPageAtIndex:pageIndex];
00309       }
00310     }
00311   } else {
00312     _centerPageIndex = -1;
00313   }
00314 
00315   if (oldCenterPageIndex != _centerPageIndex
00316       && [self.delegate respondsToSelector:@selector(pagingScrollViewDidChangePages:)]) {
00317     [self.delegate pagingScrollViewDidChangePages:self];
00318   }
00319 }

00474 - (void)reloadData {
00475   NIDASSERT(nil != _dataSource);
00476 
00477   // Remove any visible pages from the view before we release the sets.
00478   for (UIView<NIPagingScrollViewPage>* page in _visiblePages) {
00479     [_viewRecycler recycleView:page];
00480     [(UIView *)page removeFromSuperview];
00481   }
00482 
00483   NI_RELEASE_SAFELY(_visiblePages);
00484 
00485   // If there is no data source then we can't do anything particularly interesting.
00486   if (nil == _dataSource) {
00487     _isModifyingContentOffset = YES;
00488     self.pagingScrollView.contentSize = self.bounds.size;
00489     self.pagingScrollView.contentOffset = CGPointZero;
00490     _isModifyingContentOffset = NO;
00491 
00492     // May as well just get rid of all the views then.
00493     [_viewRecycler removeAllViews];
00494 
00495     return;
00496   }
00497 
00498   _visiblePages = [[NSMutableSet alloc] init];
00499 
00500   // Cache the number of pages.
00501   _numberOfPages = [_dataSource numberOfPagesInPagingScrollView:self];
00502   self.pagingScrollView.frame = [self frameForPagingScrollView];
00503   self.pagingScrollView.contentSize = [self contentSizeForPagingScrollView];
00504 
00505   NSInteger oldCenterPageIndex = _centerPageIndex;
00506   if (oldCenterPageIndex >= 0) {
00507     _centerPageIndex = boundi(_centerPageIndex, 0, _numberOfPages - 1);
00508 
00509     // The content size is calculated based on the number of pages and the scroll view frame.
00510     _isModifyingContentOffset = YES;
00511     CGPoint offset = [self frameForPageAtIndex:_centerPageIndex].origin;
00512     offset.x -= self.pageHorizontalMargin;
00513     self.pagingScrollView.contentOffset = offset;
00514     _isModifyingContentOffset = NO;
00515   }
00516 
00517   // Begin requesting the page information from the data source.
00518   [self updateVisiblePages];
00519 }

00523 - (void)willRotateToInterfaceOrientation: (UIInterfaceOrientation)toInterfaceOrientation
00524                                 duration: (NSTimeInterval)duration {
00525   // Here, our pagingScrollView bounds have not yet been updated for the new interface
00526   // orientation. This is a good place to calculate the content offset that we will
00527   // need in the new orientation.
00528   CGFloat offset = self.pagingScrollView.contentOffset.x;
00529   CGFloat pageWidth = self.pagingScrollView.bounds.size.width;
00530 
00531   if (offset >= 0) {
00532     _firstVisiblePageIndexBeforeRotation = floorf(offset / pageWidth);
00533     _percentScrolledIntoFirstVisiblePage = ((offset
00534                                             - (_firstVisiblePageIndexBeforeRotation * pageWidth))
00535                                            / pageWidth);

00537   } else {
00538     _firstVisiblePageIndexBeforeRotation = 0;
00539     _percentScrolledIntoFirstVisiblePage = offset / pageWidth;
00540   }

00596 - (void)moveToPageAtIndex:(NSInteger)pageIndex animated:(BOOL)animated {
00597   if (_isAnimatingToPage) {
00598     // Don't allow re-entry for sliding animations.
00599     return;
00600   }

00603   offset.x -= self.pageHorizontalMargin;

00605   _isModifyingContentOffset = YES;

00609   if (animated) {

00612     [NSObject cancelPreviousPerformRequestsWithTarget: self];

00615     // changes while we're animating (like when rotating the device). To do this we need
00616     // to know the destination index for the animation.

00618                withObject: pageIndexNumber
