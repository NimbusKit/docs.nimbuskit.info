36 NSString* NIStringFromBytes(unsigned long long bytes) {
37  static const void* sOrdersOfMagnitude[] = {
38  @"bytes", @"KB", @"MB", @"GB"
39  };
40 
41  // Determine what magnitude the number of bytes is by shifting off 10 bits at a time
42  // (equivalent to dividing by 1024).
43  unsigned long magnitude = 0;
44  unsigned long long highbits = bytes;
45  unsigned long long inverseBits = ~((unsigned long long)0x3FF);
46  while ((highbits & inverseBits)
47  && magnitude + 1 < (sizeof(sOrdersOfMagnitude) / sizeof(void *))) {
48  // Shift off an order of magnitude.
49  highbits >>= 10;
50  magnitude++;
51  }
52 
53  if (magnitude > 0) {
54  unsigned long long dividend = 1024 << (magnitude - 1) * 10;
55  double result = ((double)bytes / (double)(dividend));
56  return [NSString stringWithFormat:@"%.2f %@",
57  result,
58  sOrdersOfMagnitude[magnitude]];
59 
60  } else {
61  // We don't need to bother with dividing bytes.
62  return [NSString stringWithFormat:@"%lld %@", bytes, sOrdersOfMagnitude[magnitude]];
63  }
64 }

82 + (BOOL)updateFileSystemAttributes {
83  NSError* error = nil;
84  // This path could be any path that is on the device's local disk.
85  NSArray* paths = NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES);
86 
87  // Fetch the file system information based on the path given (the user's documents directory).
88  sFileSystem =
89  [[NSFileManager defaultManager] attributesOfFileSystemForPath: [paths lastObject]
90  error: &error];
91  return (nil == error);
