00139         // Ensure that we're not modifying a property.

00141       }

00151         if (nil != _currentPropertyName) {
00152           NSMutableArray* values = [_mutatingRuleset objectForKey:_currentPropertyName];

00194     // Parse individual characters.
00195     case CSSUNKNOWN: {
00196       switch (text[0]) {
00197 
00198         // Commit the current selector and start a new one.
00199         case ',': {

00229             NSMutableDictionary* existingProperties = [_rulesets objectForKey:name];

00231             if (nil == existingProperties) {
00232               NSMutableDictionary* ruleSet = [_mutatingRuleset mutableCopy];
00233               [_rulesets setObject:ruleSet forKey:name];
00234               NI_RELEASE_SAFELY(ruleSet);

00238               // Merge the orders.
00239               {
00240                 NSMutableArray* order = [existingProperties objectForKey:kPropertyOrderKey];
00241                 [order addObjectsFromArray:[_mutatingRuleset objectForKey:kPropertyOrderKey]];
00242                 [_mutatingRuleset setObject:order forKey:kPropertyOrderKey];
00243               }

00246                 [existingProperties setObject:[_mutatingRuleset objectForKey:key] forKey:key];
00247               }
00248               // Add the order of the new properties.
00249               [[existingProperties objectForKey:kPropertyOrderKey] addObjectsFromArray:

00251             }
00252           }
00253

00255           [_scopesForActiveRuleset removeAllObjects];
00256           _state.Flags.InsideRuleset = NO;
00257           _state.Flags.InsideProperty = NO;
00258           _state.Flags.InsideFunction = NO;
00259           break;
00260         }
00261 
00262         case ':': {
00263           // Defining a property value.
00264           if (_state.Flags.InsideRuleset) {
00265             _state.Flags.InsideProperty = YES;
00266           }
00267           break;
00268         }
00269 
00270         case ')': {
00271           if (_state.Flags.InsideFunction && nil != _currentPropertyName) {
00272             NSMutableArray* values = [_mutatingRuleset objectForKey:_currentPropertyName];
00273             [values addObject:lowercaseTextAsString];
00274           }
00275           _state.Flags.InsideFunction = NO;
00276           break;
00277         }
00278 
00279         case ';': {
00280           // Committing a property value.
00281           if (_state.Flags.InsideRuleset) {
00282             _state.Flags.InsideProperty = NO;
00283           }
00284           break;
00285         }
00286       }
00287       break;
00288     }

00324 - (NSDictionary *)mergeCompositeRulesets:(NSMutableArray *)compositeRulesets dependencyFilenames:(NSSet *)dependencyFilenames {
00325   NIDASSERT([compositeRulesets count] > 0);
00326   if ([compositeRulesets count] == 0) {
00327     return nil;
00328   }
00329 
00330   NSDictionary* result = nil;
00331 
00332   if ([compositeRulesets count] == 1) {
00333     if ([dependencyFilenames count] > 0) {
00334       [[compositeRulesets objectAtIndex:0] setObject:dependencyFilenames
00335                                               forKey:kDependenciesSelectorKey];
00336     }
00337     result = [[[compositeRulesets objectAtIndex:0] copy] autorelease];
00338 
00339   } else {
00340     NSMutableDictionary* mergedResult = [[[compositeRulesets lastObject] retain] autorelease];
00341     [compositeRulesets removeLastObject];
00342 
00343     // Merge all of the rulesets into one.
00344     for (NSDictionary* ruleSet in [compositeRulesets reverseObjectEnumerator]) {
00345       for (NSString* scope in ruleSet) {
00346         if (![mergedResult objectForKey:scope]) {
00347           // New scope means we can just add it.
00348           [mergedResult setObject:[ruleSet objectForKey:scope] forKey:scope];
00349 
00350         } else {
00351           // Existing scope means we need to overwrite existing properties.
00352           NSMutableDictionary* mergedScopeProperties = [mergedResult objectForKey:scope];
00353           NSMutableDictionary* properties = [ruleSet objectForKey:scope];
00354 
00355           for (NSString* propertyName in properties) {
00356 
00357             if (!(nil != [mergedScopeProperties objectForKey:propertyName]
00358                   && [propertyName isEqualToString:kPropertyOrderKey])) {
00359               // Overwrite the existing property's value.
00360               [mergedScopeProperties setObject:[properties objectForKey:propertyName]
00361                                         forKey:propertyName];
00362 
00363             } else {
00364               // Append the property order.
00365               [[mergedScopeProperties objectForKey:kPropertyOrderKey] addObjectsFromArray:
00366                [properties objectForKey:kPropertyOrderKey]];
00367             }
00368           }
00369         }
00370       }
00371     }
00372     
00373     if ([dependencyFilenames count] > 0) {
00374       [mergedResult setObject:dependencyFilenames forKey:kDependenciesSelectorKey];
00375     }
00376     result = [[mergedResult copy] autorelease];
00377   }
00378 
00379   return result;

00402 - (NSDictionary *)dictionaryForPath:(NSString *)aPath
00403                          pathPrefix:(NSString *)pathPrefix
00404                            delegate:(id<NICSSParserDelegate>)delegate {
00405   // Bail out early if there was no path given.
00406   if ([aPath length] == 0) {
00407     _didFailToParse = YES;
00408     return nil;
00409   }
00410 
00411   _didFailToParse = NO;
00412 
00413   NSMutableArray* compositeRulesets = [[[NSMutableArray alloc] init] autorelease];
00414 
00415   // Maintain a set of filenames that we've looked at for two reasons:
00416   // 1) To avoid visiting the same CSS file twice.
00417   // 2) To collect a list of dependencies for this stylesheet.
00418   NSMutableSet* processedFilenames = [[[NSMutableSet alloc] init] autorelease];
00419 
00420   // Imported CSS files will be added to the queue.
00421   NILinkedList* filenameQueue = [NILinkedList linkedList];
00422   [filenameQueue addObject:aPath];
00423 
00424   while ([filenameQueue count] > 0) {
00425     // Om nom nom
00426     NSString* path = [filenameQueue firstObject];
00427     [filenameQueue removeFirstObject];
00428 
00429     // Skip files that we've already processed in order to avoid infinite loops.
00430     if ([processedFilenames containsObject:path]) {
00431       continue;
00432     }
00433     [processedFilenames addObject:path];
00434 
00435     // Allow the delegate to rename the file.
00436     if ([delegate respondsToSelector:@selector(cssParser:pathFromPath:)]) {
00437       NSString* reprocessedFilename = [delegate cssParser:self pathFromPath:path];
00438       if (nil != reprocessedFilename) {
00439         path = reprocessedFilename;
00440       }
00441     }
00442 
00443     // Add the prefix, if it exists.
00444     if (pathPrefix.length > 0) {
00445       path = [pathPrefix stringByAppendingPathComponent:path];
00446     }
00447 
00448     // Verify that the file exists.
00449     if (![[NSFileManager defaultManager] fileExistsAtPath:path]) {
00450       [self shutdown];
00451       return nil;
00452     }
00453 
00454     [self setup];
00455     [self parseFileAtPath:path];
00456     if (self.didFailToParse) {
00457       break;
00458     }
00459 
00460     [filenameQueue addObjectsFromArray:_importedFilenames];
00461     [_importedFilenames removeAllObjects];
00462 
00463     [compositeRulesets addObject:_rulesets];
00464     [self shutdown];
00465   }
00466 
00467   NSDictionary* result = nil;
00468 
00469   if (!self.didFailToParse) {
00470     // processedFilenames will be the set of dependencies, so remove the initial path.
00471     [processedFilenames removeObject:aPath];
00472 
00473     result = [self mergeCompositeRulesets:compositeRulesets
00474                       dependencyFilenames:processedFilenames];
00475 
00476   }
00477 
00478   [self shutdown];
00479 
00480   return result;
00481 }
