132 -(void)attach:(id)element withSelector:(SEL)selector withControlState: (UIControlState) state hasControlState: (BOOL) hasControlState
133 {
134  NIUserInterfaceStringAttachment *attachment = [[NIUserInterfaceStringAttachment alloc] init];
135  attachment.element = element;
136  attachment.controlState = state;
137  attachment.setterIsWithControlState = hasControlState;
138  attachment.setter = selector;
139 
140  // If we're keeping track of attachments, set all that up. Else just call the selector
141  if ([[NIUserInterfaceString stringResolver] isChangeTrackingEnabled]) {
142  NSMutableDictionary *viewMap = self.viewMap;
143  @synchronized (viewMap) {
144  // Call this first, because if there's an existing association, it will detach it in dealloc
145  [NIUserInterfaceStringDeallocTracker attachString:self withInfo:attachment];
146  id existing = [viewMap objectForKey:_originalKey];
147  if (!existing) {
148  // Simple, no map exists, make one
149  [viewMap setObject:attachment forKey:_originalKey];
150  } else if ([existing isKindOfClass: [NIUserInterfaceStringAttachment class]]) {
151  // An attachment exists, convert it to a list
152  NSMutableArray *list = [[NSMutableArray alloc] initWithCapacity:2];
153  [list addObject:existing];
154  [list addObject:attachment];
155  [viewMap setObject:list forKey:_originalKey];
156  } else {
157  // NSMutableArray*
158  NSMutableArray *a = (NSMutableArray*) existing;
159  [a addObject: attachment];
160  }
161  }
162  }
163  [attachment attach: _string];
164 }

217 -(void)stringsDidChange: (NSNotification*) notification
218 {
219  NSString *path = [notification.userInfo objectForKey:NIStringsDidChangeFilePathKey];
220  NSString *content = [[NSString alloc] initWithContentsOfFile:path encoding:NSUTF8StringEncoding error:nil];
221 
222  if ([content hasPrefix:@"/* SHOW KEYS */"]) {
223  self.returnKeys = YES;
224  } else {
225  self.returnKeys = NO;
226  }
227  self.overrides = [[NSDictionary alloc] initWithContentsOfFile:path];
228  if (sStringToViewMap && self.overrides.count > 0) {
229  @synchronized (sStringToViewMap) {
230  [sStringToViewMap enumerateKeysAndObjectsUsingBlock:^(NSString* key, id obj, BOOL *stop) {
231  NSString *o = self.returnKeys ? key : [self.overrides objectForKey:key];
232  if (o) {
233  if ([obj isKindOfClass:[NIUserInterfaceStringAttachment class]]) {
234  [((NIUserInterfaceStringAttachment*)obj) attach: o];
235  } else {
236  NSArray *attachments = (NSArray*) obj;
237  for (NIUserInterfaceStringAttachment *a in attachments) {
238  [a attach:o];
239  }
240  }
241  }
242  }];
243  }
244  }
245 }

273 -(void)attach: (NSString*) value
274 {
275  if (self.setterIsWithControlState) {
276  NSInvocation *inv = [NSInvocation invocationWithMethodSignature:[_element methodSignatureForSelector:_setter]];
277  [inv setSelector:_setter];
278  [inv setTarget:_element];
279  [inv setArgument:&value atIndex:2]; //this is the string to set (0 and 1 are self and message respectively)
280  [inv setArgument:&_controlState atIndex:3];
281  [inv invoke];
282  } else {
283 #pragma clang diagnostic push
284 #pragma clang diagnostic ignored "-Warc-performSelector-leaks"
285  [_element performSelector:_setter withObject:value];
286 #pragma clang diagnostic pop
287  }
288 }

292 @implementation NIUserInterfaceStringDeallocTracker
293 +(void)attachString:(NIUserInterfaceString *)string withInfo:(NIUserInterfaceStringAttachment *)attachment
294 {
295  NIUserInterfaceStringDeallocTracker *tracker = [[NIUserInterfaceStringDeallocTracker alloc] init];
296  tracker.attachment = attachment;
297  tracker.string = string;
298  char* key = &sBaseStringAssocationKey;
299  if (attachment.setterIsWithControlState) {
300  key += attachment.controlState;
301  }
302  objc_setAssociatedObject(attachment.element, key, tracker, OBJC_ASSOCIATION_RETAIN_NONATOMIC);
303 }
