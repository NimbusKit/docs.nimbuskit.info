93 - (void)didTapShareButton {
94  // Dismiss the action menu if the user taps the action button again on the iPad.
95  if ([self.actionSheet isVisible]) {
96  // It shouldn't be possible to tap the share action button again on anything but the iPad.
97  NIDASSERT(NIIsPad());
98 
99  [self.actionSheet dismissWithClickedButtonIndex:[self.actionSheet cancelButtonIndex] animated:YES];
100 
101  // We remove the action sheet here just in case the delegate isn't properly implemented.
102  self.actionSheet.delegate = nil;
103  self.actionSheet = nil;
104  self.actionSheetURL = nil;
105 
106  // Don't show the menu again.
107  return;
108  }
109 
110  // Remember the URL at this point
111  self.actionSheetURL = [self.URL copy];
112 
113  if (nil == self.actionSheet) {
114  self.actionSheet =
115  [[UIActionSheet alloc] initWithTitle:[self.actionSheetURL absoluteString]
116  delegate:self
117  cancelButtonTitle:nil
118  destructiveButtonTitle:nil
119  otherButtonTitles:nil];
120 
121  // Let -shouldPresentActionSheet: setup the action sheet
122  if (![self shouldPresentActionSheet:self.actionSheet] || self.actionSheet.numberOfButtons == 0) {
123  // A subclass decided to handle the action in another way
124  self.actionSheet = nil;
125  self.actionSheetURL = nil;
126  return;
127  }
128  // Add "Cancel" button except for iPads
129  if (!NIIsPad()) {
130  [self.actionSheet setCancelButtonIndex:[self.actionSheet addButtonWithTitle:NSLocalizedString(@"Cancel", @"")]];
131  }
132  }
133 
134  if (NIIsPad()) {
135  [self.actionSheet showFromBarButtonItem:self.actionButton animated:YES];
136  } else {
137  [self.actionSheet showInView:self.view];
138  }
139 }

141 - (void)updateToolbarWithOrientation:(UIInterfaceOrientation)interfaceOrientation {
142  if (!self.toolbarHidden) {
143  CGRect toolbarFrame = self.toolbar.frame;
144  toolbarFrame.size.height = NIToolbarHeightForOrientation(interfaceOrientation);
145  toolbarFrame.origin.y = self.view.bounds.size.height - toolbarFrame.size.height;
146  self.toolbar.frame = toolbarFrame;
147 
148  CGRect webViewFrame = self.webView.frame;
149  webViewFrame.size.height = self.view.bounds.size.height - toolbarFrame.size.height;
150  self.webView.frame = webViewFrame;
151 
152  } else {
153  self.webView.frame = self.view.bounds;
154  }
155 }

169 - (void)loadView {
170  [super loadView];
171 
172  CGRect bounds = self.view.bounds;
173 
174  CGFloat toolbarHeight = NIToolbarHeightForOrientation(NIInterfaceOrientation());
175  CGRect toolbarFrame = CGRectMake(0, bounds.size.height - toolbarHeight,
176  bounds.size.width, toolbarHeight);
177 
178  self.toolbar = [[UIToolbar alloc] initWithFrame:toolbarFrame];
179  self.toolbar.autoresizingMask = (UIViewAutoresizingFlexibleTopMargin
180  | UIViewAutoresizingFlexibleWidth);
181  self.toolbar.tintColor = self.toolbarTintColor;
182  self.toolbar.hidden = self.toolbarHidden;
183 
184  UIActivityIndicatorView* spinner =
185  [[UIActivityIndicatorView alloc] initWithActivityIndicatorStyle:
186  UIActivityIndicatorViewStyleWhite];
187  [spinner startAnimating];
188  self.activityItem = [[UIBarButtonItem alloc] initWithCustomView:spinner];
189 
190  UIImage* backIcon = [UIImage imageWithContentsOfFile:
191  NIPathForBundleResource(nil, @"NimbusWebController.bundle/gfx/backIcon.png")];
192  // We weren't able to find the forward or back icons in your application's resources.
193  // Ensure that you've dragged the NimbusWebController.bundle from src/webcontroller/resources
194  //into your application with the "Create Folder References" option selected. You can verify that
195  // you've done this correctly by expanding the NimbusPhotos.bundle file in your project
196  // and verifying that the 'gfx' directory is blue. Also verify that the bundle is being
197  // copied in the Copy Bundle Resources phase.
198  NIDASSERT(nil != backIcon);
199 
200  self.backButton =
201  [[UIBarButtonItem alloc] initWithImage:backIcon
202  style:UIBarButtonItemStylePlain
203  target:self
204  action:@selector(didTapBackButton)];
205  self.backButton.tag = 2;
206  self.backButton.enabled = NO;
207 
208  UIImage* forwardIcon = [UIImage imageWithContentsOfFile:
209  NIPathForBundleResource(nil, @"NimbusWebController.bundle/gfx/forwardIcon.png")];
210  // We weren't able to find the forward or back icons in your application's resources.
211  // Ensure that you've dragged the NimbusWebController.bundle from src/webcontroller/resources
212  // into your application with the "Create Folder References" option selected. You can verify that
213  // you've done this correctly by expanding the NimbusPhotos.bundle file in your project
214  // and verifying that the 'gfx' directory is blue. Also verify that the bundle is being
215  // copied in the Copy Bundle Resources phase.
216  NIDASSERT(nil != forwardIcon);
217 
218  self.forwardButton =
219  [[UIBarButtonItem alloc] initWithImage:forwardIcon
220  style:UIBarButtonItemStylePlain
221  target:self
222  action:@selector(didTapForwardButton)];
223  self.forwardButton.tag = 1;
224  self.forwardButton.enabled = NO;
225  self.refreshButton = [[UIBarButtonItem alloc] initWithBarButtonSystemItem:
226  UIBarButtonSystemItemRefresh target:self action:@selector(didTapRefreshButton)];
227  self.refreshButton.tag = 3;
228  self.stopButton = [[UIBarButtonItem alloc] initWithBarButtonSystemItem:
229  UIBarButtonSystemItemStop target:self action:@selector(didTapStopButton)];
230  self.stopButton.tag = 3;
231  self.actionButton = [[UIBarButtonItem alloc] initWithBarButtonSystemItem:
232  UIBarButtonSystemItemAction target:self action:@selector(didTapShareButton)];
233 
234  UIBarItem* flexibleSpace =
235  [[UIBarButtonItem alloc] initWithBarButtonSystemItem: UIBarButtonSystemItemFlexibleSpace
236  target: nil
237  action: nil];
238 
239  self.toolbar.items = [NSArray arrayWithObjects:
240  self.backButton,
241  flexibleSpace,
242  self.forwardButton,
243  flexibleSpace,
244  self.refreshButton,
245  flexibleSpace,
246  self.actionButton,
247  nil];
248 
249  self.webView = [[UIWebView alloc] initWithFrame:CGRectZero];
250  [self updateWebViewFrame];
251  self.webView.delegate = self;
252  self.webView.autoresizingMask = (UIViewAutoresizingFlexibleWidth
253  | UIViewAutoresizingFlexibleHeight);
254  self.webView.scalesPageToFit = YES;
255 
256  if ([UIColor respondsToSelector:@selector(underPageBackgroundColor)]) {
257  self.webView.backgroundColor = [UIColor underPageBackgroundColor];
258  }
259 
260  [self.view addSubview:self.webView];
261  [self.view addSubview:self.toolbar];
262 
263  if (nil != self.loadRequest) {
264  [self.webView loadRequest:self.loadRequest];
265  }
266 }

302 - (void)webViewDidStartLoad:(UIWebView*)webView {
303  self.title = NSLocalizedString(@"Loading...", @"");
304  if (!self.navigationItem.rightBarButtonItem) {
305  [self.navigationItem setRightBarButtonItem:self.activityItem animated:YES];
306  }
307 
308  NSInteger buttonIndex = 0;
309  for (UIBarButtonItem* button in self.toolbar.items) {
310  if (button.tag == 3) {
311  NSMutableArray* newItems = [NSMutableArray arrayWithArray:self.toolbar.items];
312  [newItems replaceObjectAtIndex:buttonIndex withObject:self.stopButton];
313  self.toolbar.items = newItems;
314  break;
315  }
316  ++buttonIndex;
317  }
318  self.backButton.enabled = [self.webView canGoBack];
319  self.forwardButton.enabled = [self.webView canGoForward];
320 }
321 
322 - (void)webViewDidFinishLoad:(UIWebView*)webView {
323  self.loadingURL = nil;
324  self.title = [self.webView stringByEvaluatingJavaScriptFromString:@"document.title"];
325  if (self.navigationItem.rightBarButtonItem == self.activityItem) {
326  [self.navigationItem setRightBarButtonItem:nil animated:YES];
327  }
328 
329  NSInteger buttonIndex = 0;
330  for (UIBarButtonItem* button in self.toolbar.items) {
331  if (button.tag == 3) {
332  NSMutableArray* newItems = [NSMutableArray arrayWithArray:self.toolbar.items];
333  [newItems replaceObjectAtIndex:buttonIndex withObject:self.refreshButton];
334  self.toolbar.items = newItems;
335  break;
336  }
337  ++buttonIndex;
338  }
339 
340  self.backButton.enabled = [self.webView canGoBack];
341  self.forwardButton.enabled = [self.webView canGoForward];
342 }
