00268 - (void)layoutPages {
00269   if (nil == _scrollView || CGRectIsEmpty(_scrollView.frame)) {
00270     // Bail out early; the scroll view hasn't been laid out yet.
00271     return;
00272   }
00273 
00274   CGSize buttonDimensions = CGSizeZero;
00275   NSInteger numberOfRows = 0;
00276   NSInteger numberOfColumns = 0;
00277   CGFloat buttonHorizontalSpacing = 0;
00278   CGFloat buttonVerticalSpacing = 0;
00279   [self calculateLayoutForFrame: _scrollView.frame
00280                buttonDimensions: &buttonDimensions
00281                    numberOfRows: &numberOfRows
00282                 numberOfColumns: &numberOfColumns
00283         buttonHorizontalSpacing: &buttonHorizontalSpacing
00284           buttonVerticalSpacing: &buttonVerticalSpacing];
00285 
00286   NIDASSERT(numberOfRows > 0);
00287   NIDASSERT(numberOfColumns > 0);
00288 
00289   CGFloat pageWidth = _scrollView.frame.size.width;
00290 
00291   for (NSInteger ixPage = 0; ixPage < [_pagesOfButtons count]; ++ixPage) {
00292     CGFloat pageOffset = ixPage * pageWidth;
00293 
00294     NSArray* page = [_pagesOfButtons objectAtIndex:ixPage];
00295 
00296     CGFloat pageBottom = 0;
00297 
00298     for (NSInteger ixItem = 0; ixItem < [page count]; ++ixItem) {
00299       NSInteger col = ixItem % numberOfColumns;
00300       NSInteger row = ixItem / numberOfColumns;
00301 
00302       UIButton* button = [page objectAtIndex:ixItem];
00303       button.frame = CGRectMake(_padding.left + col * buttonDimensions.width
00304                                 + (col * buttonHorizontalSpacing),
00305                                 _padding.top + row * buttonDimensions.height
00306                                 + (row * buttonVerticalSpacing),
00307                                 buttonDimensions.width, buttonDimensions.height);
00308 
00309       pageBottom = MAX(pageBottom, CGRectGetMaxY(button.frame));
00310     }
00311 
00312     UIScrollView* pageScrollView = [_pagesOfScrollViews objectAtIndex:ixPage];
00313     pageScrollView.frame = CGRectMake(pageOffset, 0, pageWidth, _scrollView.frame.size.height);
00314     pageScrollView.contentSize = CGSizeMake(pageWidth, pageBottom + _padding.bottom);
00315   }

00473                                   [self.dataSource launcherView: self
00474                                           numberOfButtonsInPage: ixPage]);
00475 
00476     NSMutableArray* page = [[[NSMutableArray alloc] initWithCapacity:numberOfItems]
00477                             autorelease];
00478 
00479     UIScrollView* pageScrollView = [[[UIScrollView alloc] init] autorelease];
00480     pageScrollView.indicatorStyle = UIScrollViewIndicatorStyleWhite;
00481 
00482     for (NSInteger ixItem = 0 ; ixItem < numberOfItems; ++ixItem) {
00483       UIButton* item = [self.dataSource launcherView: self
00484                                        buttonForPage: ixPage
00485                                              atIndex: ixItem];

00487                      action: @selector(didTapButton:)
