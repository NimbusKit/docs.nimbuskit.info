54 - (id)initWithStylesheetCache:(NIStylesheetCache *)stylesheetCache host:(NSString *)host {
55  if ((self = [super init])) {
56  // You must provide a stylesheet cache.
57  NIDASSERT(nil != stylesheetCache);
58  _stylesheetCache = stylesheetCache;
59  _stylesheetPaths = [[NSMutableArray alloc] init];
60  _queue = [[NSOperationQueue alloc] init];
61 
62  if ([host hasSuffix:@"/"]) {
63  _host = [host copy];
64 
65  } else if (host) {
66  _host = [[host stringByAppendingString:@"/"] copy];
67  }
68 
69  NSFileManager* fm = [NSFileManager defaultManager];
70  NSDirectoryEnumerator* de = [fm enumeratorAtPath:_stylesheetCache.pathPrefix];
71 
72  NSString* filename;
73  while ((filename = [de nextObject])) {
74  BOOL isFolder = NO;
75  NSString* path = [_stylesheetCache.pathPrefix stringByAppendingPathComponent:filename];
76  if ([fm fileExistsAtPath:path isDirectory:&isFolder]
77  && !isFolder
78  && [[filename pathExtension] isEqualToString:@"css"]) {
79  [_stylesheetPaths addObject:filename];
80  NSString* cachePath = NIPathForDocumentsResource([self pathFromPath:filename]);
81  NSError* error = nil;
82  [fm removeItemAtPath:cachePath error:&error];
83  error = nil;
84  [fm copyItemAtPath:path toPath:cachePath error:&error];
85  NIDASSERT(nil == error);
86  }
87  }
88  }
89  return self;
90 }

92 - (void)downloadStylesheetWithFilename:(NSString *)path {
93  NSURL* url = [NSURL URLWithString:[_host stringByAppendingString:path]];
94  NSMutableURLRequest* request = [NSMutableURLRequest requestWithURL:url];
95 
96  AFHTTPRequestOperation* requestOp = [[AFHTTPRequestOperation alloc] initWithRequest:request];
97 
98  // This stylesheet has the changed stylesheet as a dependency so let's refresh it.[requestOp setCompletionBlockWithSuccess:^(AFHTTPRequestOperation *operation, id responseObject) {
99  NSMutableArray* changedStylesheets = [NSMutableArray array];
100  NSArray* pathParts = [[operation.request.URL absoluteString] pathComponents];
101  NSString* resultPath = [[pathParts subarrayWithRange:NSMakeRange(2, [pathParts count] - 2)]
102  componentsJoinedByString:@"/"];
103  NSString* rootPath = NIPathForDocumentsResource(nil);
104  NSString* hashedPath = [self pathFromPath:resultPath];
105  NSString* diskPath = [rootPath stringByAppendingPathComponent:hashedPath];
106  [responseObject writeToFile:diskPath atomically:YES];
107 
108  NIStylesheet* stylesheet = [_stylesheetCache stylesheetWithPath:resultPath loadFromDisk:NO];
109  if ([stylesheet loadFromPath:resultPath pathPrefix:rootPath delegate:self]) {
110  [changedStylesheets addObject:stylesheet];
111  }
112 
113  for (NSString* iteratingPath in _stylesheetPaths) {
114  stylesheet = [_stylesheetCache stylesheetWithPath:iteratingPath loadFromDisk:NO];
115  if ([stylesheet.dependencies containsObject:resultPath]) {
116 
117  if ([stylesheet loadFromPath:iteratingPath pathPrefix:rootPath delegate:self]) {
118  [changedStylesheets addObject:stylesheet];
119  }
120  }
121  }
122 
123  NSNotificationCenter* nc = [NSNotificationCenter defaultCenter];
124  for (NIStylesheet* changedStylesheet in changedStylesheets) {
125  [nc postNotificationName:NIStylesheetDidChangeNotification
126  object:changedStylesheet
127  userInfo:nil];
128  }
129 
130  } failure:^(AFHTTPRequestOperation *operation, NSError *error) {
131  }];
132  [_queue addOperation:requestOp];
133 }

202 - (void)watchSkinChanges {
203  NSURL* url = [NSURL URLWithString:[_host stringByAppendingString:@"watch"]];
204  NSMutableURLRequest* request = [NSMutableURLRequest requestWithURL:url];
205  request.timeoutInterval = kTimeoutInterval;
206  AFHTTPRequestOperation* requestOp = [[AFHTTPRequestOperation alloc] initWithRequest:request];
207 
208  // Immediately start watching for more skin changes.[requestOp setCompletionBlockWithSuccess:^(AFHTTPRequestOperation *operation, id responseObject) {
209  NSString* stringData = [[NSString alloc] initWithData:responseObject
210  encoding:NSUTF8StringEncoding];
211 
212  NSArray* files = [stringData componentsSeparatedByString:@"\n"];
213  for (NSString* filename in files) {
214  if ([[filename lowercaseString] hasSuffix:@".strings"]) {
215  [self downloadStringsWithFilename: filename];
216  } else if ([[filename lowercaseString] hasSuffix:@".json"]) {
217  [self downloadJSONWithFilename:filename];
218  } else {
219  [self downloadStylesheetWithFilename:filename];
220  }
221  }
222 
223 
224  _retryCount = 0;
225  [self watchSkinChanges];
226 
227  } failure:^(AFHTTPRequestOperation *operation, NSError *error) {
228  if (_retryCount < kMaxNumberOfRetries) {
229  ++_retryCount;
230 
231  dispatch_time_t popTime = dispatch_time(DISPATCH_TIME_NOW, (int64_t)(kRetryInterval * NSEC_PER_MSEC));
232  dispatch_after(popTime, dispatch_get_main_queue(), ^(void){
233  [self watchSkinChanges];
234  });
235  }
236  }];
237 
238  [_queue addOperation:requestOp];
239 }
