44 // Called after a certain amount of time has passed since all network activity has stopped.
45 // By delaying the turnoff of the network activity we avoid "flickering" effects when network
46 // activity is starting and stopping rapidly.
47 + (void)disableNetworkActivity {
48  pthread_mutex_lock(&gMutex);
49  if (nil != gScheduledDelayTimer) {
50  [UIApplication sharedApplication].networkActivityIndicatorVisible = NO;
51  gScheduledDelayTimer = nil;
52  }
53  pthread_mutex_unlock(&gMutex);
54 }
55 
56 @end

59 void NINetworkActivityTaskDidStart(void) {
60  pthread_mutex_lock(&gMutex);
61 
62  BOOL enableNetworkActivityIndicator = (0 == gNetworkTaskCount);
63 
64  ++gNetworkTaskCount;
65  [gScheduledDelayTimer invalidate];
66  gScheduledDelayTimer = nil;
67 
68  if (enableNetworkActivityIndicator) {
69  [UIApplication sharedApplication].networkActivityIndicatorVisible = YES;
70  }
71 
72  pthread_mutex_unlock(&gMutex);
73 }
74 
75 void NINetworkActivityTaskDidFinish(void) {
76  pthread_mutex_lock(&gMutex);
77 
78  --gNetworkTaskCount;
79  // If this asserts, you don't have enough stop requests to match your start requests.
80  NIDASSERT(gNetworkTaskCount >= 0);
81  gNetworkTaskCount = MAX(0, gNetworkTaskCount);
82 
83  if (gNetworkTaskCount == 0) {
84  [gScheduledDelayTimer invalidate];
85  gScheduledDelayTimer = nil;
86 
87  // Ensure that the timer is scheduled on the main loop, otherwise it will die when the thread
88  // dies.
89  dispatch_async(dispatch_get_main_queue(), ^{
90  pthread_mutex_lock(&gMutex);
91  gScheduledDelayTimer = [NSTimer scheduledTimerWithTimeInterval:kDelayBeforeDisablingActivity
92  target:[NINetworkActivity class]
93  selector:@selector(disableNetworkActivity)
94  userInfo:nil
95  repeats:NO];
96  pthread_mutex_unlock(&gMutex);
97  });
98  }
99 
100  pthread_mutex_unlock(&gMutex);
101 }

114 - (void)nimbusDebugSetNetworkActivityIndicatorVisible:(BOOL)visible {
115  // This method will only be used when swizzled, so this will actually call
116  // setNetworkActivityIndicatorVisible:
117  [self nimbusDebugSetNetworkActivityIndicatorVisible:visible];
118 
119  // Sanity check that this method isn't being called directly when debugging isn't enabled.
120  NIDASSERT(gNetworkActivityDebuggingEnabled);
121 
122  // If either of the following assertions fail then you should look at the call stack to
123  // determine what code is erroneously calling setNetworkActivityIndicatorVisible: directly.
124  if (visible) {
125  // The only time we should be enabling the network activity indicator is when the task
126  // count is one.
127  NIDASSERT(1 == gNetworkTaskCount);
128 
129  } else {
130  // The only time we should be disabling the network activity indicator is when the task
131  // count is zero.
132  NIDASSERT(0 == gNetworkTaskCount);
133  }
134 }

136 @end

139 void NISwizzleMethodsForNetworkActivityDebugging(void) {
