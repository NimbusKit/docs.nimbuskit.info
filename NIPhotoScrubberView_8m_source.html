176 - (NSInteger)numberOfVisiblePhotos {
177  CGSize photoSize = [self photoSize];
178  CGFloat spaceBetweenPhotos = [self spaceBetweenPhotos];
179 
180  // Here's where we take into account the container layer's border because we don't want to
181  // display thumbnails on top of the border.
182  CGFloat maxContentWidth = ([self maxContentWidth]
183  - _containerView.layer.borderWidth * 2);
184 
185  NSInteger numberOfPhotosThatFit = (NSInteger)floor((maxContentWidth + spaceBetweenPhotos)
186  / (photoSize.width + spaceBetweenPhotos));
187  return MIN(_numberOfPhotos, numberOfPhotosThatFit);
188 }

190 - (CGRect)frameForSelectionAtIndex:(NSInteger)photoIndex {
191  CGSize photoSize = [self photoSize];
192  CGSize selectionSize = [self selectionSize];
193 
194  CGFloat containerWidth = _containerView.bounds.size.width;
195  // TODO (jverkoey July 21, 2011): I need to figure out why this is necessary.
196  // Basically, when there are a lot of photos it seems like the selection frame
197  // slowly gets offset from the thumbnail frame it's supposed to be representing until by the end
198  // it's off the right edge by a noticeable amount. Trimming off some fat from the right
199  // edge seems to fix this.
200  if (_numberOfVisiblePhotos < _numberOfPhotos) {
201  containerWidth -= photoSize.width / 2;
202  }
203 
204  // Calculate the offset into the container view based on index/numberOfPhotos.
205  CGFloat relativeOffset = floorf((((CGFloat)photoIndex * containerWidth)
206  / (CGFloat)MAX(1, _numberOfPhotos)));
207 
208  return CGRectMake(floorf(_containerView.frame.origin.x
209  + relativeOffset
210  + photoSize.width / 2 - selectionSize.width / 2),
211  floorf(_containerView.center.y - selectionSize.height / 2),
212  selectionSize.width, selectionSize.height);
213 }

215 - (CGRect)frameForThumbAtIndex:(NSInteger)thumbIndex {
216  CGSize photoSize = [self photoSize];
217  CGFloat spaceBetweenPhotos = [self spaceBetweenPhotos];
218  return CGRectMake(_containerView.layer.borderWidth
219  + (photoSize.width + spaceBetweenPhotos) * thumbIndex,
220  _containerView.layer.borderWidth,
221  photoSize.width, photoSize.height);
222 }

224 - (void)layoutSubviews {
225  [super layoutSubviews];
226 
227  CGSize boundsSize = self.bounds.size;
228 
229  CGSize photoSize = [self photoSize];
230  CGFloat spaceBetweenPhotos = [self spaceBetweenPhotos];
231  CGFloat maxContentWidth = [self maxContentWidth];
232 
233  // Update the total number of visible photos.
234  _numberOfVisiblePhotos = [self numberOfVisiblePhotos];
235 
236  // Hide views if there isn't any interesting information to show.
237  _containerView.hidden = (0 == _numberOfVisiblePhotos);
238  _selectionView.hidden = (_selectedPhotoIndex < 0 || _containerView.hidden);
239 
240  // Calculate the container width using the number of visible photos.
241  CGFloat containerWidth = ((_numberOfVisiblePhotos * photoSize.width)
242  + (MAX(0, _numberOfVisiblePhotos - 1) * spaceBetweenPhotos)
243  + _containerView.layer.borderWidth * 2);
244 
245  // Then we center the container in the content area.
246  CGFloat containerMargins = MAX(0, floorf((maxContentWidth - containerWidth) / 2));
247  CGFloat horizontalMargins = [self horizontalMargins];
248  CGFloat containerHeight = photoSize.height + _containerView.layer.borderWidth * 2;
249 
250  CGFloat containerLeftMargin = horizontalMargins + containerMargins;
251  CGFloat containerTopMargin = floorf((boundsSize.height - containerHeight) / 2);
252 
253  _containerView.frame = CGRectMake(containerLeftMargin,
254  containerTopMargin,
255  containerWidth,
256  containerHeight);
257 
258  // Don't bother updating the selected photo index if there isn't a selection; the
259  // selection view will be hidden anyway.
260  if (_selectedPhotoIndex >= 0) {
261  _selectionView.frame = [self frameForSelectionAtIndex:_selectedPhotoIndex];
262  }
263 
264  // Update the frames for all of the thumbnails.
265  [self updateVisiblePhotos];
266 }

268 // Transforms an index into the number of visible photos into an index into the total
269 // number of photos.
270 - (NSInteger)photoIndexAtScrubberIndex:(NSInteger)scrubberIndex {
271  return (NSInteger)(ceilf((CGFloat)(scrubberIndex * _numberOfPhotos)
272  / (CGFloat)_numberOfVisiblePhotos)
273  + 0.5f);
274 }

276 - (void)updateVisiblePhotos {
277  if (nil == self.dataSource) {
278  return;
279  }
280 
281  // This will update the number of visible photos if the layout did indeed change.
282  [self layoutIfNeeded];
283 
284  // Recycle any views that we no longer need.
285  while ([_visiblePhotoViews count] > (NSUInteger)_numberOfVisiblePhotos) {
286  UIView* photoView = [_visiblePhotoViews lastObject];
287  [photoView removeFromSuperview];
288 
289  [_recycledPhotoViews addObject:photoView];
290 
291  [_visiblePhotoViews removeLastObject];
292  }
293 
294  // Lay out the visible photos.
295  for (NSUInteger ix = 0; ix < (NSUInteger)_numberOfVisiblePhotos; ++ix) {
296  UIImageView* photoView = nil;
297 
298  // We must first get the photo view at this index.
299 
300  // If there aren't enough visible photo views then try to recycle another view.
301  if (ix >= [_visiblePhotoViews count]) {
302  photoView = [_recycledPhotoViews anyObject];
303  if (nil == photoView) {
304  // Couldn't recycle the view, so create a new one.
305  photoView = [self photoView];
306 
307  } else {
308  [_recycledPhotoViews removeObject:photoView];
309  }
310  [_containerView addSubview:photoView];
311  [_visiblePhotoViews addObject:photoView];
312 
313  } else {
314  photoView = [_visiblePhotoViews objectAtIndex:ix];
315  }
316 
317  NSInteger photoIndex = [self photoIndexAtScrubberIndex:ix];
318 
319  // Only request the thumbnail if this thumbnail's photo index has changed. Otherwise
320  // we assume that this photo either already has the thumbnail or it's still loading.
321  if (photoView.tag != photoIndex) {
322  photoView.tag = photoIndex;
323 
324  UIImage* image = [self.dataSource photoScrubberView:self thumbnailAtIndex:photoIndex];
325  photoView.image = image;
326 
327  if (_selectedPhotoIndex == photoIndex) {
328  _selectionView.image = image;
329  }
330  }
331 
332  photoView.frame = [self frameForThumbAtIndex:ix];
333  }
334 }

339 - (NSInteger)photoIndexAtPoint:(CGPoint)point {
340  NSInteger photoIndex;
341 
342  if (point.x <= 0) {
343  // Beyond the left edge
344  photoIndex = 0;
345 
346  } else if (point.x >= _containerView.bounds.size.width) {
347  // Beyond the right edge
348  photoIndex = (_numberOfPhotos - 1);
349 
350  } else {
351  // Somewhere in between
352  photoIndex = (NSInteger)(floorf((point.x / _containerView.bounds.size.width) * _numberOfPhotos)
353  + 0.5f);
354  }
355 
356  return photoIndex;
357 }

395 - (void)didLoadThumbnail: (UIImage *)image
396  atIndex: (NSInteger)photoIndex {
397  for (UIImageView* thumbView in _visiblePhotoViews) {
398  if (thumbView.tag == photoIndex) {
399  thumbView.image = image;
400  break;
401  }
402  }
403 
404  // Update the selected thumbnail if it's the one that just received a photo.
405  if (_selectedPhotoIndex == photoIndex) {
406  _selectionView.image = image;
407  }

410 - (void)reloadData {
411  NIDASSERT(nil != _dataSource);
412 
413  // Remove any visible photos from the view before we release the sets.
414  for (UIView* photoView in _visiblePhotoViews) {
415  [photoView removeFromSuperview];
416  }
417 
418  // If there is no data source then we can't do anything particularly interesting.
419  if (nil == _dataSource) {
420  return;
421  }
422 
423  _visiblePhotoViews = [[NSMutableArray alloc] init];
424  _recycledPhotoViews = [[NSMutableSet alloc] init];
425 
426  // Cache the number of photos.
427  _numberOfPhotos = [_dataSource numberOfPhotosInScrubberView:self];
428 
429  [self setNeedsLayout];
430 
431  // This will call layoutIfNeeded and layoutSubviews will then be called because we
432  // set the needsLayout flag.
433  [self updateVisiblePhotos];
434 }

436 - (void)setSelectedPhotoIndex:(NSInteger)photoIndex animated:(BOOL)animated {
437  if (_selectedPhotoIndex != photoIndex) {
438  // Don't animate the selection if it was previously invalid.
439  animated = animated && (_selectedPhotoIndex >= 0);
440 
441  _selectedPhotoIndex = photoIndex;
442 
443  if (animated) {
444  [UIView beginAnimations:nil context:nil];
445  [UIView setAnimationDuration:0.2];
446  [UIView setAnimationCurve:UIViewAnimationCurveEaseOut];
447  [UIView setAnimationBeginsFromCurrentState:YES];
448  }
449 
450  _selectionView.frame = [self frameForSelectionAtIndex:_selectedPhotoIndex];
451 
452  if (animated) {
453  [UIView commitAnimations];
454  }
455 
456  _selectionView.image = [self.dataSource photoScrubberView: self
457  thumbnailAtIndex: _selectedPhotoIndex];
458  }
459 }
