00028 // From: http://developer.apple.com/mac/library/qa/qa2004/qa1361.html
00029 int NIIsInDebugger(void) {
00030   int                 mib[4];
00031   struct kinfo_proc   info;
00032   size_t              size;
00033 
00034   // Initialize the flags so that, if sysctl fails for some bizarre
00035   // reason, we get a predictable result.
00036 
00037   info.kp_proc.p_flag = 0;
00038 
00039   // Initialize mib, which tells sysctl the info we want, in this case
00040   // we're looking for information about a specific process ID.
00041 
00042   mib[0] = CTL_KERN;
00043   mib[1] = KERN_PROC;
00044   mib[2] = KERN_PROC_PID;
00045   mib[3] = getpid();
00046 
00047   // Call sysctl.
00048 
00049   size = sizeof(info);
00050   sysctl(mib, sizeof(mib) / sizeof(*mib), &info, &size, NULL, 0);
00051 
00052   // We're being debugged if the P_TRACED flag is set.
00053 
00054   return (info.kp_proc.p_flag & P_TRACED) != 0;
00055 }
