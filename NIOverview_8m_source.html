68 void NIOverviewLogMethod(const char* message, unsigned length, BOOL withSyslogBanner) {
69  static NSDateFormatter* formatter = nil;
70  if (nil == formatter) {
71  formatter = [[NSDateFormatter alloc] init];
72  [formatter setTimeStyle:NSDateFormatterMediumStyle];
73  [formatter setDateStyle:NSDateFormatterMediumStyle];
74  }
75 
76  // Don't autorelease here in an attempt to minimize autorelease thrashing in tight
77  // loops.
78 
79  NSString* formattedLogMessage = [[NSString alloc] initWithCString: message
80  encoding: NSUTF8StringEncoding];
81 
82  dispatch_async(dispatch_get_main_queue(), ^{
83  NIOverviewConsoleLogEntry* entry = [[NIOverviewConsoleLogEntry alloc]
84  initWithLog:formattedLogMessage];
85 
86  [[NIOverview logger] addConsoleLog:entry];
87  });
88 
89  formattedLogMessage = [[NSString alloc] initWithFormat:
90  @"%@: %s\n", [formatter stringFromDate:[NSDate date]], message];
91 
92  fprintf(stderr, "%s", [formattedLogMessage UTF8String]);
93 }

104 + (void)didChangeOrientation {
105  static UIDeviceOrientation lastOrientation = UIDeviceOrientationUnknown;
106 
107  // Don't animate the overview if the device didn't actually change orientations.
108  if (lastOrientation != [[UIDevice currentDevice] orientation]
109  && [[UIDevice currentDevice] orientation] != UIDeviceOrientationUnknown
110  && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceUp
111  && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceDown) {
112 
113  // When we flip from landscape to landscape or portait to portait the animation lasts
114  // twice as long.
115  UIDeviceOrientation currentOrientation = [[UIDevice currentDevice] orientation];
116  BOOL isLongAnimation = (UIDeviceOrientationIsLandscape(lastOrientation)
117  == UIDeviceOrientationIsLandscape(currentOrientation));
118  lastOrientation = currentOrientation;
119 
120  // Hide the overview right away, we'll make it fade back in when the rotation is
121  // finished.
122  sOverviewView.hidden = YES;
123 
124  // Delay showing the overview again until the rotation finishes.
125  [self cancelPreviousPerformRequestsWithTarget:self];
126  [self performSelector:@selector(showoverviewAfterRotation) withObject:nil
127  afterDelay:NIDeviceRotationDuration(isLongAnimation)];
128  }
129 }

140 + (void)showoverviewAfterRotation {
141  // Don't modify the overview's frame directly, just modify the transform/center/bounds
142  // properties so that the view is rotated with the device.
143 
144  sOverviewView.transform = NIRotateTransformForOrientation(NIInterfaceOrientation());
145 
146  // Fetch the frame only to calculate the center.
147  CGRect frame = [NIOverview frame];
148  sOverviewView.center = CGPointMake(CGRectGetMidX(frame), CGRectGetMidY(frame));
149 
150  CGRect bounds = sOverviewView.bounds;
151  if (UIInterfaceOrientationIsLandscape(NIInterfaceOrientation())) {
152  bounds.size.width = frame.size.height;
153  bounds.size.height = frame.size.width;
154  } else {
155  bounds.size = frame.size;
156  }
157  sOverviewView.bounds = bounds;
158 
159  // Get ready to fade the overview back in.
160  sOverviewView.hidden = NO;
161  sOverviewView.alpha = 0;
162 
163  [sOverviewView flashScrollIndicators];
164 
165  // Fade!
166  [UIView beginAnimations:nil context:nil];
167  [UIView setAnimationDuration:0.2];
168  sOverviewView.alpha = 1;
169  [UIView commitAnimations];
170 }

272 + (CGRect)frame {
273 #if defined(DEBUG) || defined(NI_DEBUG)
274  UIInterfaceOrientation orient = NIInterfaceOrientation();
275  CGFloat overviewWidth;
276  CGRect frame;
277  CGFloat overviewHeight =
278  sOverviewHasOverridenStatusBarHeight ? NIOverviewStatusBarHeight() : NIStatusBarHeight();
279 
280  // We can't take advantage of automatic view positioning because the overview exists
281  // at the topmost view level (even above the root view controller). As such, we have to
282  // calculate the frame depending on the interface orientation.
283  if (orient == UIInterfaceOrientationLandscapeLeft) {
284  overviewWidth = [UIScreen mainScreen].bounds.size.height;
285  frame = CGRectMake(overviewHeight, 0, sOverviewHeight, overviewWidth);
286 
287  } else if (orient == UIInterfaceOrientationLandscapeRight) {
288  overviewWidth = [UIScreen mainScreen].bounds.size.height;
289  frame = CGRectMake([UIScreen mainScreen].bounds.size.width
290  - (overviewHeight + sOverviewHeight), 0, sOverviewHeight, overviewWidth);
291 
292  } else if (orient == UIInterfaceOrientationPortraitUpsideDown) {
293  overviewWidth = [UIScreen mainScreen].bounds.size.width;
294  frame = CGRectMake(0, [UIScreen mainScreen].bounds.size.height
295  - (overviewHeight + sOverviewHeight), overviewWidth, sOverviewHeight);
296 
297  } else if (orient == UIInterfaceOrientationPortrait) {
298  overviewWidth = [UIScreen mainScreen].bounds.size.width;
299  frame = CGRectMake(0, overviewHeight, overviewWidth, sOverviewHeight);
300 
301  } else {
302  overviewWidth = [UIScreen mainScreen].bounds.size.width;
303  frame = CGRectMake(0, overviewHeight, overviewWidth, sOverviewHeight);
304  }
305  if ([[UIApplication sharedApplication] isStatusBarHidden]) {
306  // When the status bar is hidden we want to position the overview offscreen.
307  switch (orient) {
308  case UIInterfaceOrientationLandscapeLeft: {
309  frame = CGRectOffset(frame, -frame.size.width, 0);
310  break;
311  }
312  case UIInterfaceOrientationLandscapeRight: {
313  frame = CGRectOffset(frame, frame.size.width, 0);
314  break;
315  }
316  case UIInterfaceOrientationPortrait: {
317  frame = CGRectOffset(frame, 0, -frame.size.height);
318  break;
319  }
320  case UIInterfaceOrientationPortraitUpsideDown: {
321  frame = CGRectOffset(frame, 0, frame.size.height);
322  break;
323  }
324  }
325  }
326  return frame;
327 
328 #else
329  return CGRectZero;
330 #endif
331 }
