00182 - (void)_compileSectionIndex {
00183   NI_RELEASE_SAFELY(_sectionIndexTitles);
00184 
00185   // Prime the section index and the map
00186   NSMutableArray* titles = nil;
00187   NSMutableDictionary* sectionPrefixToSectionIndex = nil;
00188   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00189     titles = [NSMutableArray array];
00190     sectionPrefixToSectionIndex = [NSMutableDictionary dictionary];
00191 
00192     // The search symbol is always first in the index.
00193     if (_sectionIndexShowsSearch) {
00194       [titles addObject:UITableViewIndexSearch];
00195     }
00196   }
00197 
00198   // A dynamic index shows the first letter of every section in the index in whatever order the
00199   // sections are ordered (this may not be alphabetical).
00200   if (NITableViewModelSectionIndexDynamic == _sectionIndexType) {
00201     for (NITableViewModelSection* section in _sections) {
00202       NSString* headerTitle = section.headerTitle;
00203       if ([headerTitle length] > 0) {
00204         NSString* prefix = [headerTitle substringToIndex:1];
00205         [titles addObject:prefix];
00206       }
00207     }
00208 
00209   } else if (NITableViewModelSectionIndexAlphabetical == _sectionIndexType) {
00210     // Use the localized indexed collation to create the index. In English, this will always be
00211     // the entire alphabet.
00212     NSArray* sectionIndexTitles = [[UILocalizedIndexedCollation currentCollation] sectionIndexTitles];
00213 
00214     // The localized indexed collection sometimes includes a # for summaries, but we might
00215     // not want to show a summary in the index, so prune it out. It's not guaranteed that
00216     // a # will actually be included in the section index titles, so we always attempt to
00217     // remove it for consistency's sake and then add it back down below if it is requested.
00218     for (NSString* letter in sectionIndexTitles) {
00219       if (![letter isEqualToString:@"#"]) {
00220         [titles addObject:letter];
00221       }
00222     }
00223   }
00224 
00225   // Add the section summary symbol if it was requested.
00226   if (_sectionIndexShowsSummary) {
00227     [titles addObject:@"#"];
00228   }
00229 
00230   // Build the prefix => section index map.
00231   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00232 
00233     // Map all of the sections to indices.
00234     NSInteger sectionIndex = 0;
00235     for (NITableViewModelSection* section in _sections) {
00236       NSString* headerTitle = section.headerTitle;
00237       if ([headerTitle length] > 0) {
00238         NSString* prefix = [headerTitle substringToIndex:1];
00239         if (nil == [sectionPrefixToSectionIndex objectForKey:prefix]) {
00240           [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:sectionIndex] forKey:prefix];
00241         }
00242       }
00243       ++sectionIndex;
00244     }
00245 
00246     // Map the unmapped section titles to the next closest earlier section.
00247     NSInteger lastIndex = 0;
00248     for (NSString* title in titles) {
00249       NSString* prefix = [title substringToIndex:1];
00250       if (nil != [sectionPrefixToSectionIndex objectForKey:prefix]) {
00251         lastIndex = [[sectionPrefixToSectionIndex objectForKey:prefix] intValue];
00252         
00253       } else {
00254         [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:lastIndex] forKey:prefix];
00255       }
00256     }
00257   }
00258 
00259   self.sectionIndexTitles = titles;
00260   self.sectionPrefixToSectionIndex = sectionPrefixToSectionIndex;
00261 }
