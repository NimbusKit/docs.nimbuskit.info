48 - (BOOL)shouldForwardSelector:(SEL)selector {
49  struct objc_method_description description;
50  description = protocol_getMethodDescription(@protocol(UITableViewDelegate), selector, NO, YES);
51  return (description.name != NULL && description.types != NULL);
52 }
53 
54 - (BOOL)respondsToSelector:(SEL)selector {
55  if ([super respondsToSelector:selector]) {
56  return YES;
57 
58  } else if ([self shouldForwardSelector:selector]) {
59  for (id delegate in self.forwardDelegates) {
60  if ([delegate respondsToSelector:selector]) {
61  return YES;
62  }
63  }
64  }
65  return NO;
66 }
67 
68 - (NSMethodSignature *)methodSignatureForSelector:(SEL)selector {
69  NSMethodSignature *signature = [super methodSignatureForSelector:selector];
70  if (signature == nil) {
71  for (id delegate in self.forwardDelegates) {
72  if ([delegate respondsToSelector:selector]) {
73  signature = [delegate methodSignatureForSelector:selector];
74  }
75  }
76  }
77  return signature;
78 }
79 
80 - (void)forwardInvocation:(NSInvocation *)invocation {
81  BOOL didForward = NO;
82 
83  if ([self shouldForwardSelector:invocation.selector]) {
84  for (id delegate in self.forwardDelegates) {
85  if ([delegate respondsToSelector:invocation.selector]) {
86  [invocation invokeWithTarget:delegate];
87  didForward = YES;
88  break;
89  }
90  }
91  }
92 
93  if (!didForward) {
94  [super forwardInvocation:invocation];
95  }
96 }

110 - (UITableViewCellAccessoryType)accessoryTypeForObject:(id)object {
111  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
112  // Detail disclosure indicator takes precedence over regular disclosure indicator.
113  if (nil != action.detailAction || nil != action.detailSelector) {
114  return UITableViewCellAccessoryDetailDisclosureButton;
115 
116  } else if (nil != action.navigateAction || nil != action.navigateSelector) {
117  return UITableViewCellAccessoryDisclosureIndicator;
118 
119  }
120  // We must maintain consistency of modifications to the accessoryType within this call due
121  // to the fact that cells will be reused.
122  return UITableViewCellAccessoryNone;
123 }
124 
125 - (UITableViewCellSelectionStyle)selectionStyleForObject:(id)object {
126  // If the cell is tappable, reflect that in the selection style.
127  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
128  if (action.navigateAction || action.tapAction
129  || action.navigateSelector || action.tapSelector) {
130  return self.tableViewCellSelectionStyle;
131 
132  }
133  return UITableViewCellSelectionStyleNone;
134 }

139 - (void)tableView:(UITableView *)tableView willDisplayCell:(UITableViewCell *)cell forRowAtIndexPath:(NSIndexPath *)indexPath {
140  NIDASSERT([tableView.dataSource isKindOfClass:[NITableViewModel class]]);
141  if ([tableView.dataSource isKindOfClass:[NITableViewModel class]]) {
142  NITableViewModel* model = (NITableViewModel *)tableView.dataSource;
143  id object = [model objectAtIndexPath:indexPath];
144  if ([self isObjectActionable:object]) {
145  cell.accessoryType = [self accessoryTypeForObject:object];
146  cell.selectionStyle = [self selectionStyleForObject:object];
147  } else {
148  cell.accessoryType = UITableViewCellAccessoryNone;
149  cell.selectionStyle = UITableViewCellSelectionStyleNone;
150  }
151  }
152 
153  // Forward the invocation along.
154  for (id<UITableViewDelegate> delegate in self.forwardDelegates) {
155  if ([delegate respondsToSelector:_cmd]) {
156  [delegate tableView:tableView willDisplayCell:cell forRowAtIndexPath:indexPath];
157  }
158  }
159 }

161 - (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath {
162  NIDASSERT([tableView.dataSource isKindOfClass:[NITableViewModel class]]);
163  if ([tableView.dataSource isKindOfClass:[NITableViewModel class]]) {
164  NITableViewModel* model = (NITableViewModel *)tableView.dataSource;
165  id object = [model objectAtIndexPath:indexPath];
166 
167  if ([self isObjectActionable:object]) {
168  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
169 
170  BOOL shouldDeselect = NO;
171  if (action.tapAction) {
172  // Tap actions can deselect the row if they return YES.
173  shouldDeselect = action.tapAction(object, self.target, indexPath);
174  }
175  if (action.tapSelector && [self.target respondsToSelector:action.tapSelector]) {
176  NSMethodSignature *methodSignature = [self.target methodSignatureForSelector:action.tapSelector];
177  NSInvocation *invocation = [NSInvocation invocationWithMethodSignature:methodSignature];
178  invocation.selector = action.tapSelector;
179  if (methodSignature.numberOfArguments >= 3) {
180  [invocation setArgument:&object atIndex:2];
181  }
182  if (methodSignature.numberOfArguments >= 4) {
183  [invocation setArgument:&indexPath atIndex:3];
184  }
185  [invocation invokeWithTarget:self.target];
186 
187  NSUInteger length = invocation.methodSignature.methodReturnLength;
188  if (length > 0) {
189  char *buffer = (void *)malloc(length);
190  memset(buffer, 0, sizeof(char) * length);
191  [invocation getReturnValue:buffer];
192  for (NSUInteger index = 0; index < length; ++index) {
193  if (buffer[index]) {
194  shouldDeselect = YES;
195  break;
196  }
197  }
198  free(buffer);
199  }
200  }
201  if (shouldDeselect) {
202  [tableView deselectRowAtIndexPath:indexPath animated:YES];
203  }
204 
205  if (action.navigateAction) {
206  action.navigateAction(object, self.target, indexPath);
207  }
208  if (action.navigateSelector && [self.target respondsToSelector:action.navigateSelector]) {
209 #pragma clang diagnostic push
210 #pragma clang diagnostic ignored "-Warc-performSelector-leaks"
211  [self.target performSelector:action.navigateSelector withObject:object withObject:indexPath];
212 #pragma clang diagnostic pop
213  }
214  }
215  }
216 
217  // Forward the invocation along.
218  for (id<UITableViewDelegate> delegate in self.forwardDelegates) {
219  if ([delegate respondsToSelector:_cmd]) {
220  [delegate tableView:tableView didSelectRowAtIndexPath:indexPath];
221  }
222  }
223 }

225 - (void)tableView:(UITableView *)tableView accessoryButtonTappedForRowWithIndexPath:(NSIndexPath *)indexPath {
226  NIDASSERT([tableView.dataSource isKindOfClass:[NITableViewModel class]]);
227  if ([tableView.dataSource isKindOfClass:[NITableViewModel class]]) {
228  NITableViewModel* model = (NITableViewModel *)tableView.dataSource;
229  id object = [model objectAtIndexPath:indexPath];
230 
231  if ([self isObjectActionable:object]) {
232  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
233 
234  if (action.detailAction) {
235  action.detailAction(object, self.target, indexPath);
236  }
237  if (action.detailSelector && [self.target respondsToSelector:action.detailSelector]) {
238 #pragma clang diagnostic push
239 #pragma clang diagnostic ignored "-Warc-performSelector-leaks"
240  [self.target performSelector:action.detailSelector withObject:object withObject:indexPath];
241 #pragma clang diagnostic pop
242  }
243  }
244  }
245 
246  // Forward the invocation along.
247  for (id<UITableViewDelegate> delegate in self.forwardDelegates) {
248  if ([delegate respondsToSelector:_cmd]) {
