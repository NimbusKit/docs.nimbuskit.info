00081 - (void)refreshModel {
00082   NSMutableArray* contents = [NSMutableArray array];
00083   NSString* summary = nil;
00084 
00085   // Display a summary.
00086   if ([self.cache isKindOfClass:[NIImageMemoryCache class]]) {
00087     NIImageMemoryCache* imageCache = (NIImageMemoryCache *)self.cache;
00088     summary = [NSString stringWithFormat:
00089                @"Number of images: %d\nNumber of pixels: %@/%@\nStress limit: %@",
00090                self.cache.count,
00091                NIStringFromBytes(imageCache.numberOfPixels),
00092                NIStringFromBytes(imageCache.maxNumberOfPixels),
00093                NIStringFromBytes(imageCache.maxNumberOfPixelsUnderStress)];
00094 
00095   } else {
00096     summary = [NSString stringWithFormat:
00097                @"Number of objects: %d",
00098                self.cache.count];
00099   }
00100   [contents addObject:[NITableViewModelFooter footerWithTitle:summary]];
00101 
00102   NSDateFormatter* formatter = [[[NSDateFormatter alloc] init] autorelease];
00103   // We care more about time than date here.
00104   [formatter setDateStyle:NSDateFormatterShortStyle];
00105   [formatter setTimeStyle:NSDateFormatterMediumStyle];
00106 
00107   // Add each of the cache objects to the model.
00108   for (id cacheObject in self.cache.lruCacheObjects) {
00109     NSString* name = nil;
00110     UIImage* image = nil;
00111     NSDate* lastAccessTime = nil;
00112 
00113     // We're accessing a private object from the core here, so we can't assume that any of
00114     // these selectors will be around forever. That being said, we should try to remember to
00115     // update this controller if the cache object internals ever change.
00116     // If any of these assertions fire it means we've changed the cache object signatures but
00117     // haven't gotten around to updating this controller yet.
00118     NIDASSERT([cacheObject respondsToSelector:@selector(name)]);
00119     if ([cacheObject respondsToSelector:@selector(name)]) {
00120       name = [cacheObject performSelector:@selector(name)];
00121     }
00122     NIDASSERT([cacheObject respondsToSelector:@selector(object)]);
00123     if ([cacheObject respondsToSelector:@selector(object)]) {
00124       id object = [cacheObject performSelector:@selector(object)];
00125       if ([object isKindOfClass:[UIImage class]]) {
00126         image = object;
00127       }
00128     }
00129     NIDASSERT([cacheObject respondsToSelector:@selector(lastAccessTime)]);
00130     if ([cacheObject respondsToSelector:@selector(lastAccessTime)]) {
00131       lastAccessTime = [cacheObject performSelector:@selector(lastAccessTime)];
00132     }
00133 
00134     [contents addObject:
00135      [NISubtitleCellObject objectWithTitle:name
00136                                   subtitle:[NSString stringWithFormat:
00137                                             @"Last access: %@",
00138                                             [formatter stringFromDate:lastAccessTime]]
00139                                      image:image]];
00140   }
00141 
00142   if (0 == self.cache.count) {
00143     [contents addObject:[NITitleCellObject objectWithTitle:@"No cache objects"]];
00144   }
00145 
00146   [contents addObject:[NITableViewModelFooter footerWithTitle:
00147                        @"The most-recently-used object is here at the bottom"]];
00148 
00149   self.model = [[[NITableViewModel alloc] initWithSectionedArray:contents
00150                                                         delegate:(id)[NICellFactory class]] autorelease];
00151   self.tableView.dataSource = self.model;
00152   [self.tableView reloadData];
00153 }
