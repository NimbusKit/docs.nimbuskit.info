00315 - (void)setChromeVisibility:(BOOL)isVisible animated:(BOOL)animated {
00316   if (_isAnimatingChrome
00317       || (!isVisible && _isChromeHidden)
00318       || (isVisible && !_isChromeHidden)
00319       || !self.chromeCanBeHidden) {
00320     // Nothing to do here.
00321     return;

00325   CGRect bounds = self.view.bounds;
00326 
00327   if (self.toolbarIsTranslucent) {
00328     // Reset the toolbar's initial position.
00329     if (!isVisible) {
00330       toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00331 
00332     } else {
00333       // Ensure that the toolbar is visible through the animation.
00334       self.toolbar.hidden = NO;
00335 
00336       toolbarFrame.origin.y = bounds.size.height;
00337     }
00338     self.toolbar.frame = toolbarFrame;
00339   }
00340 
00341   // Show/hide the system chrome.
00342   if ([[UIApplication sharedApplication] respondsToSelector:
00343        @selector(setStatusBarHidden:withAnimation:)]) {
00344     // On 3.2 and higher we can slide the status bar out.
00345     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00346                                             withAnimation: (animated
00347                                                             ? UIStatusBarAnimationSlide
00348                                                             : UIStatusBarAnimationNone)];
00349 
00350   } else {
00351 #if __IPHONE_OS_VERSION_MIN_REQUIRED < NIIOS_3_2
00352     // On 3.0 devices we use the boring fade animation.
00353     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00354                                                  animated: animated];
00355 #endif
00356   }
00357 
00358   if (self.toolbarIsTranslucent) {
00359     // Place the toolbar at its final location.
00360     if (isVisible) {
00361       // Slide up.
00362       toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00363 
00364     } else {
00365       // Slide down.
00366       toolbarFrame.origin.y = bounds.size.height;
00367     }
00368   }
00369

00371   CGRect navigationBarFrame = CGRectZero;

00431 - (void)didTap {
00432   SEL selector = @selector(toggleChromeVisibility);
00433   if (self.photoAlbumView.zoomingIsEnabled) {
00434     // Cancel any previous delayed performs so that we don't stack them.
00435     [NSObject cancelPreviousPerformRequestsWithTarget:self selector:selector object:nil];
00436 
00437     // We need to delay taking action on the first tap in case a second tap comes in, causing
00438     // a double-tap gesture to be recognized and the photo to be zoomed.
00439     [self performSelector: selector
00440                withObject: nil
00441                afterDelay: 0.3];
00442 
00443   } else {
00444     // When zooming is disabled, double-tap-to-zoom is also disabled so we don't have to
00445     // be as careful; just toggle the chrome immediately.
00446     [self toggleChromeVisibility];
00447   }
