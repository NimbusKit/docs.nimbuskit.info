93 + (void)pushAppearanceForNavigationController:(UINavigationController *)navigationController {
94  if (!sAppearanceStack) {
95  sAppearanceStack = [[NSMutableArray alloc] init];
96  }
97 
98  NINavigationAppearanceSnapshot *snapshot = [[NINavigationAppearanceSnapshot alloc] initForNavigationController:navigationController];
99  [sAppearanceStack addObject:snapshot];
100 }

102 + (void)popAppearanceForNavigationController:(UINavigationController *)navigationController animated:(BOOL)animated {
103  NIDASSERT([sAppearanceStack count] > 0);
104  if ([sAppearanceStack count]) {
105  NINavigationAppearanceSnapshot *snapshot = [sAppearanceStack objectAtIndex:0];
106  [snapshot restoreForNavigationController:navigationController animated:animated];
107  [sAppearanceStack removeObject:snapshot];
108  }
109 
110  if (![sAppearanceStack count]) {
111  sAppearanceStack = nil;
112  }
113 }

130 - (id)initForNavigationController:(UINavigationController *)navigationController {
131  self = [super init];
132  if (self) {
133  _statusBarStyle = [[UIApplication sharedApplication] statusBarStyle];
134  _navBarStyle = navigationController.navigationBar.barStyle;
135  _navBarTranslucent = navigationController.navigationBar.translucent;
136  _navBarTintColor = navigationController.navigationBar.tintColor;
137  if ([navigationController.navigationBar respondsToSelector:@selector(backgroundImageForBarMetrics:)])
138  {
139  _navBarDefaultImage = [navigationController.navigationBar
140  backgroundImageForBarMetrics:UIBarMetricsDefault];
141  _navBarLandscapePhoneImage = [navigationController.navigationBar
142  backgroundImageForBarMetrics:UIBarMetricsLandscapePhone];
143  }
144  }
145 
146  return self;
147 }
148 
149 - (void)restoreForNavigationController:(UINavigationController *)navigationController animated:(BOOL)animated {
150  [[UIApplication sharedApplication] setStatusBarStyle:self.statusBarStyle animated:animated];
151  navigationController.navigationBar.barStyle = self.navBarStyle;
152  navigationController.navigationBar.translucent = self.navBarTranslucent;
153  navigationController.navigationBar.tintColor = self.navBarTintColor;
154  if ([navigationController.navigationBar respondsToSelector:@selector(setBackgroundImage:forBarMetrics:)])
155  {
156  [navigationController.navigationBar setBackgroundImage:self.navBarDefaultImage
157  forBarMetrics:UIBarMetricsDefault];
158  [navigationController.navigationBar setBackgroundImage:self.navBarLandscapePhoneImage
159  forBarMetrics:UIBarMetricsLandscapePhone];
160  }
161 }
