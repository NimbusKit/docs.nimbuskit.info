53 CGSize NISizeOfAttributedStringConstrainedToSize(NSAttributedString* attributedString, CGSize constraintSize, NSInteger numberOfLines) {
54  if (nil == attributedString) {
55  return CGSizeZero;
56  }
57 
58  CFAttributedStringRef attributedStringRef = (__bridge CFAttributedStringRef)attributedString;
59  CTFramesetterRef framesetter = CTFramesetterCreateWithAttributedString(attributedStringRef);
60  CFRange range = CFRangeMake(0, 0);
61 
62  // This logic adapted from @mattt's TTTAttributedLabel
63  // https://github.com/mattt/TTTAttributedLabel
64 
65  if (numberOfLines == 1) {
66  constraintSize = CGSizeMake(CGFLOAT_MAX, CGFLOAT_MAX);
67 
68  } else if (numberOfLines > 0 && nil != framesetter) {
69  CGMutablePathRef path = CGPathCreateMutable();
70  CGPathAddRect(path, NULL, CGRectMake(0, 0, constraintSize.width, constraintSize.height));
71  CTFrameRef frame = CTFramesetterCreateFrame(framesetter, CFRangeMake(0, 0), path, NULL);
72  CFArrayRef lines = CTFrameGetLines(frame);
73 
74  if (nil != lines && CFArrayGetCount(lines) > 0) {
75  NSInteger lastVisibleLineIndex = MIN(numberOfLines, CFArrayGetCount(lines)) - 1;
76  CTLineRef lastVisibleLine = CFArrayGetValueAtIndex(lines, lastVisibleLineIndex);
77 
78  CFRange rangeToLayout = CTLineGetStringRange(lastVisibleLine);
79  range = CFRangeMake(0, rangeToLayout.location + rangeToLayout.length);
80  }
81 
82  CFRelease(frame);
83  CFRelease(path);
84  }
85 
86  CGSize newSize = CTFramesetterSuggestFrameSizeWithConstraints(framesetter, range, NULL, constraintSize, NULL);
87 
88  if (nil != framesetter) {
89  CFRelease(framesetter);
90  framesetter = nil;
91  }
92 
93  return CGSizeMake(NICGFloatCeil(newSize.width), NICGFloatCeil(newSize.height));
94 }

633 - (void)_processLinksInAttributedString:(NSAttributedString *)attributedString {
634  // Pull any attributes matching the link attribute from the attributed string and store them as
635  // the current set of explicit links. This properly handles the value of the attribute being
636  // either an NSURL or an NSString.
637  __block NSMutableArray *links = [NSMutableArray array];
638  [attributedString enumerateAttribute:NIAttributedLabelLinkAttributeName
639  inRange:NSMakeRange(0, attributedString.length)
640  options:0
641  usingBlock:^(id value, NSRange range, BOOL *stop) {
642  if (value != nil) {
643  [links addObject:value];
644  }
645  }];
646  self.explicitLinkLocations = links;
647 }

649 - (CGFloat)_verticalOffsetForBounds:(CGRect)bounds {
650  CGFloat verticalOffset = 0;
651  if (NIVerticalTextAlignmentTop != self.verticalTextAlignment) {
652  // When the text is attached to the top we can easily just start drawing and leave the
653  // remainder. This is the most performant case.
654  // With other alignment modes we must calculate the size of the text first.
655  CGSize textSize = [self sizeThatFits:CGSizeMake(bounds.size.width, CGFLOAT_MAX)];
656 
657  if (NIVerticalTextAlignmentMiddle == self.verticalTextAlignment) {
658  verticalOffset = floorf((bounds.size.height - textSize.height) / 2.f);
659 
660  } else if (NIVerticalTextAlignmentBottom == self.verticalTextAlignment) {
661  verticalOffset = bounds.size.height - textSize.height;
662  }
663  }
664  return verticalOffset;
665 }

672 - (NSTextCheckingResult *)linkAtPoint:(CGPoint)point {
673  if (!CGRectContainsPoint(CGRectInset(self.bounds, 0, -kVMargin), point)) {
674  return nil;
675  }
676 
677  CFArrayRef lines = CTFrameGetLines(self.textFrame);
678  if (!lines) return nil;
679  CFIndex count = CFArrayGetCount(lines);
680 
681  CGPoint origins[count];
682  CTFrameGetLineOrigins(self.textFrame, CFRangeMake(0,0), origins);
683 
684  CGAffineTransform transform = [self _transformForCoreText];
685  CGFloat verticalOffset = [self _verticalOffsetForBounds:self.bounds];
686 
687  for (int i = 0; i < count; i++) {
688  CGPoint linePoint = origins[i];
689

691  CGRect flippedRect = [self getLineBounds:line point:linePoint];

724  // intersect with the range.
