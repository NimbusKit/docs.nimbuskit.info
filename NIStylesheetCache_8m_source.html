44 - (NIStylesheet *)stylesheetWithPath:(NSString *)path loadFromDisk:(BOOL)loadFromDisk {
45  NIStylesheet* stylesheet = [_pathToStylesheet objectForKey:path];
46 
47  if (nil == stylesheet) {
48  stylesheet = [[NIStylesheet alloc] init];
49  if (loadFromDisk) {
50  BOOL didSucceed = [stylesheet loadFromPath:path
51  pathPrefix:_pathPrefix];
52 
53  if (didSucceed) {
54  [_pathToStylesheet setObject:stylesheet forKey:path];
55 
56  } else {
57  [_pathToStylesheet removeObjectForKey:path];
58  stylesheet = nil;
59  }
60 
61  } else {
62  [_pathToStylesheet setObject:stylesheet forKey:path];
63  }
64  }
65 
66  return stylesheet;
67 }
