00058     if ([host hasSuffix:@"/"]) {
00059       _host = [host copy];

00062       _host = [[host stringByAppendingString:@"/"] copy];
00063     }
00064 
00065     NSFileManager* fm = [NSFileManager defaultManager];
00066     NSDirectoryEnumerator* de = [fm enumeratorAtPath:_stylesheetCache.pathPrefix];
00067 
00068     NSString* filename;
00069     while ((filename = [de nextObject])) {
00070       BOOL isFolder = NO;
00071       NSString* path = [_stylesheetCache.pathPrefix stringByAppendingPathComponent:filename];
00072       if ([fm fileExistsAtPath:path isDirectory:&isFolder]
00073           && !isFolder
00074           && [[filename pathExtension] isEqualToString:@"css"]) {
00075         [_stylesheetPaths addObject:filename];
00076         NSString* cachePath = NIPathForDocumentsResource([self pathFromPath:filename]);
00077         NSError* error = nil;
00078         [fm removeItemAtPath:cachePath error:&error];
00079         error = nil;
00080         [fm copyItemAtPath:path toPath:cachePath error:&error];
00081         NIDASSERT(nil == error);
00082       }
00083     }
00084   }

00120 - (void)nimbusOperationWillFinish:(NINetworkRequestOperation *)operation {
00121   if (![operation.url.path isEqualToString:@"/watch"]) {
00122     NSMutableArray* changedStylesheets = [NSMutableArray array];
00123     NSArray* pathParts = [[operation.url absoluteString] pathComponents];
00124     NSString* path = [[pathParts subarrayWithRange:NSMakeRange(2, [pathParts count] - 2)]
00125                       componentsJoinedByString:@"/"];
00126     NSString* rootPath = NIPathForDocumentsResource(nil);
00127     NSString* hashedPath = [self pathFromPath:path];
00128     NSString* diskPath = [rootPath stringByAppendingPathComponent:hashedPath];
00129     [operation.data writeToFile:diskPath atomically:YES];
00130 
00131     NIStylesheet* stylesheet = [_stylesheetCache stylesheetWithPath:path loadFromDisk:NO];
00132     if ([stylesheet loadFromPath:path pathPrefix:rootPath delegate:self]) {
00133       [changedStylesheets addObject:stylesheet];
00134     }
00135     
00136     for (NSString* iteratingPath in _stylesheetPaths) {
00137       stylesheet = [_stylesheetCache stylesheetWithPath:iteratingPath loadFromDisk:NO];
00138       if ([stylesheet.dependencies containsObject:path]) {
00139         // This stylesheet has the changed stylesheet as a dependency so let's refresh it.
00140         if ([stylesheet loadFromPath:iteratingPath pathPrefix:rootPath delegate:self]) {
00141           [changedStylesheets addObject:stylesheet];
00142         }
00143       }
00144     }
00145 
00146     operation.processedObject = changedStylesheets;
00147   }
