36 - (void)drawGraphWithContext:(CGContextRef)context {
37  CGSize contentSize = self.bounds.size;
38 
39  CGFloat xRange = [self.dataSource graphViewXRange:self];
40  CGFloat yRange = [self.dataSource graphViewYRange:self];
41  if (xRange == 0 || yRange == 0) {
42  return;
43  }
44 
45  [self.dataSource resetPointIterator];
46 
47  CGContextSetLineWidth(context, 1);
48  CGContextSetShouldAntialias(context, YES);
49 
50  BOOL isFirstPoint = YES;
51  CGPoint point = CGPointZero;
52  while ([self.dataSource nextPointInGraphView:self point:&point]) {
53  CGPoint scaledPoint = CGPointMake(point.x / xRange, point.y / yRange);
54  CGPoint plotPoint = CGPointMake(floorf(scaledPoint.x * contentSize.width) - 0.5f,
55  contentSize.height
56  - floorf((scaledPoint.y * 0.8f + 0.1f)
57  * contentSize.height) - 0.5f);
58  if (!isFirstPoint) {
59  CGContextAddLineToPoint(context, plotPoint.x, plotPoint.y);
60  }
61  CGContextMoveToPoint(context, plotPoint.x, plotPoint.y);
62  isFirstPoint = NO;
63  }
64 
65  CGContextSetStrokeColorWithColor(context, [UIColor colorWithWhite:1 alpha:0.6f].CGColor);
66  CGContextStrokePath(context);
67 
68  [self.dataSource resetEventIterator];
69 
70  CGFloat xValue = 0;
71  UIColor* color = nil;
72  while ([self.dataSource nextEventInGraphView:self xValue:&xValue color:&color]) {
73  CGFloat scaledXValue = xValue / xRange;
74  CGFloat plotXValue = floorf(scaledXValue * contentSize.width) - 0.5f;
75  CGContextMoveToPoint(context, plotXValue, 0);
76  CGContextAddLineToPoint(context, plotXValue, contentSize.height);
77 
78  CGContextSetStrokeColorWithColor(context, color.CGColor);
79  CGContextStrokePath(context);
80  }
81 }
82 
83 - (void)drawRect:(CGRect)rect {
84  CGContextRef context = UIGraphicsGetCurrentContext();
85 
86  CGRect bounds = self.bounds;
87 
88  UIGraphicsPushContext(context);
89 
90  [self drawGraphWithContext:context];
91 
92  CGContextSetFillColorWithColor(context, [UIColor colorWithWhite:1 alpha:0.2f].CGColor);
93  CGContextFillRect(context, bounds);
94 
95  CGGradientRef glossGradient = nil;
96  CGColorSpaceRef colorspace = nil;
97  size_t numberOfLocations = 2;
98  CGFloat locations[2] = { 0.0f, 1.0f };
99  CGFloat components[8] = {
100  1.0f, 1.0f, 1.0f, 0.35f,
101  1.0f, 1.0f, 1.0f, 0.06f
102  };
103 
104  colorspace = CGColorSpaceCreateDeviceRGB();
105  glossGradient = CGGradientCreateWithColorComponents(colorspace,
106  components, locations, numberOfLocations);
107 
108  CGPoint topCenter = CGPointMake(CGRectGetMidX(bounds), 0.0f);
109  CGPoint midCenter = CGPointMake(CGRectGetMidX(bounds), CGRectGetMidY(bounds));
110  CGContextDrawLinearGradient(context, glossGradient, topCenter, midCenter, 0);
111 
112  CGGradientRelease(glossGradient);
113  glossGradient = nil;
114  CGColorSpaceRelease(colorspace);
115  colorspace = nil;
116 
117  UIGraphicsPopContext();
118 }
