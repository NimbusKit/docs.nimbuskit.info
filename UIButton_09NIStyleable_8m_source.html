47 - (void)applyButtonStyleWithRuleSet:(NICSSRuleset *)ruleSet inDOM:(NIDOM *)dom {
48  if ([ruleSet hasTextColor]) {
49  // If you want to reset this color, set none as the color
50  [self setTitleColor:ruleSet.textColor forState:UIControlStateNormal];
51  }
52  if ([ruleSet hasTextShadowColor]) {
53  // If you want to reset this color, set none as the color
54  [self setTitleShadowColor:ruleSet.textShadowColor forState:UIControlStateNormal];
55  }
56  if (ruleSet.hasImage) {
57  [self setImage:[UIImage imageNamed:ruleSet.image] forState:UIControlStateNormal];
58  }
59  if (ruleSet.hasBackgroundImage) {
60  UIImage *backImage = [UIImage imageNamed:ruleSet.backgroundImage];
61  if (ruleSet.hasBackgroundStretchInsets) {
62  backImage = [backImage resizableImageWithCapInsets:ruleSet.backgroundStretchInsets];
63  }
64  [self setBackgroundImage:backImage forState:UIControlStateNormal];
65  }
66  if ([ruleSet hasTextShadowOffset]) {
67  self.titleLabel.shadowOffset = ruleSet.textShadowOffset;
68  }
69  if ([ruleSet hasTitleInsets]) { self.titleEdgeInsets = ruleSet.titleInsets; }
70  if ([ruleSet hasContentInsets]) { self.contentEdgeInsets = ruleSet.contentInsets; }
71  if ([ruleSet hasImageInsets]) { self.imageEdgeInsets = ruleSet.imageInsets; }
72  if ([ruleSet hasButtonAdjust]) {
73  self.adjustsImageWhenDisabled = ((ruleSet.buttonAdjust & NICSSButtonAdjustDisabled) != 0);
74  self.adjustsImageWhenHighlighted = ((ruleSet.buttonAdjust & NICSSButtonAdjustHighlighted) != 0);
75  }
76 }

88 -(void)applyStyleWithRuleSet:(NICSSRuleset *)ruleSet forPseudoClass:(NSString *)pseudo inDOM:(NIDOM *)dom
89 {
90  UIControlState state = UIControlStateNormal;
91  if ([pseudo caseInsensitiveCompare:@"selected"] == NSOrderedSame) {
92  state = UIControlStateSelected;
93  } else if ([pseudo caseInsensitiveCompare:@"highlighted"] == NSOrderedSame) {
94  state = UIControlStateHighlighted;
95  } else if ([pseudo caseInsensitiveCompare:@"disabled"] == NSOrderedSame) {
96  state = UIControlStateDisabled;
97  }
98  if (ruleSet.hasTextKey) {
99  NIUserInterfaceString *nis = [[NIUserInterfaceString alloc] initWithKey:ruleSet.textKey];
100  [nis attach:self withSelector:@selector(setTitle:forState:) forControlState:state];
101  }
102  if (ruleSet.hasTextColor) {
103  [self setTitleColor:ruleSet.textColor forState:state];
104  }
105  if (ruleSet.hasTextShadowColor) {
106  [self setTitleShadowColor:ruleSet.textShadowColor forState:state];
107  }
108  if (ruleSet.hasImage) {
109  [self setImage:[UIImage imageNamed:ruleSet.image] forState:state];
110  }
111  if (ruleSet.hasBackgroundImage) {
112  UIImage *backImage = [UIImage imageNamed:ruleSet.backgroundImage];
113  if (ruleSet.hasBackgroundStretchInsets) {
114  backImage = [backImage resizableImageWithCapInsets:ruleSet.backgroundStretchInsets];
115  }
116  [self setBackgroundImage:backImage forState:state];
117  }
118 }
