00083 - (void)willDisplayPage:(NIPhotoScrollView *)page {
00084   // When we ask the data source for the image we expect the following to happen:
00085   // 1) If the data source has any image at this index, it should return it and set the
00086   //    photoSize accordingly.
00087   // 2) If the returned photo is not the highest quality available, the data source should
00088   //    start loading the high quality photo and set isLoading to YES.
00089   // 3) If no photo was available, then the data source should start loading the photo
00090   //    at its highest available quality and nil should be returned. The loadingImage property
00091   //    will be displayed until the image is loaded. isLoading should be set to YES.
00092   NIPhotoScrollViewPhotoSize photoSize = NIPhotoScrollViewPhotoSizeUnknown;
00093   BOOL isLoading = NO;
00094   CGSize originalPhotoDimensions = CGSizeZero;
00095   UIImage* image = [self.dataSource photoAlbumScrollView: self
00096                                             photoAtIndex: page.pageIndex
00097                                                photoSize: &photoSize
00098                                                isLoading: &isLoading
00099                                  originalPhotoDimensions: &originalPhotoDimensions];
00100 
00101   page.photoDimensions = originalPhotoDimensions;
00102 
00103   if (nil == image) {
00104     page.zoomingIsEnabled = NO;
00105     [page setImage:self.loadingImage photoSize:NIPhotoScrollViewPhotoSizeUnknown];
00106 
00107   } else {
00108     page.zoomingIsEnabled = ([self isZoomingEnabled]
00109                              && (NIPhotoScrollViewPhotoSizeOriginal == photoSize));
00110     if (photoSize > page.photoSize) {
00111       [page setImage:image photoSize:photoSize];
00112 
00113       if (NIPhotoScrollViewPhotoSizeOriginal == photoSize) {
00114         [self notifyDelegatePhotoDidLoadAtIndex:page.pageIndex];
00115       }
00116     }
00117   }
00118 }

00175 - (void)didLoadPhoto: (UIImage *)image
00176              atIndex: (NSInteger)pageIndex
00177            photoSize: (NIPhotoScrollViewPhotoSize)photoSize {
00178   for (NIPhotoScrollView* page in self.visiblePages) {
00179     if (page.pageIndex == pageIndex) {
00180 
00181       // Only replace the photo if it's of a higher quality than one we're already showing.
00182       if (photoSize > page.photoSize) {
00183         [page setImage:image photoSize:photoSize];
00184 
00185         page.zoomingIsEnabled = ([self isZoomingEnabled]
00186                                  && (NIPhotoScrollViewPhotoSizeOriginal == photoSize));
00187 
00188         // Notify the delegate that the photo has been loaded.
00189         if (NIPhotoScrollViewPhotoSizeOriginal == photoSize) {
00190           [self notifyDelegatePhotoDidLoadAtIndex:pageIndex];
00191         }
00192       }
00193       break;
00194     }
