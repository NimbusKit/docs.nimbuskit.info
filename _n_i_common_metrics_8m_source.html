00057 CGFloat NIStatusBarHeight(void) {
00058   CGRect statusBarFrame = [[UIApplication sharedApplication] statusBarFrame];
00059 
00060   // We take advantage of the fact that the status bar will always be wider than it is tall
00061   // in order to avoid having to check the status bar orientation.
00062   CGFloat statusBarHeight = MIN(statusBarFrame.size.width, statusBarFrame.size.height);
00063 
00064   return statusBarHeight;
00065 }
