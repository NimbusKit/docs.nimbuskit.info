56 - (UIView<NIRecyclableView> *)dequeueReusableViewWithIdentifier:(NSString *)reuseIdentifier {
57  NSMutableArray* views = [_reuseIdentifiersToRecycledViews objectForKey:reuseIdentifier];
58  UIView<NIRecyclableView>* view = [views lastObject];
59  if (nil != view) {
60  [views removeLastObject];
61  if ([view respondsToSelector:@selector(prepareForReuse)]) {
62  [view prepareForReuse];
63  }
64  }
65  return view;
66 }

68 - (void)recycleView:(UIView<NIRecyclableView> *)view {
69  NIDASSERT([view isKindOfClass:[UIView class]]);
70 
71  NSString* reuseIdentifier = nil;
72  if ([view respondsToSelector:@selector(reuseIdentifier)]) {
73  reuseIdentifier = [view reuseIdentifier];;
74  }
75  if (nil == reuseIdentifier) {
76  reuseIdentifier = NSStringFromClass([view class]);
77  }
78 
79  NIDASSERT(nil != reuseIdentifier);
80  if (nil == reuseIdentifier) {
81  return;
82  }
83 
84  NSMutableArray* views = [_reuseIdentifiersToRecycledViews objectForKey:reuseIdentifier];
85  if (nil == views) {
86  views = [[NSMutableArray alloc] init];
87  [_reuseIdentifiersToRecycledViews setObject:views forKey:reuseIdentifier];
88  }
89  [views addObject:view];
90 }
91 
92 - (void)removeAllViews {
93  [_reuseIdentifiersToRecycledViews removeAllObjects];
94 }
