99 - (CGFloat)pageHorizontalMargin {
100  return 10;
101 }

103 - (CGRect)frameForPagingScrollView {
104  CGRect frame = self.bounds;
105 
106  // We make the paging scroll view a little bit wider on the side edges so that there
107  // there is space between the pages when flipping through them.
108  frame.origin.x -= self.pageHorizontalMargin;
109  frame.size.width += (2 * self.pageHorizontalMargin);
110 
111  return frame;
112 }
113 
114 - (CGSize)contentSizeForPagingScrollView {
115  CGRect bounds = _pagingScrollView.bounds;
116  return CGSizeMake(bounds.size.width * [_pageViews count], bounds.size.height);
117 }
118 
119 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
120  // We have to use our paging scroll view's bounds, not frame, to calculate the page
121  // placement. When the device is in landscape orientation, the frame will still be in
122  // portrait because the pagingScrollView is the root view controller's view, so its
123  // frame is in window coordinate space, which is never rotated. Its bounds, however,
124  // will be in landscape because it has a rotation transform applied.
125  CGRect bounds = _pagingScrollView.bounds;
126  CGRect pageFrame = bounds;
127 
128  // We need to counter the extra spacing added to the paging scroll view in
129  // frameForPagingScrollView:
130  pageFrame.size.width -= self.pageHorizontalMargin * 2;
131  pageFrame.origin.x = (bounds.size.width * pageIndex) + self.pageHorizontalMargin;
132 
133  return pageFrame;
134 }
135 
136 - (void)layoutPages {
137  _pagingScrollView.contentSize = [self contentSizeForPagingScrollView];
138 
139  for (NSUInteger ix = 0; ix < [_pageViews count]; ++ix) {
140  UIView* pageView = [_pageViews objectAtIndex:ix];
141  pageView.frame = [self frameForPageAtIndex:ix];
142  }
143 }

145 - (NSInteger)visiblePageIndex {
146  CGFloat offset = _pagingScrollView.contentOffset.x;
147  CGFloat pageWidth = _pagingScrollView.bounds.size.width;
148 
149  return (NSInteger)(offset / pageWidth);
150 }
151 
152 - (void)setBounds:(CGRect)bounds {
153  NSInteger visiblePageIndex = [self visiblePageIndex];
154 
155  [super setBounds:bounds];
156 
157  [self layoutPages];
158

160  CGFloat newOffset = (visiblePageIndex * pageWidth);
161  _pagingScrollView.contentOffset = CGPointMake(newOffset, 0);

230 - (void)didPanMapWithGestureRecognizer:(UIPanGestureRecognizer *)gestureRecognizer {
231  if (!_enableDraggingVertically || _panGestureRecognizer != gestureRecognizer) {
232  return;
233  }
234 
235  if (gestureRecognizer.state == UIGestureRecognizerStateBegan) {
236  _initialFrame = self.frame;
237  } else if (gestureRecognizer.state == UIGestureRecognizerStateChanged) {
238  CGPoint translation = [gestureRecognizer translationInView:self.superview];
239  UIInterfaceOrientation orientation = NIInterfaceOrientation();
240  CGRect rect = self.frame;
241  CGRect superRect = self.superview.frame;
242  if (UIInterfaceOrientationIsPortrait(orientation)) {
243  CGFloat y = _initialFrame.origin.y + translation.y;
244  rect.origin.y = MIN(MAX(y, 0), superRect.size.height - _initialFrame.size.height);
245  } else if (UIInterfaceOrientationIsLandscape(orientation)) {
246  CGFloat x = _initialFrame.origin.x + translation.x;
247  rect.origin.x = MIN(MAX(x, 0), superRect.size.width - _initialFrame.size.width);
248  }
249  self.frame = rect;
250  }
251 }
