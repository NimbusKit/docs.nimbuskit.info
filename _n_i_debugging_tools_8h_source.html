00099 // We leave the __asm__ in this macro so that when a break occurs, we don't have to step out of
00100 // a "breakInDebugger" function.
00101 #define NIDASSERT(xx) { if (!(xx)) { NIDPRINT(@"NIDASSERT failed: %s", #xx); \
00102 if (NIDebugAssertionsShouldBreak && NIIsInDebugger()) { __asm__("int $3\n" : : ); } } \
00103 } ((void)0)
00104 #else
00105 #define NIDASSERT(xx) { if (!(xx)) { NIDPRINT(@"NIDASSERT failed: %s", #xx); \
00106 if (NIDebugAssertionsShouldBreak && NIIsInDebugger()) { raise(SIGTRAP); } } \
00107 } ((void)0)
00108 #endif // #if TARGET_IPHONE_SIMULATOR
