161 - (void)_removeNode:(NILinkedListNode *)node {
162  if (nil == node) {
163  return;
164  }
165 
166  if (nil != node.prev) {
167  node.prev.next = node.next;
168 
169  } else {
170  _head = node.next;
171  }
172 
173  if (nil != node.next) {
174  node.next.prev = node.prev;
175 
176  } else {
177  _tail = node.prev;
178  }
179 
180  node.next = nil;
181  node.prev = nil;
182 
183  --_count;
184  ++_modificationNumber;
185 }

216 - (id)initWithCoder:(NSCoder *)decoder {
217  if ((self = [super init])) {
218  // We'll let addObject modify the count, so create a local count here so that we don't
219  // double count every object.
220  NSUInteger count = 0;
221  [decoder decodeValueOfObjCType:@encode(NSUInteger) at:&count];
222 
223  for (NSUInteger ix = 0; ix < count; ++ix) {
224  id object = [decoder decodeObject];
225 
226  [self addObject:object];
227  }
228 
229  // Sanity check.
230  NIDASSERT(count == self.count);
231  }
232  return self;
233 }

235 #pragma mark - NSFastEnumeration
236 
237 - (NSUInteger)countByEnumeratingWithState:(NSFastEnumerationState *)state
238  objects:(__unsafe_unretained id *)stackbuf
239  count:(NSUInteger)len {
240  // Initialization condition.
241  if (0 == state->state) {
242  // Whenever the linked list is modified, the modification number increases. This allows
243  // enumeration to bail out if the linked list is modified mid-flight.
244  state->mutationsPtr = &_modificationNumber;
245  }
246 
247  NSUInteger numberOfItemsReturned = 0;
248 
249  // If there is no _tail (i.e. this is an empty list) then this will end immediately.
250  if ((void *)state->state != (__bridge void *)_tail) {
251  state->itemsPtr = stackbuf;
252 
253  if (0 == state->state) {
254  // Initialize the state here instead of above when we check 0 == state.state because
255  // for single item linked lists head == tail. If we initialized it in the initialization
256  // condition, state.state != _tail check would fail and we wouldn't return the single
257  // object.
258  state->state = (unsigned long)_head;
259  }
260 
261  // Return *at most* the number of request objects.
262  while ((0 != state->state) && (numberOfItemsReturned < len)) {
263  NILinkedListNode* node = (__bridge NILinkedListNode *)(void *)state->state;
264  stackbuf[numberOfItemsReturned] = node.object;
265  state->state = (unsigned long)node.next;
266  ++numberOfItemsReturned;
267  }
268 
269  if (0 == state->state) {
270  // Final step condition. We allow the above loop to overstep the end one iteration,
271  // because we rewind it one step here (to ensure that the next time enumeration occurs,
272  // state == _tail.
273  state->state = (unsigned long)_tail;
274  }
275 
276  } // else we've returned all of the items that we can; leave numberOfItemsReturned as 0 to
277  // signal that there is nothing left to be done.
278 
279  return numberOfItemsReturned;
280 }

347 - (NILinkedListLocation *)addObject:(id)object {
348  // nil objects can not be added to a linked list.
349  NIDASSERT(nil != object);
350  if (nil == object) {
351  return nil;
352  }
353 
354  NILinkedListNode* node = [[NILinkedListNode alloc] init];
355  node.object = object;
356 
357  // Empty condition.
358  if (nil == _tail) {
359  _head = node;
360  _tail = node;
361 
362  } else {
363  // Non-empty condition.
364  _tail.next = node;
365  node.prev = _tail;
366  _tail = node;
367  }
368 
369  ++self.count;
370  ++_modificationNumber;
371 
372  return [NILinkedListLocation locationWithNode:node];
373 }

387  node.prev = nil;
388  node.next = nil;

393  _tail = nil;
394 
395  self.count = 0;
