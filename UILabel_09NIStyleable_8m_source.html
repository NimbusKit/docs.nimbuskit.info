37 - (void)applyLabelStyleWithRuleSet:(NICSSRuleset *)ruleSet inDOM:(NIDOM *)dom {
38  if ([ruleSet hasTextColor]) { self.textColor = ruleSet.textColor; }
39  if ([ruleSet hasHighlightedTextColor]) { self.highlightedTextColor = ruleSet.highlightedTextColor; }
40  if ([ruleSet hasTextAlignment]) { self.textAlignment = ruleSet.textAlignment; }
41  if ([ruleSet hasFont]) { self.font = ruleSet.font; }
42  if ([ruleSet hasTextShadowColor]) { self.shadowColor = ruleSet.textShadowColor; }
43  if ([ruleSet hasTextShadowOffset]) { self.shadowOffset = ruleSet.textShadowOffset; }
44  if ([ruleSet hasLineBreakMode]) { self.lineBreakMode = ruleSet.lineBreakMode; }
45  if ([ruleSet hasNumberOfLines]) { self.numberOfLines = ruleSet.numberOfLines; }
46  if ([ruleSet hasMinimumFontSize]) { self.minimumFontSize = ruleSet.minimumFontSize; }
47  if ([ruleSet hasAdjustsFontSize]) { self.adjustsFontSizeToFitWidth = ruleSet.adjustsFontSize; }
48  if ([ruleSet hasBaselineAdjustment]) { self.baselineAdjustment = ruleSet.baselineAdjustment; }
49 }

51 -(void)applyLabelStyleBeforeViewWithRuleSet:(NICSSRuleset *)ruleSet inDOM:(NIDOM *)dom
52 {
53  if (ruleSet.hasTextKey) {
54  NIUserInterfaceString *nis = [[NIUserInterfaceString alloc] initWithKey:ruleSet.textKey];
55  [nis attach:self withSelector:@selector(setText:)];
