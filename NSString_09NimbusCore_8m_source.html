00068 - (NSDictionary*)queryContentsUsingEncoding:(NSStringEncoding)encoding {
00069   NSCharacterSet* delimiterSet = [NSCharacterSet characterSetWithCharactersInString:@"&;"];
00070   NSMutableDictionary* pairs = [NSMutableDictionary dictionary];
00071   NSScanner* scanner = [[NSScanner alloc] initWithString:self];
00072   while (![scanner isAtEnd]) {
00073     NSString* pairString = nil;
00074     [scanner scanUpToCharactersFromSet:delimiterSet intoString:&pairString];
00075     [scanner scanCharactersFromSet:delimiterSet intoString:NULL];
00076     NSArray* kvPair = [pairString componentsSeparatedByString:@"="];
00077     if (kvPair.count == 1 || kvPair.count == 2) {
00078       NSString* key = [[kvPair objectAtIndex:0]
00079                        stringByReplacingPercentEscapesUsingEncoding:encoding];
00080       NSMutableArray* values = [pairs objectForKey:key];
00081       if (nil == values) {
00082         values = [NSMutableArray array];
00083         [pairs setObject:values forKey:key];
00084       }
00085       if (kvPair.count == 1) {
00086         [values addObject:[NSNull null]];
00087 
00088       } else if (kvPair.count == 2) {
00089         NSString* value = [[kvPair objectAtIndex:1]
00090                            stringByReplacingPercentEscapesUsingEncoding:encoding];
00091         [values addObject:value];
00092       }
00093     }
00094   }
00095   return [NSDictionary dictionaryWithDictionary:pairs];
00096 }

00124 - (NSString*)stringByAddingQueryDictionary:(NSDictionary*)query {
00125   NSMutableArray* pairs = [NSMutableArray array];
00126   for (NSString* key in [query keyEnumerator]) {
00127     NSString* value = [[query objectForKey:key] stringByAddingPercentEscapesForURLParameter];
00128     NSString* pair = [NSString stringWithFormat:@"%@=%@", key, value];
00129     [pairs addObject:pair];
00130   }
00131 
00132   NSString* params = [pairs componentsJoinedByString:@"&"];
00133   if ([self rangeOfString:@"?"].location == NSNotFound) {
00134     return [self stringByAppendingFormat:@"?%@", params];
00135 
00136   } else {
00137     return [self stringByAppendingFormat:@"&%@", params];
00138   }

00179 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00180   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00181   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00182 
00183   // The parts before the "a"
00184   NSString *oneMain = [oneComponents objectAtIndex:0];
00185   NSString *twoMain = [twoComponents objectAtIndex:0];
00186 
00187   // If main parts are different, return that result, regardless of alpha part
00188   NSComparisonResult mainDiff;
00189   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00190     return mainDiff;
00191   }
00192 
00193   // At this point the main parts are the same; just deal with alpha stuff
00194   // If one has an alpha part and the other doesn't, the one without is newer
00195   if ([oneComponents count] < [twoComponents count]) {
00196     return NSOrderedDescending;
00197 
00198   } else if ([oneComponents count] > [twoComponents count]) {
00199     return NSOrderedAscending;
00200 
00201   } else if ([oneComponents count] == 1) {
00202     // Neither has an alpha part, and we know the main parts are the same
00203     return NSOrderedSame;
00204   }
00205 
00206   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00207   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00208   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00209   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00210   return [oneAlpha compare:twoAlpha];
00211 }
