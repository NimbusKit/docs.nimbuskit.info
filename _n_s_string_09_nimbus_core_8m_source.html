00081 - (NSDictionary*)queryContentsUsingEncoding:(NSStringEncoding)encoding {
00082   NSCharacterSet* delimiterSet = [NSCharacterSet characterSetWithCharactersInString:@"&;"];
00083   NSMutableDictionary* pairs = [NSMutableDictionary dictionary];
00084   NSScanner* scanner = [[[NSScanner alloc] initWithString:self] autorelease];
00085   while (![scanner isAtEnd]) {
00086     NSString* pairString = nil;
00087     [scanner scanUpToCharactersFromSet:delimiterSet intoString:&pairString];
00088     [scanner scanCharactersFromSet:delimiterSet intoString:NULL];
00089     NSArray* kvPair = [pairString componentsSeparatedByString:@"="];
00090     if (kvPair.count == 1 || kvPair.count == 2) {
00091       NSString* key = [[kvPair objectAtIndex:0]
00092                        stringByReplacingPercentEscapesUsingEncoding:encoding];
00093       NSMutableArray* values = [pairs objectForKey:key];
00094       if (nil == values) {
00095         values = [NSMutableArray array];
00096         [pairs setObject:values forKey:key];
00097       }
00098       if (kvPair.count == 1) {
00099         [values addObject:[NSNull null]];
00100 
00101       } else if (kvPair.count == 2) {
00102         NSString* value = [[kvPair objectAtIndex:1]
00103                            stringByReplacingPercentEscapesUsingEncoding:encoding];
00104         [values addObject:value];
00105       }
00106     }
00107   }
00108   return [NSDictionary dictionaryWithDictionary:pairs];

00185 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00186   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00187   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00188 
00189   // The parts before the "a"
00190   NSString *oneMain = [oneComponents objectAtIndex:0];
00191   NSString *twoMain = [twoComponents objectAtIndex:0];
00192 
00193   // If main parts are different, return that result, regardless of alpha part
00194   NSComparisonResult mainDiff;
00195   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00196     return mainDiff;
00197   }
00198 
00199   // At this point the main parts are the same; just deal with alpha stuff
00200   // If one has an alpha part and the other doesn't, the one without is newer
00201   if ([oneComponents count] < [twoComponents count]) {
00202     return NSOrderedDescending;
00203 
00204   } else if ([oneComponents count] > [twoComponents count]) {
00205     return NSOrderedAscending;
00206 
00207   } else if ([oneComponents count] == 1) {
00208     // Neither has an alpha part, and we know the main parts are the same
00209     return NSOrderedSame;
00210   }
00211 
00212   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00213   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00214   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00215   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00216   return [oneAlpha compare:twoAlpha];
00217 }
