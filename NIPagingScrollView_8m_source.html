97 - (CGRect)frameForPagingScrollView {
98  CGRect frame = self.bounds;
99 
100  if (NIPagingScrollViewHorizontal == self.type) {
101  // We make the paging scroll view a little bit wider on the side edges so that there
102  // there is space between the pages when flipping through them.
103  frame = CGRectInset(frame, -self.pageMargin, 0);
104 
105  } else if (NIPagingScrollViewVertical == self.type) {
106  frame = CGRectInset(frame, 0, -self.pageMargin);
107  }
108 
109  return frame;
110 }
111 
112 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
113  // We have to use our paging scroll view's bounds, not frame, to calculate the page
114  // placement. When the device is in landscape orientation, the frame will still be in
115  // portrait because the pagingScrollView is the root view controller's view, so its
116  // frame is in window coordinate space, which is never rotated. Its bounds, however,
117  // will be in landscape because it has a rotation transform applied.
118  CGRect bounds = _scrollView.bounds;
119  CGRect pageFrame = bounds;
120 
121  if (NIPagingScrollViewHorizontal == self.type) {
122  pageFrame.origin.x = (bounds.size.width * pageIndex);
123  // We need to counter the extra spacing added to the paging scroll view in
124  // frameForPagingScrollView.
125  pageFrame = CGRectInset(pageFrame, self.pageMargin, 0);
126 
127  } else if (NIPagingScrollViewVertical == self.type) {
128  pageFrame.origin.y = (bounds.size.height * pageIndex);
129  pageFrame = CGRectInset(pageFrame, 0, self.pageMargin);
130  }
131 
132  return pageFrame;
133 }
134 
135 - (CGSize)contentSizeForPagingScrollView {
136  // We use the paging scroll view's bounds to calculate the contentSize, for the same reason
137  // outlined above.
138  CGRect bounds = _scrollView.bounds;
139  if (NIPagingScrollViewHorizontal == self.type) {
140  return CGSizeMake(bounds.size.width * self.numberOfPages, bounds.size.height);
141 
142  } else if (NIPagingScrollViewVertical == self.type) {
143  return CGSizeMake(bounds.size.width, bounds.size.height * self.numberOfPages);
144  }
145 
146  return CGSizeZero;
147 }

195 - (BOOL)isDisplayingPageForIndex:(NSInteger)pageIndex {
196  BOOL foundPage = NO;
197 
198  // There will never be more than 3 visible pages in this array, so this lookup is
199  // effectively O(C) constant time.
200  for (UIView <NIPagingScrollViewPage>* page in _visiblePages) {
201  if (page.pageIndex == pageIndex) {
202  foundPage = YES;
203  break;
204  }
205  }
206 
207  return foundPage;
208 }
209 
210 - (NSInteger)currentVisiblePageIndex {
211  CGPoint contentOffset = _scrollView.contentOffset;
212  CGSize boundsSize = _scrollView.bounds.size;
213 
214  if (NIPagingScrollViewHorizontal == self.type) {
215  // Whatever image is currently displayed in the center of the screen is the currently
216  // visible image.
217  return NIBoundi((NSInteger)(floorf((contentOffset.x + boundsSize.width / 2) / boundsSize.width)
218  + 0.5f),
219  0, self.numberOfPages - 1);
220 
221  } else if (NIPagingScrollViewVertical == self.type) {
222  return NIBoundi((NSInteger)(floorf((contentOffset.y + boundsSize.height / 2) / boundsSize.height)
223  + 0.5f),
224  0, self.numberOfPages - 1);
225  }
226 
227  return 0;
228 }
229 
230 - (NSRange)rangeOfVisiblePages {
231  if (0 >= self.numberOfPages) {
232  return NSMakeRange(0, 0);
233  }
234 
235  NSInteger currentVisiblePageIndex = [self currentVisiblePageIndex];
236 
237  NSInteger firstVisiblePageIndex = NIBoundi(currentVisiblePageIndex - 1, 0, self.numberOfPages - 1);
238  NSInteger lastVisiblePageIndex = NIBoundi(currentVisiblePageIndex + 1, 0, self.numberOfPages - 1);
239 
240  return NSMakeRange(firstVisiblePageIndex, lastVisiblePageIndex - firstVisiblePageIndex + 1);
241 }

273 - (UIView<NIPagingScrollViewPage> *)loadPageAtIndex:(NSInteger)pageIndex {
274  UIView<NIPagingScrollViewPage>* page = [self.dataSource pagingScrollView:self pageViewForIndex:pageIndex];
275 
276  NIDASSERT([page isKindOfClass:[UIView class]]);
277  NIDASSERT([page conformsToProtocol:@protocol(NIPagingScrollViewPage)]);
278 
279  if (nil == page || ![page isKindOfClass:[UIView class]]
280  || ![page conformsToProtocol:@protocol(NIPagingScrollViewPage)]) {
281  // Bail out! This page is malformed.
282  return nil;
283  }
284 
285  return page;
286 }

288 - (void)displayPageAtIndex:(NSInteger)pageIndex {
289  UIView<NIPagingScrollViewPage>* page = [self loadPageAtIndex:pageIndex];
290  if (nil == page) {
291  return;
292  }
293 
294  // This will only be called once, before the page is shown.
295  [self willDisplayPage:page atIndex:pageIndex];
296

298  [_visiblePages addObject:page];

301 - (void)recyclePageAtIndex:(NSInteger)pageIndex {
302  for (UIView<NIPagingScrollViewPage>* page in [_visiblePages copy]) {
303  if (page.pageIndex == pageIndex) {
304  [_viewRecycler recycleView:page];
305  [page removeFromSuperview];
306 
307  [self didRecyclePage:page];
308 
309  [_visiblePages removeObject:page];
310  }
311  }

314 - (void)updateVisiblePagesShouldNotifyDelegate:(BOOL)shouldNotifyDelegate {
315  // Before updating _centerPageIndex, notify delegate
316  if (shouldNotifyDelegate && (self.numberOfPages > 0) &&
317  ([self currentVisiblePageIndex] != self.centerPageIndex) &&
318  [self.delegate respondsToSelector:@selector(pagingScrollViewWillChangePages:)]) {
319  [self.delegate pagingScrollViewWillChangePages:self];
320  }
321 
322  NSRange rangeOfVisiblePages = [self rangeOfVisiblePages];
323  // Recycle no-longer-visible pages. We copy _visiblePages because we may modify it while we're
324  // iterating over it.
325  for (UIView<NIPagingScrollViewPage>* page in [_visiblePages copy]) {
326  if (!NSLocationInRange(page.pageIndex, rangeOfVisiblePages)) {
327  [_viewRecycler recycleView:page];
328  [page removeFromSuperview];
329 
330  [self didRecyclePage:page];
331 
332  [_visiblePages removeObject:page];
333  }
334  }
335 
336  NSInteger oldCenterPageIndex = self.centerPageIndex;
337 
338  if (self.numberOfPages > 0) {
339  _centerPageIndex = [self currentVisiblePageIndex];
340 
341  [self didChangeCenterPageIndexFrom:oldCenterPageIndex to:_centerPageIndex];
342 
343  // Prioritize displaying the currently visible page.
344  if (![self isDisplayingPageForIndex:_centerPageIndex]) {
345  [self displayPageAtIndex:_centerPageIndex];
346  }
347 
348  // Add missing pages.
349  for (int pageIndex = rangeOfVisiblePages.location;
350  pageIndex < (NSInteger)NSMaxRange(rangeOfVisiblePages); ++pageIndex) {
351  if (![self isDisplayingPageForIndex:pageIndex]) {
352  [self displayPageAtIndex:pageIndex];
353  }
354  }
355  } else {
356  _centerPageIndex = -1;
357  }
358 
359  if (shouldNotifyDelegate && oldCenterPageIndex != _centerPageIndex
360  && [self.delegate respondsToSelector:@selector(pagingScrollViewDidChangePages:)]) {
361  [self.delegate pagingScrollViewDidChangePages:self];
362  }
363 }

365 - (void)layoutVisiblePages {
366  for (UIView<NIPagingScrollViewPage>* page in _visiblePages) {
367  CGRect pageFrame = [self frameForPageAtIndex:page.pageIndex];
368  if ([page respondsToSelector:@selector(setFrameAndMaintainState:)]) {
369  [page setFrameAndMaintainState:pageFrame];
370 
371  } else {
372  [page setFrame:pageFrame];
373  }
374  }

400 - (void)scrollViewDidScroll:(UIScrollView *)scrollView {
401  if ([scrollView isTracking] && [scrollView isDragging]) {
402  if ([self.delegate respondsToSelector:@selector(pagingScrollViewDidScroll:)]) {
403  [self.delegate pagingScrollViewDidScroll:self];
404  }
405  }
406  if (_shouldUpdateVisiblePagesWhileScrolling
407  && ![scrollView isTracking] && ![scrollView isDragging]) {
408  [self updateVisiblePagesShouldNotifyDelegate:YES];
409  }
410 
411  if ([self.delegate respondsToSelector:_cmd]) {
412  [self.delegate scrollViewDidScroll:scrollView];
413  }
414 
415  if (_isKillingAnimation) {
416  // The content size is calculated based on the number of pages and the scroll view frame.
417  CGPoint offset = [self frameForPageAtIndex:_centerPageIndex].origin;
418  offset = [self contentOffsetFromPageOffset:offset];
419  _scrollView.contentOffset = offset;
420  }
421 }
422 
423 - (void)scrollViewDidEndDragging:(UIScrollView *)scrollView willDecelerate:(BOOL)decelerate {
424  _isKillingAnimation = NO;
425 
426  if (!decelerate) {
427  [self updateVisiblePagesShouldNotifyDelegate:YES];
428  [self resetSurroundingPages];
429  }
430 
431  if ([self.delegate respondsToSelector:_cmd]) {
432  [self.delegate scrollViewDidEndDragging:scrollView willDecelerate:decelerate];
433  }
434 }

439 
440  if ([self.delegate respondsToSelector:_cmd]) {
441  [self.delegate scrollViewDidEndDecelerating:scrollView];
442  }

456 - (BOOL)shouldForwardSelectorToDelegate:(SEL)aSelector {
457  struct objc_method_description description;
458  // Only forward the selector if it's part of the UIScrollViewDelegate protocol.
459  description = protocol_getMethodDescription(@protocol(UIScrollViewDelegate),
460  aSelector,
461  NO,
462  YES);
463 
464  BOOL isSelectorInScrollViewDelegate = (description.name != NULL && description.types != NULL);
465  return (isSelectorInScrollViewDelegate
466  && [self.delegate respondsToSelector:aSelector]);
467 }

513 - (void)reloadData {
514  _animatingToPageIndex = -1;
515  NIDASSERT(nil != _dataSource);
516 
517  // Remove any visible pages from the view before we release the sets.
518  for (UIView<NIPagingScrollViewPage>* page in _visiblePages) {
519  [_viewRecycler recycleView:page];
520  [(UIView *)page removeFromSuperview];
521 
522  [self didRecyclePage:page];
523  }
524 
525  _visiblePages = nil;
526 
527  // If there is no data source then we can't do anything particularly interesting.
528  if (nil == _dataSource) {
529  _scrollView.contentSize = self.bounds.size;
530  _scrollView.contentOffset = CGPointZero;
531 
532  // May as well just get rid of all the views then.
533  [_viewRecycler removeAllViews];
534 
535  return;
536  }
537 
538  _visiblePages = [[NSMutableSet alloc] init];
539 
540  // Cache the number of pages.
541  _numberOfPages = [_dataSource numberOfPagesInPagingScrollView:self];
542  _scrollView.frame = [self frameForPagingScrollView];
543  _scrollView.contentSize = [self contentSizeForPagingScrollView];
544 
545  [self didReloadNumberOfPages];
546 
547  NSInteger oldCenterPageIndex = _centerPageIndex;
548  if (oldCenterPageIndex >= 0) {
549  _centerPageIndex = NIBoundi(_centerPageIndex, 0, self.numberOfPages - 1);
550 
551  if (![_scrollView isTracking] && ![_scrollView isDragging]) {
552  // The content size is calculated based on the number of pages and the scroll view frame.
553  CGPoint offset = [self frameForPageAtIndex:_centerPageIndex].origin;
554  offset = [self contentOffsetFromPageOffset:offset];
555  _scrollView.contentOffset = offset;
556 
557  _isKillingAnimation = YES;
558  }
559  }
560 
561  // Begin requesting the page information from the data source.
562  [self updateVisiblePagesShouldNotifyDelegate:NO];
563 }

565 - (void)willRotateToInterfaceOrientation: (UIInterfaceOrientation)toInterfaceOrientation
566  duration: (NSTimeInterval)duration {
567  // Here, our pagingScrollView bounds have not yet been updated for the new interface
568  // orientation. This is a good place to calculate the content offset that we will
569  // need in the new orientation.
570  CGFloat offset = [self scrolledPageOffset];
571  CGFloat pageScrollableDimension = [self pageScrollableDimension];
572 
573  if (offset >= 0) {
574  _firstVisiblePageIndexBeforeRotation = (NSInteger)floorf(offset / pageScrollableDimension);
575  _percentScrolledIntoFirstVisiblePage = ((offset
576  - (_firstVisiblePageIndexBeforeRotation * pageScrollableDimension))
577  / pageScrollableDimension);
578 
579  } else {
580  _firstVisiblePageIndexBeforeRotation = 0;
581  _percentScrolledIntoFirstVisiblePage = offset / pageScrollableDimension;
582  }
583 }
584 
585 - (void)willAnimateRotationToInterfaceOrientation: (UIInterfaceOrientation)toInterfaceOrientation
586  duration: (NSTimeInterval)duration {
587  // Recalculate contentSize based on current orientation.
588  _scrollView.contentSize = [self contentSizeForPagingScrollView];
589 
590  [self layoutVisiblePages];
591 
592  // Adjust contentOffset to preserve page location based on values collected prior to location.
593  CGFloat pageScrollableDimension = [self pageScrollableDimension];
594  CGFloat newOffset = ((_firstVisiblePageIndexBeforeRotation * pageScrollableDimension)
595  + (_percentScrolledIntoFirstVisiblePage * pageScrollableDimension));
596  _scrollView.contentOffset = [self contentOffsetFromOffset:newOffset];
597 }

607 - (void)didAnimateToPage:(NSInteger)pageIndex {
608  _shouldUpdateVisiblePagesWhileScrolling = NO;
609  _animatingToPageIndex = -1;
610  if (_queuedAnimationPageIndex >= 0 && _queuedAnimationPageIndex != pageIndex) {
611  [self moveToPageAtIndex:_queuedAnimationPageIndex animated:YES];
612  return;
613  }
614 
615  // Reset the content offset once the animation completes, just to be sure that the
616  // viewer sits on a page bounds even if we rotate the device while animating.
617  CGPoint offset = [self frameForPageAtIndex:pageIndex].origin;
618  offset = [self contentOffsetFromPageOffset:offset];
619 
620  _scrollView.contentOffset = offset;
621 
622  [self updateVisiblePagesShouldNotifyDelegate:YES];
623 }

625 - (BOOL)moveToPageAtIndex:(NSInteger)pageIndex animated:(BOOL)animated {
626  return [self moveToPageAtIndex:pageIndex animated:animated updateVisiblePagesWhileScrolling:NO];
627 }
628 
629 - (BOOL)moveToPageAtIndex:(NSInteger)pageIndex animated:(BOOL)animated updateVisiblePagesWhileScrolling:(BOOL)updateVisiblePagesWhileScrolling {
630  if (_animatingToPageIndex >= 0) {
631  // Don't allow re-entry for sliding animations.
632  _queuedAnimationPageIndex = pageIndex;
633  return NO;
634  }
635  _shouldUpdateVisiblePagesWhileScrolling = updateVisiblePagesWhileScrolling;
636  _isKillingAnimation = NO;
637  _queuedAnimationPageIndex = -1;
638 
639  CGPoint offset = [self frameForPageAtIndex:pageIndex].origin;
640  offset = [self contentOffsetFromPageOffset:offset];
641 
642  // The paging scroll view won't actually animate if the offsets are identical.
643  animated = animated && !CGPointEqualToPoint(offset, _scrollView.contentOffset);
644 
645  if (animated) {
646  _animatingToPageIndex = pageIndex;
647  }
648  [_scrollView setContentOffset:offset animated:animated];
649  if (!animated) {
650  [self resetSurroundingPages];
651  [self didAnimateToPage:pageIndex];
652  }
653  return YES;
654 }
