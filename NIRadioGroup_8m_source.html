81 - (BOOL)shouldForwardSelector:(SEL)selector {
82  struct objc_method_description description;
83  description = protocol_getMethodDescription(@protocol(UITableViewDelegate), selector, NO, YES);
84  return (description.name != NULL && description.types != NULL);

87 - (BOOL)respondsToSelector:(SEL)selector {
88  if ([super respondsToSelector:selector]) {
89  return YES;
90 
91  } else if ([self shouldForwardSelector:selector]) {
92  for (id delegate in self.forwardDelegates) {
93  if ([delegate respondsToSelector:selector]) {
94  return YES;
95  }
96  }
97  }
98  return NO;
99 }

101 - (NSMethodSignature *)methodSignatureForSelector:(SEL)selector {
102  NSMethodSignature *signature = [super methodSignatureForSelector:selector];
103  if (signature == nil) {
104  for (id delegate in self.forwardDelegates) {
105  if ([delegate respondsToSelector:selector]) {
106  signature = [delegate methodSignatureForSelector:selector];
107  }
108  }
109  }
110  return signature;
111 }

113 - (void)forwardInvocation:(NSInvocation *)invocation {
114  BOOL didForward = NO;
115 
116  if ([self shouldForwardSelector:invocation.selector]) {
117  for (id delegate in self.forwardDelegates) {
118  if ([delegate respondsToSelector:invocation.selector]) {
119  [invocation invokeWithTarget:delegate];
120  didForward = YES;
121  break;
122  }
123  }
124  }
125 
126  if (!didForward) {
127  [super forwardInvocation:invocation];
128  }
129 }

143 - (id)mapObject:(id)object toIdentifier:(NSInteger)identifier {
144  NIDASSERT(nil != object);
145  NIDASSERT(identifier != kInvalidSelection);
146  NIDASSERT(![self isObjectInRadioGroup:object]);
147  if (nil == object) {
148  return nil;
149  }
150  if (kInvalidSelection == identifier) {
151  return nil;
152  }
153  if ([self isObjectInRadioGroup:object]) {
154  return nil;
155  }
156  [self.objectMap setObject:object forKey:[self keyForIdentifier:identifier]];
157  [self.objectSet addObject:object];
158  [self.objectOrder addObject:object];
159  return object;
160 }
161 
162 - (void)setSelectedIdentifier:(NSInteger)selectedIdentifier {
163  id key = [self keyForIdentifier:selectedIdentifier];
164  NIDASSERT(nil != [self.objectMap objectForKey:key]);
165  if (nil != [self.objectMap objectForKey:key]) {
166  _selectedIdentifier = selectedIdentifier;
167  self.hasSelection = YES;
168  } else {
169  // If we set an invalid identifier then clear the current selection.
170  self.hasSelection = NO;
171  }
172 }
173

175  return self.hasSelection ? _selectedIdentifier : kInvalidSelection;
176 }

182 - (BOOL)isObjectInRadioGroup:(id)object {
183  if (nil == object) {
184  return NO;
185  }
186  return [self.objectSet containsObject:object];
187 }
188 
189 - (BOOL)isObjectSelected:(id)object {
190  if (nil == object) {
191  return NO;
192  }
193  NIDASSERT(nil != object);
194  NIDASSERT([self isObjectInRadioGroup:object]);
195  NSArray* keys = [self.objectMap allKeysForObject:object];
196  NSInteger selectedIdentifier = self.selectedIdentifier;
197  for (NSNumber* key in keys) {
198  if ([key intValue] == selectedIdentifier) {
199  return YES;
200  }
201  }
202  return NO;
203 }
204 
205 - (NSInteger)identifierForObject:(id)object {
206  if (nil == object) {
207  return NO;
208  }
209  NIDASSERT(nil != object);
210  NIDASSERT([self isObjectInRadioGroup:object]);
211  NSArray* keys = [self.objectMap allKeysForObject:object];
212  return keys.count > 0 ? [[keys objectAtIndex:0] intValue] : kInvalidSelection;
213 }

222 - (void)tableView:(UITableView *)tableView willDisplayCell:(UITableViewCell *)cell forRowAtIndexPath:(NSIndexPath *)indexPath {
223  NIDASSERT([tableView.dataSource isKindOfClass:[NITableViewModel class]]);
224  if ([tableView.dataSource isKindOfClass:[NITableViewModel class]]) {
225  NITableViewModel* model = (NITableViewModel *)tableView.dataSource;
226  id object = [model objectAtIndexPath:indexPath];
227  if ([self isObjectInRadioGroup:object]) {
228  cell.accessoryType = ([self isObjectSelected:object]
229  ? UITableViewCellAccessoryCheckmark
230  : UITableViewCellAccessoryNone);
231  cell.selectionStyle = self.tableViewCellSelectionStyle;
232  }
233  }
234 
235  // Forward the invocation along.
236  for (id<UITableViewDelegate> delegate in self.forwardDelegates) {
237  if ([delegate respondsToSelector:_cmd]) {
238  [delegate tableView:tableView willDisplayCell:cell forRowAtIndexPath:indexPath];
239  }
240  }
241 }

243 - (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath {
244  NIDASSERT([tableView.dataSource isKindOfClass:[NITableViewModel class]]);
245  if ([tableView.dataSource isKindOfClass:[NITableViewModel class]]) {
246  NITableViewModel* model = (NITableViewModel *)tableView.dataSource;
247  id object = [model objectAtIndexPath:indexPath];
248 
249  if (object == self) {
250  // You must provide a controller in the initWithController: initializer.
251  NIDASSERT(nil != self.controller);
252 
253  NIRadioGroupController* controller = [[NIRadioGroupController alloc] initWithRadioGroup:self tappedCell:(id<NICell>)[tableView cellForRowAtIndexPath:indexPath]];
254  controller.title = self.controllerTitle;
255 
256  BOOL shouldPush = YES;
257  // Notify the delegate that the controller is about to appear.
258  if ([self.delegate respondsToSelector:@selector(radioGroup:radioGroupController:willAppear:)]) {
259  shouldPush = [self.delegate radioGroup:self radioGroupController:controller willAppear:YES];
260  }
261 
262  if (shouldPush) {
263  [self.controller.navigationController pushViewController:controller animated:YES];
264  }
265 
266  } else if ([self isObjectInRadioGroup:object]) {
267  NSInteger newSelection = [self identifierForObject:object];
268 
269  if (newSelection != self.selectedIdentifier) {
270  [self setSelectedIdentifier:newSelection];
271 
272  // It's easiest to simply reload the visible table cells. Reloading only the radio group
273  // cells would require iterating through the visible cell objects and determining whether
274  // each was in the radio group. This is more complex behavior that should be relegated to
275  // the controller.
276  [tableView reloadRowsAtIndexPaths:tableView.indexPathsForVisibleRows
277  withRowAnimation:UITableViewRowAnimationNone];
278 
279  // After we reload the table view the selection will be lost, so set the selection again.
280  [tableView selectRowAtIndexPath:indexPath
281  animated:NO
282  scrollPosition:UITableViewScrollPositionNone];
283 
284  [self.delegate radioGroup:self didSelectIdentifier:newSelection];
285  }
286 
287  // Fade the selection out.
288  [tableView deselectRowAtIndexPath:indexPath animated:YES];
289  }
290  }
291 
292  // Forward the invocation along.
293  for (id<UITableViewDelegate> delegate in self.forwardDelegates) {
294  if ([delegate respondsToSelector:_cmd]) {
295  [delegate tableView:tableView didSelectRowAtIndexPath:indexPath];
296  }
297  }
298 }
