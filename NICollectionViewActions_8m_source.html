34 - (BOOL)collectionView:(UICollectionView *)collectionView shouldHighlightItemAtIndexPath:(NSIndexPath *)indexPath {
35  BOOL shouldHighlight = NO;
36 
37  NIDASSERT([collectionView.dataSource isKindOfClass:[NICollectionViewModel class]]);
38  if ([collectionView.dataSource isKindOfClass:[NICollectionViewModel class]]) {
39  NICollectionViewModel* model = (NICollectionViewModel *)collectionView.dataSource;
40  id object = [model objectAtIndexPath:indexPath];
41 
42  if ([self isObjectActionable:object]) {
43  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
44 
45  // If the cell is tappable, reflect that in the selection style.
46  if (nil != action.tapAction || nil != action.tapSelector
47  || nil != action.detailAction || nil != action.detailSelector
48  || nil != action.navigateAction || nil != action.navigateSelector) {
49  shouldHighlight = YES;
50  }
51  }
52  }
53 
54  return shouldHighlight;
55 }

57 - (void)collectionView:(UICollectionView *)collectionView didSelectItemAtIndexPath:(NSIndexPath *)indexPath {
58  NIDASSERT([collectionView.dataSource isKindOfClass:[NICollectionViewModel class]]);
59  if ([collectionView.dataSource isKindOfClass:[NICollectionViewModel class]]) {
60  NICollectionViewModel* model = (NICollectionViewModel *)collectionView.dataSource;
61  id object = [model objectAtIndexPath:indexPath];
62 
63  if ([self isObjectActionable:object]) {
64  NIObjectActions* action = [self actionForObjectOrClassOfObject:object];
65 
66  BOOL shouldDeselect = NO;
67  if (action.tapAction) {
68  // Tap actions can deselect the cell if they return YES.
69  shouldDeselect = action.tapAction(object, self.target, indexPath);
70  }
71  if (action.tapSelector && [self.target respondsToSelector:action.tapSelector]) {
72  NSMethodSignature *methodSignature = [self.target methodSignatureForSelector:action.tapSelector];
73  NSInvocation *invocation = [NSInvocation invocationWithMethodSignature:methodSignature];
74  invocation.selector = action.tapSelector;
75  if (methodSignature.numberOfArguments >= 3) {
76  [invocation setArgument:&object atIndex:2];
77  }
78  if (methodSignature.numberOfArguments >= 4) {
79  [invocation setArgument:&indexPath atIndex:3];
80  }
81  [invocation invokeWithTarget:self.target];
82 
83  NSUInteger length = invocation.methodSignature.methodReturnLength;
84  if (length > 0) {
85  char *buffer = (void *)malloc(length);
86  memset(buffer, 0, sizeof(char) * length);
87  [invocation getReturnValue:buffer];
88  for (NSUInteger index = 0; index < length; ++index) {
89  if (buffer[index]) {
90  shouldDeselect = YES;
91  break;
92  }
93  }
94  free(buffer);
95  }
96  }
97 
98  if (action.detailAction) {
99  // Tap actions can deselect the cell if they return YES.
100  action.detailAction(object, self.target, indexPath);
101  }
102  if (action.detailSelector && [self.target respondsToSelector:action.detailSelector]) {
103 #pragma clang diagnostic push
104 #pragma clang diagnostic ignored "-Warc-performSelector-leaks"
105  [self.target performSelector:action.detailSelector withObject:object withObject:indexPath];
106 #pragma clang diagnostic pop
107  }
108 
109  if (action.navigateAction) {
110  // Tap actions can deselect the cell if they return YES.
111  action.navigateAction(object, self.target, indexPath);
112  }
113  if (action.navigateSelector && [self.target respondsToSelector:action.navigateSelector]) {
114 #pragma clang diagnostic push
115 #pragma clang diagnostic ignored "-Warc-performSelector-leaks"
116  [self.target performSelector:action.navigateSelector withObject:object withObject:indexPath];
117 #pragma clang diagnostic pop
118  }
119 
120  if (shouldDeselect) {
121  [collectionView deselectItemAtIndexPath:indexPath animated:YES];
122  }
123  }
124  }
125 }
