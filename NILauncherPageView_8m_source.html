41 - (void)layoutSubviews {
42  [super layoutSubviews];
43 
44  const CGFloat leftEdge = self.contentInset.left;
45  const CGFloat topEdge = self.contentInset.top;
46  const CGFloat rightEdge = self.bounds.size.width - self.contentInset.right;
47  const CGSize viewSize = self.viewSize;
48  const CGSize viewMargins = self.viewMargins;
49 
50  CGFloat contentWidth = (self.bounds.size.width - self.contentInset.left - self.contentInset.right);
51  NSInteger numberOfColumns = floorf((contentWidth + viewMargins.width) / (viewSize.width + viewMargins.width));
52  CGFloat viewWidth = numberOfColumns * viewSize.width;
53  CGFloat distributedHorizontalMargin = floorf((contentWidth - viewWidth) / (CGFloat)(numberOfColumns + 1));
54 
55  const CGFloat horizontalDelta = viewSize.width + distributedHorizontalMargin;
56  const CGFloat verticalDelta = viewSize.height + viewMargins.height;
57 
58  CGFloat x = leftEdge + distributedHorizontalMargin;
59  CGFloat y = topEdge;
60 
61  for (UIView* view in self.mutableRecyclableViews) {
62  view.frame = CGRectMake(x, y, viewSize.width, viewSize.height);
63  x += horizontalDelta;
64  if (x + viewSize.width > rightEdge) {
65  x = leftEdge + distributedHorizontalMargin;
66  y += verticalDelta;
67  }
68  }
69 }

74 - (void)prepareForReuse {
75  // You forgot to provide a view recycler.
76  NIDASSERT(nil != self.viewRecycler);
77 
78  for (UIView<NIRecyclableView>* view in self.mutableRecyclableViews) {
79  [view removeFromSuperview];
80  [self.viewRecycler recycleView:view];
81  }
82  [self.mutableRecyclableViews removeAllObjects];
83 }
