50 - (void)_configureDefaults {
51  // We handle autoresizing ourselves.
52  [self setAutoresizesSubviews:NO];
53 
54  _viewRecycler = [[NIViewRecycler alloc] init];
55 
56  _buttonSize = CGSizeMake(kDefaultButtonDimensions, kDefaultButtonDimensions);
57  _numberOfColumns = NILauncherViewGridBasedOnButtonSize;
58  _numberOfRows = NILauncherViewGridBasedOnButtonSize;
59 
60  _maxNumberOfButtonsPerPage = NSIntegerMax;
61  _contentInsetForPages = UIEdgeInsetsMake(kDefaultPadding, kDefaultPadding, kDefaultPadding, kDefaultPadding);
62 
63  // The paging scroll view.
64  _pagingScrollView = [[NIPagingScrollView alloc] initWithFrame:self.bounds];
65  _pagingScrollView.dataSource = self;
66  _pagingScrollView.delegate = self;
67 
68  [self addSubview:_pagingScrollView];
69 
70  // The pager displayed below the paging scroll view.
71  _pager = [[UIPageControl alloc] init];
72  _pager.hidesForSinglePage = YES;
73 
74  // So, this is weird. Apparently if you don't set a background color on the pager control
75  // then taps won't be handled anywhere but within the dot area. If you do set a background
76  // color, however, then taps outside of the dot area DO change the selected page.
77  // \(o.o)/
78  _pager.backgroundColor = [UIColor blackColor];
79 
80  // Similarly for the scroll view anywhere there isn't a subview.
81  // We update these background colors when the launcher view's own background color is set.
82  _pagingScrollView.backgroundColor = [UIColor blackColor];
83 
84  // Don't update the pager when the user taps until we've animated to the new page.
85  // This allows us to reset the page index forcefully if necessary without flickering the
86  // pager's current selection.
87  _pager.defersCurrentPageDisplay = YES;
88 
89  // When the user taps the pager control it fires a UIControlEventValueChanged notification.
90  [_pager addTarget:self action:@selector(pagerDidChangePage:) forControlEvents:UIControlEventValueChanged];
91 
92  [self addSubview:_pager];
93 }

128 - (void)calculateLayoutForFrame:(CGRect)frame
129  buttonDimensions:(CGSize *)pButtonDimensions
130  numberOfRows:(NSInteger *)pNumberOfRows
131  numberOfColumns:(NSInteger *)pNumberOfColumns
132  buttonMargins:(CGSize *)pButtonMargins {
133  NIDASSERT(nil != pButtonDimensions);
134  NIDASSERT(nil != pNumberOfRows);
135  NIDASSERT(nil != pNumberOfColumns);
136  NIDASSERT(nil != pButtonMargins);
137  if (nil == pButtonDimensions
138  || nil == pNumberOfRows
139  || nil == pNumberOfColumns
140  || nil == pButtonMargins) {
141  return;
142  }
143  CGFloat pageWidth = frame.size.width - self.contentInsetForPages.left - self.contentInsetForPages.right;
144  CGFloat pageHeight = frame.size.height - self.contentInsetForPages.top - self.contentInsetForPages.bottom;
145 
146  CGSize buttonDimensions = self.buttonSize;
147  NSInteger numberOfColumns = self.numberOfColumns;
148  NSInteger numberOfRows = self.numberOfRows;
149 
150  // Override point
151  if ([self.dataSource respondsToSelector:@selector(numberOfRowsPerPageInLauncherView:)]) {
152  numberOfRows = [self.dataSource numberOfRowsPerPageInLauncherView:self];
153  }
154  if ([self.dataSource respondsToSelector:@selector(numberOfColumnsPerPageInLauncherView:)]) {
155  numberOfColumns = [self.dataSource numberOfColumnsPerPageInLauncherView:self];
156  }
157 
158  if (NILauncherViewGridBasedOnButtonSize == numberOfColumns) {
159  numberOfColumns = floorf(pageWidth / buttonDimensions.width);
160  }
161  if (NILauncherViewGridBasedOnButtonSize == numberOfRows) {
162  numberOfRows = floorf(pageHeight / buttonDimensions.height);
163  }
164  NIDASSERT(numberOfRows > 0);
165  NIDASSERT(numberOfColumns > 0);
166  numberOfRows = MAX(1, numberOfRows);
167  numberOfColumns = MAX(1, numberOfColumns);
168 
169  CGFloat totalButtonWidth = numberOfColumns * buttonDimensions.width;
170  CGFloat buttonHorizontalSpacing = 0;
171  if (numberOfColumns > 1) {
172  buttonHorizontalSpacing = floorf((pageWidth - totalButtonWidth) / (numberOfColumns - 1));
173  }
174  CGFloat totalButtonHeight = numberOfRows * buttonDimensions.height;
175  CGFloat buttonVerticalSpacing = 0;
176  if (numberOfRows > 1) {
177  buttonVerticalSpacing = floorf((pageHeight - totalButtonHeight) / (numberOfRows - 1));
178  }
179 
180  *pButtonDimensions = buttonDimensions;
181  *pNumberOfRows = numberOfRows;
182  *pNumberOfColumns = numberOfColumns;
183  pButtonMargins->width = buttonHorizontalSpacing;
184  pButtonMargins->height = buttonVerticalSpacing;
185 }

192  [self calculateLayoutForFrame:self.pagingScrollView.frame

225 - (BOOL)pageAndIndexOfButton:(UIButton *)searchButton page:(NSInteger *)pPage index:(NSInteger *)pIndex {
226  NIDASSERT(nil != pPage);
227  NIDASSERT(nil != pIndex);
228  if (nil == pPage
229  || nil == pIndex) {
230  return NO;
231  }
232 
233  for (NILauncherPageView* pageView in self.pagingScrollView.visiblePages) {
234  for (NSInteger buttonIndex = 0; buttonIndex < pageView.recyclableViews.count; ++buttonIndex) {
235  UIView<NILauncherButtonView>* buttonView = [pageView.recyclableViews objectAtIndex:buttonIndex];
236  if (buttonView.button == searchButton) {
237  *pPage = pageView.pageIndex;
238  *pIndex = buttonIndex;
239  return YES;
240  }
241  }
242  }
243 
244  return NO;
245 }

247 - (void)didTapButton:(UIButton *)tappedButton {
248  NSInteger page = -1;
249  NSInteger buttonIndex = 0;
250  if ([self pageAndIndexOfButton:tappedButton
251  page:&page
252  index:&buttonIndex]) {
253 
254  if ([self.delegate respondsToSelector:@selector(launcherView:didSelectItemOnPage:atIndex:)]) {
255  [self.delegate launcherView:self didSelectItemOnPage:page atIndex:buttonIndex];
256  }
257 
258  } else {
259  // How exactly did we tap a button that wasn't a part of the launcher view?
260  NIDASSERT(NO);
261  }
262 }

271 - (UIView<NIPagingScrollViewPage> *)pagingScrollView:(NIPagingScrollView *)pagingScrollView pageViewForIndex:(NSInteger)pageIndex {
272  NILauncherPageView* page = (NILauncherPageView *)[self.pagingScrollView dequeueReusablePageWithIdentifier:kPageReuseIdentifier];
273  if (nil == page) {
274  page = [[NILauncherPageView alloc] initWithReuseIdentifier:kPageReuseIdentifier];
275  page.viewRecycler = self.viewRecycler;
276  }
277 
278  [self updateLayoutForPage:page];
279 
280  NSInteger numberOfButtons = [self.dataSource launcherView:self numberOfButtonsInPage:pageIndex];
281  numberOfButtons = MIN(numberOfButtons, self.maxNumberOfButtonsPerPage);
282 
283  for (NSInteger buttonIndex = 0 ; buttonIndex < numberOfButtons; ++buttonIndex) {
284  UIView<NILauncherButtonView>* buttonView = [self.dataSource launcherView:self buttonViewForPage:pageIndex atIndex:buttonIndex];
285  NSAssert(nil != buttonView, @"A non-nil UIView must be returned.");
286  [buttonView.button addTarget:self action:@selector(didTapButton:) forControlEvents:UIControlEventTouchUpInside];
287  [page addRecyclableView:(UIView<NIRecyclableView> *)buttonView];
288  }
289 
290  return page;
291 }
