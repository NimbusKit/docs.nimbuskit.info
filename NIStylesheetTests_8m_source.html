53 - (void)assertColor:(UIColor *)color1 equalsColor:(UIColor *)color2 {
54  size_t nColors1 = CGColorGetNumberOfComponents(color1.CGColor);
55  size_t nColors2 = CGColorGetNumberOfComponents(color2.CGColor);
56  STAssertEquals(nColors1, nColors2, @"Should have the same number of components.");
57 
58  const float* colors1 = CGColorGetComponents(color1.CGColor);
59  const float* colors2 = CGColorGetComponents(color2.CGColor);
60  for (NSInteger ix = 0; ix < nColors1; ++ix) {
61  STAssertEqualsWithAccuracy(colors1[ix], colors2[ix], 0.0001, @"Colors should match.");
62  }
63 }

65 - (void)testApplyStyleToUILabel {
66  // Sadly nearly all of these tests don't work with SenTest. The error we get when we run these
67  // tests is:
68  // ERROR: System image table has not been initialized. Do not ask for images or set up UI before UIApplicationMain() has been called.
69  return;
70 
71  NIStylesheet* stylesheet = [[NIStylesheet alloc] init];
72  NSString* pathToFile = NIPathForBundleResource(_unitTestBundle, @"UILabel.css");
73 
74  STAssertTrue([stylesheet loadFromPath:pathToFile], @"The stylesheet should have been parsed.");
75 
76  UILabel* label = [[UILabel alloc] initWithFrame:CGRectZero];
77  [stylesheet applyStyleToView:label withClassName:NSStringFromClass([label class]) inDOM:nil];
78 
79  [self assertColor:label.textColor equalsColor:[UIColor redColor]];
80  [self assertColor:label.shadowColor equalsColor:[UIColor greenColor]];
81  STAssertEquals(label.textAlignment, UITextAlignmentRight, @"Alignment should match.");
82  STAssertEquals(label.shadowOffset.width, 20.f, @"Shadow offset should match.");
83  STAssertEquals(label.shadowOffset.height, -30.f, @"Shadow offset should match.");
84  STAssertEquals(label.lineBreakMode, NSLineBreakByTruncatingTail, @"Should match.");
85  STAssertEquals(label.numberOfLines, 5, @"Should match.");
86  STAssertEquals(label.minimumFontSize, 5.f, @"Should match.");
87  STAssertTrue(label.adjustsFontSizeToFitWidth, @"Should match.");
88  STAssertEquals(label.baselineAdjustment, UIBaselineAdjustmentAlignCenters, @"Should match.");
89  STAssertEquals(label.alpha, 0.5f, @"Should match.");
90 }
