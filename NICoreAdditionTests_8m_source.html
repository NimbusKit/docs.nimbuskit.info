35 - (void)testNSString_queryContentsUsingEncoding {
36  NSDictionary* query;
37 
38  query = NIQueryDictionaryFromStringUsingEncoding(@"", NSUTF8StringEncoding);
39  STAssertTrue([query count] == 0, @"Query: %@", query);
40 
41  query = NIQueryDictionaryFromStringUsingEncoding(@"q", NSUTF8StringEncoding);
42  STAssertTrue([query[@"q"] isEqual:@[[NSNull null]]],
43  @"Query: %@", query);
44 
45  query = NIQueryDictionaryFromStringUsingEncoding(@"q=", NSUTF8StringEncoding);
46  STAssertTrue([query[@"q"] isEqual:@[@""]],
47  @"Query: %@", query);
48 
49  query = NIQueryDictionaryFromStringUsingEncoding(@"q=three20", NSUTF8StringEncoding);
50  STAssertTrue([query[@"q"] isEqual:@[@"three20"]],
51  @"Query: %@", query);
52 
53  query = NIQueryDictionaryFromStringUsingEncoding(@"q=three20%20github", NSUTF8StringEncoding);
54  STAssertTrue([query[@"q"] isEqual:@[@"three20 github"]],
55  @"Query: %@", query);
56 
57  query = NIQueryDictionaryFromStringUsingEncoding(@"q=three20&hl=en", NSUTF8StringEncoding);
58  STAssertTrue([query[@"q"] isEqual:@[@"three20"]],
59  @"Query: %@", query);
60  STAssertTrue([query[@"hl"] isEqual:@[@"en"]],
61  @"Query: %@", query);
62 
63  query = NIQueryDictionaryFromStringUsingEncoding(@"q=three20&hl=", NSUTF8StringEncoding);
64  STAssertTrue([query[@"q"] isEqual:@[@"three20"]],
65  @"Query: %@", query);
66  STAssertTrue([query[@"hl"] isEqual:@[@""]],
67  @"Query: %@", query);
68 
69  query = NIQueryDictionaryFromStringUsingEncoding(@"q=&&hl=", NSUTF8StringEncoding);
70  STAssertTrue([query[@"q"] isEqual:@[@""]],
71  @"Query: %@", query);
72  STAssertTrue([query[@"hl"] isEqual:@[@""]],
73  @"Query: %@", query);
74 
75  query = NIQueryDictionaryFromStringUsingEncoding(@"q=three20=repo&hl=en", NSUTF8StringEncoding);
76  STAssertNil(query[@"q"], @"Query: %@", query);
77  STAssertTrue([query[@"hl"] isEqual:@[@"en"]],
78  @"Query: %@", query);
79 
80  query = NIQueryDictionaryFromStringUsingEncoding(@"&&", NSUTF8StringEncoding);
81  STAssertTrue([query count] == 0, @"Query: %@", query);
82 
83  query = NIQueryDictionaryFromStringUsingEncoding(@"q=foo&q=three20", NSUTF8StringEncoding);
84  NSArray* qArr = @[@"foo", @"three20"];
85  STAssertTrue([query[@"q"] isEqual:qArr], @"Query: %@", query);
86 
87  query = NIQueryDictionaryFromStringUsingEncoding(@"q=foo&q=three20&hl=en", NSUTF8StringEncoding);
88  qArr = @[@"foo", @"three20"];
89  STAssertTrue([query[@"q"] isEqual:qArr], @"Query: %@", query);
90  STAssertTrue([query[@"hl"] isEqual:@[@"en"]],
91  @"Query: %@", query);
92 
93  query = NIQueryDictionaryFromStringUsingEncoding(@"q=foo&q=three20&hl=en&g", NSUTF8StringEncoding);
94  qArr = @[@"foo", @"three20"];
95  STAssertTrue([query[@"q"] isEqual:qArr], @"Query: %@", query);
96  STAssertTrue([query[@"hl"] isEqual:@[@"en"]],
97  @"Query: %@", query);
98  STAssertTrue([query[@"g"] isEqual:@[[NSNull null]]],
99  @"Query: %@", query);
100 
101  query = NIQueryDictionaryFromStringUsingEncoding(@"q&q=three20&hl=en&g", NSUTF8StringEncoding);
102  qArr = @[[NSNull null], @"three20"];
103  STAssertTrue([query[@"q"] isEqual:qArr], @"Query: %@", query);
104  STAssertTrue([query[@"hl"] isEqual:@[@"en"]],
105  @"Query: %@", query);
106 }

108 - (void)testNSString_stringByAddingQueryDictionary {
109  NSString* baseUrl = @"http://google.com/search";
110  STAssertTrue([NIStringByAddingQueryDictionaryToString(baseUrl, nil) isEqualToString:
111  [baseUrl stringByAppendingString:@"?"]], @"Empty dictionary fail.");
112 
113  STAssertTrue([NIStringByAddingQueryDictionaryToString(baseUrl, @{}) isEqualToString:
114  [baseUrl stringByAppendingString:@"?"]], @"Empty dictionary fail.");
115 
116  STAssertTrue([NIStringByAddingQueryDictionaryToString(baseUrl, @{@"q":@"three20"}) isEqualToString:
117  [baseUrl stringByAppendingString:@"?q=three20"]], @"Single parameter fail.");
118 
119  NSDictionary* query = @{@"q": @"three20",
120  @"hl": @"en"};
121  NSString* baseUrlWithQuery = NIStringByAddingQueryDictionaryToString(baseUrl, query);
122  STAssertTrue([baseUrlWithQuery isEqualToString:[baseUrl
123  stringByAppendingString:@"?hl=en&q=three20"]]
124  || [baseUrlWithQuery isEqualToString:[baseUrl
125  stringByAppendingString:@"?q=three20&hl=en"]],
126  @"Additional query parameters not correct. %@",
127  NIStringByAddingQueryDictionaryToString(baseUrl, query));
128 }

130 - (void)testNSString_versionStringCompare {
131  STAssertTrue(NICompareVersionStrings(@"3.0", @"3.0") == NSOrderedSame, @"same version");
132  STAssertTrue(NICompareVersionStrings(@"3.0a2", @"3.0a2") == NSOrderedSame, @"same version alpha");
133  STAssertTrue(NICompareVersionStrings(@"3.0", @"2.5") == NSOrderedDescending, @"major no alpha");
134  STAssertTrue(NICompareVersionStrings(@"3.1", @"3.0") == NSOrderedDescending, @"minor no alpha");
135  STAssertTrue(NICompareVersionStrings(@"3.0a1", @"3.0") == NSOrderedAscending, @"alpha-no alpha");
136  STAssertTrue(NICompareVersionStrings(@"3.0a1", @"3.0a4") == NSOrderedAscending, @"alpha diff");
137  STAssertTrue(NICompareVersionStrings(@"3.0a2", @"3.0a19") == NSOrderedAscending, @"numeric alpha");
138  STAssertTrue(NICompareVersionStrings(@"3.0a", @"3.0a1") == NSOrderedAscending, @"empty alpha");
139  STAssertTrue(NICompareVersionStrings(@"3.02", @"3.03") == NSOrderedAscending, @"point diff");
140  STAssertTrue(NICompareVersionStrings(@"3.0.2", @"3.0.3") == NSOrderedAscending, @"point diff");
141 }
