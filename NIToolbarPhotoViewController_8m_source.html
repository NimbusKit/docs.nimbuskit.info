175 - (void)loadView {
176  [super loadView];
177 
178  self.view.backgroundColor = [UIColor blackColor];
179 
180  CGRect bounds = self.view.bounds;
181 
182  // Toolbar Setup
183 
184  CGFloat toolbarHeight = NIToolbarHeightForOrientation(NIInterfaceOrientation());
185  CGRect toolbarFrame = CGRectMake(0, bounds.size.height - toolbarHeight,
186  bounds.size.width, toolbarHeight);
187 
188  _toolbar = [[UIToolbar alloc] initWithFrame:toolbarFrame];
189  _toolbar.barStyle = UIBarStyleBlack;
190  _toolbar.translucent = self.toolbarIsTranslucent;
191  _toolbar.autoresizingMask = (UIViewAutoresizingFlexibleWidth
192  | UIViewAutoresizingFlexibleTopMargin);
193 
194  [self updateToolbarItems];
195 
196  // Photo Album View Setup
197 
198  CGRect photoAlbumFrame = bounds;
199  if (!self.toolbarIsTranslucent) {
200  photoAlbumFrame = NIRectContract(bounds, 0, toolbarHeight);
201  }
202  _photoAlbumView = [[NIPhotoAlbumScrollView alloc] initWithFrame:photoAlbumFrame];
203  _photoAlbumView.autoresizingMask = (UIViewAutoresizingFlexibleWidth
204  | UIViewAutoresizingFlexibleHeight);
205  _photoAlbumView.delegate = self;
206 
207  [self.view addSubview:_photoAlbumView];
208  [self.view addSubview:_toolbar];
209 
210  if (self.hidesChromeWhenScrolling || self.chromeCanBeHidden) {
211  [self addTapGestureToView];
212  }

293 - (void)setChromeVisibility:(BOOL)isVisible animated:(BOOL)animated {
294  if (_isAnimatingChrome
295  || (!isVisible && _isChromeHidden)
296  || (isVisible && !_isChromeHidden)
297  || !self.chromeCanBeHidden) {
298  // Nothing to do here.
299  return;
300  }
301 
302  CGRect toolbarFrame = self.toolbar.frame;
303  CGRect bounds = self.view.bounds;
304 
305  if (self.toolbarIsTranslucent) {
306  // Reset the toolbar's initial position.
307  if (!isVisible) {
308  toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
309 
310  } else {
311  // Ensure that the toolbar is visible through the animation.
312  self.toolbar.hidden = NO;
313 
314  toolbarFrame.origin.y = bounds.size.height;
315  }
316  self.toolbar.frame = toolbarFrame;
317  }
318 
319  // Show/hide the system chrome.
320  BOOL isStatusBarAppearanceSupported = [self respondsToSelector:@selector(setNeedsStatusBarAppearanceUpdate)];
321  if (!isStatusBarAppearanceSupported) {
322  [[UIApplication sharedApplication] setStatusBarHidden:!isVisible
323  withAnimation:(animated
324  ? UIStatusBarAnimationSlide
325  : UIStatusBarAnimationNone)];
326  }
327 
328  if (self.toolbarIsTranslucent) {
329  // Place the toolbar at its final location.
330  if (isVisible) {
331  // Slide up.
332  toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
333 
334  } else {
335  // Slide down.
336  toolbarFrame.origin.y = bounds.size.height;
337  }
338  }
339 
340  // If there is a navigation bar, place it at its final location.
341  CGRect navigationBarFrame = self.navigationController.navigationBar.frame;
342 
343  if (animated) {
344  [UIView beginAnimations:nil context:nil];
345  [UIView setAnimationDelegate:self];
346  [UIView setAnimationDidStopSelector:(isVisible
347  ? @selector(didShowChrome)
348  : @selector(didHideChrome))];
349 
350  // Ensure that the animation matches the status bar's.
351  [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
352  [UIView setAnimationCurve:NIStatusBarAnimationCurve()];
353  }
354 
355  if (isStatusBarAppearanceSupported) {
356  _prefersStatusBarHidden = !isVisible;
357  [self setNeedsStatusBarAppearanceUpdate];
358  }
359 
360  if (nil != self.navigationController.navigationBar) {
361  if (isVisible) {
362  [UIView setAnimationsEnabled:NO];
363  [self.navigationController setNavigationBarHidden:NO animated:NO];
364  navigationBarFrame.origin.y = 0;
365  self.navigationController.navigationBar.frame = navigationBarFrame;
366  self.navigationController.navigationBar.alpha = 0;
367  [UIView setAnimationsEnabled:YES];
368 
369  navigationBarFrame.origin.y = NIStatusBarHeight();
370 
371  } else {
372  navigationBarFrame.origin.y = 0;
373  }
374  }
375 
376  if (self.toolbarIsTranslucent) {
377  self.toolbar.frame = toolbarFrame;
378  }
379  if (nil != self.navigationController.navigationBar) {
380  self.navigationController.navigationBar.frame = navigationBarFrame;
381  self.navigationController.navigationBar.alpha = (isVisible ? 1 : 0);
382  }
383 
384  if (animated) {
385  _isAnimatingChrome = YES;
386  [UIView commitAnimations];
387 
388  } else if (!isVisible) {
389  [self didHideChrome];
390 
391  } else if (isVisible) {
392  [self didShowChrome];
393  }
394 }

403 - (void)didTap {
404  SEL selector = @selector(toggleChromeVisibility);
405  if (self.photoAlbumView.zoomingIsEnabled) {
406  // Cancel any previous delayed performs so that we don't stack them.
407  [NSObject cancelPreviousPerformRequestsWithTarget:self selector:selector object:nil];
408 
409  // We need to delay taking action on the first tap in case a second tap comes in, causing
410  // a double-tap gesture to be recognized and the photo to be zoomed.
411  [self performSelector: selector
412  withObject: nil
413  afterDelay: 0.3];
414 
415  } else {
416  // When zooming is disabled, double-tap-to-zoom is also disabled so we don't have to
417  // be as careful; just toggle the chrome immediately.
418  [self toggleChromeVisibility];
419  }
420 }
