00312       NSInteger newSelection = [self identifierForObject:object];

00315         [self setSelectedIdentifier:newSelection];

00318         // cells would require iterating through the visible cell objects and determining whether
00319         // each was in the radio group. This is more complex behavior that should be relegated to
00320         // the controller.
00321         [tableView reloadRowsAtIndexPaths:tableView.indexPathsForVisibleRows
00322                          withRowAnimation:UITableViewRowAnimationNone];

00324         // After we reload the table view the selection will be lost, so set the selection again.
00325         [tableView selectRowAtIndexPath:indexPath
00326                                animated:NO

00329         [self.delegate radioGroup:self didSelectIdentifier:newSelection];
00330       }
