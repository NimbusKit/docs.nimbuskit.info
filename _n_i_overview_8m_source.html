00073     formatter = [[NSDateFormatter alloc] init];

00075     [formatter setDateStyle:kCFDateFormatterMediumStyle];

00077 
00078   // Don't autorelease here in an attempt to minimize autorelease thrashing in tight
00079   // loops.

00082                                                            encoding: NSUTF8StringEncoding];

00085                                       initWithLog:formattedLogMessage];

00089   NI_RELEASE_SAFELY(entry);
00090

00092                          @"%@: %s\n", [formatter stringFromDate:[NSDate date]], message];

00115 + (void)didChangeOrientation {
00116   static UIDeviceOrientation lastOrientation = UIDeviceOrientationUnknown;
00117 
00118   // Don't animate the overview if the device didn't actually change orientations.
00119   if (lastOrientation != [[UIDevice currentDevice] orientation]
00120       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationUnknown
00121       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceUp
00122       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceDown) {
00123 
00124     // When we flip from landscape to landscape or portait to portait the animation lasts
00125     // twice as long.
00126     UIDeviceOrientation currentOrientation = [[UIDevice currentDevice] orientation];
00127     BOOL isLongAnimation = (UIDeviceOrientationIsLandscape(lastOrientation)
00128                             == UIDeviceOrientationIsLandscape(currentOrientation));
00129     lastOrientation = currentOrientation;
00130 
00131     // Hide the overview right away, we'll make it fade back in when the rotation is
00132     // finished.
00133     sOverviewView.hidden = YES;
00134 
00135     // Delay showing the overview again until the rotation finishes.
00136     [self cancelPreviousPerformRequestsWithTarget:self];
00137     [self performSelector:@selector(showoverviewAfterRotation) withObject:nil
00138                afterDelay:NIDeviceRotationDuration(isLongAnimation)];
00139   }

00311 + (CGRect)frame {
00312 #ifdef DEBUG
00313   UIInterfaceOrientation orient = NIInterfaceOrientation();
00314   CGFloat overviewWidth;
00315   CGRect frame;
00316 
00317   // We can't take advantage of automatic view positioning because the overview exists
00318   // at the topmost view level (even above the root view controller). As such, we have to
00319   // calculate the frame depending on the interface orientation.
00320   if (orient == UIInterfaceOrientationLandscapeLeft) {
00321     overviewWidth = [UIScreen mainScreen].bounds.size.height;
00322     frame = CGRectMake(NIOverviewStatusBarHeight(), 0, sOverviewHeight, overviewWidth);
00323 
00324   } else if (orient == UIInterfaceOrientationLandscapeRight) {
00325     overviewWidth = [UIScreen mainScreen].bounds.size.height;
00326     frame = CGRectMake([UIScreen mainScreen].bounds.size.width
00327                        - (NIOverviewStatusBarHeight() + sOverviewHeight), 0,
00328                        sOverviewHeight, overviewWidth);
00329 
00330   } else if (orient == UIInterfaceOrientationPortraitUpsideDown) {
00331     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00332     frame = CGRectMake(0, [UIScreen mainScreen].bounds.size.height
00333                        - (NIOverviewStatusBarHeight() + sOverviewHeight),
00334                        overviewWidth, sOverviewHeight);
00335 
00336   } else if (orient == UIInterfaceOrientationPortrait) {
00337     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00338     frame = CGRectMake(0, NIOverviewStatusBarHeight(), overviewWidth, sOverviewHeight);
00339 
00340   } else {
00341     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00342     frame = CGRectMake(0, NIOverviewStatusBarHeight(), overviewWidth, sOverviewHeight);
00343   }
00344 
00345   if ([[UIApplication sharedApplication] isStatusBarHidden]) {
00346     // When the status bar is hidden we want to position the overview offscreen.
00347     switch (orient) {
00348       case UIInterfaceOrientationLandscapeLeft: {
00349         frame = CGRectOffset(frame, -frame.size.width, 0);
00350         break;
00351       }
00352       case UIInterfaceOrientationLandscapeRight: {
00353         frame = CGRectOffset(frame, frame.size.width, 0);
00354         break;
00355       }
00356       case UIInterfaceOrientationPortrait: {
00357         frame = CGRectOffset(frame, 0, -frame.size.height);
00358         break;
00359       }
00360       case UIInterfaceOrientationPortraitUpsideDown: {
00361         frame = CGRectOffset(frame, 0, frame.size.height);
00362         break;
00363       }
00364     }
00365   }
00366   return frame;
00367
